### Priority Queue
```java
class Solution {
    ArrayList<Integer> sortBySetBitCount(int[] arr) {
        PriorityQueue<Integer> pq = new PriorityQueue<>(new Comparator<Integer>() {
            @Override
            public int compare(Integer idx1, Integer idx2) {
                int bits1 = Integer.bitCount(arr[idx1]);
                int bits2 = Integer.bitCount(arr[idx2]);

                if (bits1 != bits2) return Integer.compare(bits2, bits1);
                return Integer.compare(idx1, idx2);
            }
        });
        for (int i = 0; i < arr.length; i++) pq.offer(i);
        int idx = -1;
        ArrayList<Integer> res = new ArrayList<>();
        while (!pq.isEmpty()) res.add(arr[pq.poll()]);
        return res;
    }
}
```

### Bucket Sort
```java
class Solution {
    ArrayList<Integer> sortBySetBitCount(int[] arr) {
        List<List<Integer>> buckets = new ArrayList<>(32);
        for (int i = 0; i < 32; i++) {
            buckets.add(new ArrayList<>());
        }

        for (int num : arr) {
            int count = Integer.bitCount(num);
            buckets.get(count).add(num);
        }

        ArrayList<Integer> result = new ArrayList<>(arr.length);
        for (int i = 31; i >= 0; i--) {
            result.addAll(buckets.get(i));
        }

        return result;
    }
}
```
