class Solution {
    public int findMinDiff(ArrayList<Integer> arr, int m) {
        Collections.sort(arr);
        int first = 0, second = m-1;
        int ans = Integer.MAX_VALUE;
        while(second<arr.size()) {
            ans = Math.min(ans, arr.get(second)-arr.get(first));
            second++;
            first++;
        }
        return ans;
    }
}
