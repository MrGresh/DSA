class Pairs {
    int ele;
    int originalIdx;
    Pairs(int e, int i) {
        this.ele = e;
        this.originalIdx = i;
    }
}
class Solution {
    int[] indexes;
    public List<Integer> countSmaller(int[] arr) {
        indexes = new int[arr.length];
        Pairs[] newArr = new Pairs[arr.length];
        int ind = -1;
        for(int i=0;i<arr.length;i++) {
            newArr[++ind] = new Pairs(arr[i], i);
        }
        mergeSort(newArr, 0, arr.length-1);
        List<Integer> res = new ArrayList<>();
        for(int x : indexes) res.add(x);
        return res;
    }
    void mergeSort(Pairs[] arr, int l, int r) {
        if(l==r) return;
        int m = l + (r-l)/2;
        mergeSort(arr, l, m);
        mergeSort(arr, m+1, r);
        merge(arr, l, m, r);
    }
    void merge(Pairs[] arr, int l, int m, int r) {
        Pairs[] sortedArray = new Pairs[r-l+1];
        int k = 0;
        int i = l, j = m+1;
        for (int left = l; left <= m; left++) {
            int low = m + 1;
            int high = r;
            int idx = -1;

            while (low <= high) {
                int mid = low + (high - low) / 2;
                if (arr[left].ele > arr[mid].ele) {
                    idx = mid;
                    low = mid + 1;
                } else {
                    high = mid - 1;
                }
            }
            if(idx!=-1) indexes[arr[left].originalIdx] += (idx-m);
        }
        while(i<=m && j<=r) {
            if(arr[i].ele <= arr[j].ele) sortedArray[k++] = arr[i++];
            else sortedArray[k++] = arr[j++];
        }
        while(i<=m) sortedArray[k++] = arr[i++];
        while(j<=r) sortedArray[k++] = arr[j++];
        
        for(int a=0;a<sortedArray.length;a++) arr[a+l] = sortedArray[a];
    }
}
