class Solution {
    public int findMaxElements(int[] arr, int k) {
        Arrays.sort(arr);
        
        int start = 0;
        int maxCount = 0;
        
        for (int end = 0; end < arr.length; end++) {
            while (arr[end] - arr[start] > k) {
                start++;
            }
            
            maxCount = Math.max(maxCount, end - start + 1);
        }
        
        return maxCount;
    }
}
