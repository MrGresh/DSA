class Solution {
    public int findMaxElements(int[] arr, int k) {
        Arrays.sort(arr);
        
        int left = 0;
        int maxCount = 0;
        
        for (int right = 0; right < arr.length; right++) {
            while (arr[right] - arr[left] > k) {
                left++;
            }
            
            maxCount = Math.max(maxCount, right - left + 1);
        }
        
        return maxCount;
    }
}
