class Solution {
    public boolean areConsecutives(int[] arr) {
        int n = arr.length;
        if (n <= 1) return true;

        Arrays.sort(arr);

        for (int i = 1; i < n; i++) {
            if (arr[i] != arr[i - 1] + 1) {
                return false;
            }
        }

        return true;
    }
}
