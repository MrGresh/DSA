public class codefile{
    static int count(int[]  arr){
        HashMap<Integer, Integer> mp = new HashMap<Integer, Integer>();
        int ans = 0;
        for (int x : arr) 
        {
            mp.put(x, mp.getOrDefault(x, 0) + 1);
            ans = Math.max(ans, mp.get(x));
        }
        return ans;
    }
}
