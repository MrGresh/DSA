class Solution {
    void mergeSort(int arr[], int l, int r) {
        if(l<r) {
            int m = l + (r-l)/2;
            mergeSort(arr, l, m);
            mergeSort(arr, m+1, r);
            merge(arr, l, m, r);
        }
    }
    void merge(int arr[], int l, int m, int r) {
        int[] newArr = new int[r-l+1];
        int i = l, j = m+1, k = -1;
        while(i<=m && j<=r) {
            if(arr[i]<=arr[j]) newArr[++k] = arr[i++];
            else newArr[++k] = arr[j++];
        }
        while(i<=m) newArr[++k] = arr[i++];
        while(j<=r) newArr[++k] = arr[j++];
        
        for(int a=0;a<newArr.length;a++) arr[a+l] = newArr[a];
    }
}
