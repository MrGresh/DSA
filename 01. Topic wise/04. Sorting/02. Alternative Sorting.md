public class codefile{
    static int[] sort(int[]  arr){
        Arrays.sort(arr);
        int[] res = new int[arr.length];
        int i = 0, j = arr.length-1, k = 0;
        while(i<j) {
            res[k++] = arr[j--];
            res[k++] = arr[i++];
        }
        if(i==j) res[k] = arr[i];
        return res;
    }
}
