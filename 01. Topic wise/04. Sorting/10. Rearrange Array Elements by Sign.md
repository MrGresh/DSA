class Solution {
    public int[] rearrangeArray(int[] arr) {
        int pos = 0, neg = 0;
        for(int n : arr) {
            if(n>=0) pos++;
            else neg++;
        }
        if(neg==0 || pos==0) return arr;
        int[] res = new int[arr.length];
        if(pos==neg) {
            int negIdx = 1;
            int posIdx = 0;
            for(int n : arr) {
                if(n>=0) {
                    res[posIdx] = n;
                    posIdx+=2;
                }
                else {
                    res[negIdx] = n;
                    negIdx+=2;
                }
            }
            return res;
        }
        else if(pos>neg) {
            int diff = pos-neg;
            int negIdx = 1;
            int posIdx = 0;
            for(int n : arr) {
                if(n>=0) {
                    res[posIdx] = n;
                    if((res.length-1-posIdx)<=diff) posIdx++;
                    else posIdx+=2;
                }
                else {
                    res[negIdx] = n;
                    negIdx+=2;
                }
            }
            return res;
        }
        else {
            int diff = neg-pos;
            int negIdx = 0;
            int posIdx = 1;
            for(int n : arr) {
                if(n>=0) {
                    res[posIdx] = n;
                    posIdx+=2;
                }
                else {
                    res[negIdx] = n;
                    if((res.length-1-negIdx)<=diff) negIdx++;
                    else negIdx+=2;
                }
            }
            return res;
        }        
    }
}
