class Solution {
    int maxOnes(int[] arr) {
        int ones = 0, maxGain = Integer.MIN_VALUE, currGain = 0;
        for(int x: arr) {
            if(x==1) {
                ones++;
                currGain--;
                if(currGain<0) currGain=0;
            } else {
                currGain++;
                maxGain = Math.max(maxGain, currGain);
            }
        }
        if(maxGain==Integer.MIN_VALUE) maxGain = 0;
        return ones + maxGain;
    }
};
