class Solution {
    public int findContentChildren(int[] childs, int[] cookies) {
        if(cookies.length==0) return 0;
        Arrays.sort(childs);
        Arrays.sort(cookies);
        int i=0;
        for(int c: cookies) {
            if(c>=childs[i]) {
                i++;
                if(i==childs.length) break;
            }
        }
        return i;
    }
}
