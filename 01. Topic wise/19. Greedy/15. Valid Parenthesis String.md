### Greedy
```java
class Solution {
    public boolean checkValidString(String s) {
        // range
        int minOpen = 0;
        int maxOpen = 0;

        for (char c : s.toCharArray()) {
            if (c == '(') {
                minOpen++;
                maxOpen++;
            } else if (c == ')') {
                minOpen--;
                maxOpen--;
            } else { // c == '*'
                minOpen--; // Treat as ')'
                maxOpen++; // Treat as '('
            }

            if (minOpen < 0) minOpen = 0;

            if (maxOpen < 0) return false;
        }

        return minOpen == 0;
    }
}
```

### Memoization
```java
class Solution {
    private Boolean[][] memo;
    public boolean checkValidString(String s) {
        memo = new Boolean[s.length()+1][101];
        return backtrack(s, 0, 0);
    }

    private boolean backtrack(String s, int index, int balance) {
        if (index == s.length()) return balance == 0;
        if (memo[index][balance]!=null) return memo[index][balance];

        char c = s.charAt(index);
        boolean isValid = false;

        if (c == '(') {
            isValid = backtrack(s, index + 1, balance + 1);
        } else if (c == ')') {
            isValid = balance>0 ? backtrack(s, index + 1, balance - 1) : false;
        } else {
            isValid = backtrack(s, index + 1, balance + 1) ||
                (balance>0 ? backtrack(s, index + 1, balance - 1) : false) ||
                backtrack(s, index + 1, balance);
        }
        return memo[index][balance] = isValid;
    }
}
```

### Tabulation
```java
class Solution {
    public boolean checkValidString(String s) {
        boolean[][] dp = new boolean[s.length()+1][101];
        for(int index=s.length();index>=0;index--) {
            for(int balance=0;balance<100;balance++) {
                if (index == s.length()) dp[index][balance] = (balance == 0);
                else {
                    char c = s.charAt(index);
                    boolean isValid = false;

                    if (c == '(') {
                        isValid = dp[index + 1][balance + 1];
                    } else if (c == ')') {
                        isValid = balance>0 ? dp[index + 1][balance - 1] : false;
                    } else {
                        isValid = dp[index + 1][balance + 1] ||
                            (balance>0 ? dp[index + 1][balance - 1] : false) ||
                            dp[index + 1][balance];
                    }
                    dp[index][balance] = isValid;
                }
            }
        }
        return dp[0][0];
    }
}
```

### Space Optimization
```java
class Solution {
    public boolean checkValidString(String s) {
        boolean[] prev = new boolean[101];
        for(int index=s.length();index>=0;index--) {
            boolean[] curr = new boolean[101];
            for(int balance=0;balance<100;balance++) {
                if (index == s.length()) curr[balance] = (balance == 0);
                else {
                    char c = s.charAt(index);
                    boolean isValid = false;

                    if (c == '(') {
                        isValid = prev[balance + 1];
                    } else if (c == ')') {
                        isValid = balance>0 ? prev[balance - 1] : false;
                    } else {
                        isValid = prev[balance + 1] ||
                            (balance>0 ? prev[balance - 1] : false) ||
                            prev[balance];
                    }
                    curr[balance] = isValid;
                }
            }
            prev = curr;
        }
        return prev[0];
    }
}
```
