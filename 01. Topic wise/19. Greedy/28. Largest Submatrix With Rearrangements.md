class Solution {
    public int largestSubmatrix(int[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        int maxArea = 0;

        // Step 1: Update the matrix to store the height of consecutive 1s
        for (int i = 1; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (matrix[i][j] == 1) {
                    matrix[i][j] += matrix[i - 1][j];
                }
            }
        }

        // Step 2: For each row, sort heights to find the max rectangle
        for (int i = 0; i < m; i++) {
            // We copy the row to avoid destroying the DP state for the next row
            int[] currentRow = matrix[i].clone();
            Arrays.sort(currentRow);

            // Step 3: Iterate backwards (largest heights first)
            for (int k = 1; k <= n; k++) {
                // currentRow[n - k] is the k-th largest height
                int height = currentRow[n - k];
                int width = k;
                maxArea = Math.max(maxArea, height * width);
            }
        }

        return maxArea;
    }
}
