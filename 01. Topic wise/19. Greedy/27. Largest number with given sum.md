class Solution {
    static String largestNumber(int n, int sum) {
        StringBuilder res = new StringBuilder();
        while(sum>0 && res.toString().length()<n) {
            if(sum>9) {
                res.append("9");
                sum -= 9;
            } else {
                res.append(String.valueOf(sum));
                sum = 0;
            }
        }
        while(res.toString().length()<n) res.append("0");
        if(res.length()==0 || sum>0) return "-1";
        return res.toString();
    }
}
