class Solution {
    public String largestSwap(String s) {
        char[] digits = s.toCharArray();
        int[] lastIndex = new int[10];
        
        // Record the last occurrence of each digit 0-9
        for (int i = 0; i < digits.length; i++) {
            lastIndex[digits[i] - '0'] = i;
        }
        
        // Traverse the string to find the first improvable digit
        for (int i = 0; i < digits.length; i++) {
            int currentDigit = digits[i] - '0';
            
            // Check if a larger digit exists later in the string
            for (int d = 9; d > currentDigit; d--) {
                if (lastIndex[d] > i) {
                    // Perform the swap
                    char temp = digits[i];
                    digits[i] = digits[lastIndex[d]];
                    digits[lastIndex[d]] = temp;
                    
                    return new String(digits);
                }
            }
        }
        
        return s; // No swap could make it larger
    }
}
