class Solution {
    public boolean divideArray(int[] arr) {
        Map<Integer, Integer> m = new HashMap<>();
        for(int x : arr) {
            m.put(x, m.getOrDefault(x, 0)+1);
        }
        for(int x : m.values()) {
            if(x%2!=0) return false;
        }
        return true;
    }
}

class Solution {
    public boolean divideArray(int[] arr) {
        Arrays.sort(arr);
        for (int i = 0; i < arr.length; i += 2) {
            if (arr[i] != arr[i + 1]) return false;
        }
        return true;
    }
}
