class Solution {
    public int maxLen(int[] nums) {
        for(int i=0;i<nums.length;i++) {
            if(nums[i]==0) nums[i] = -1;
        }
        Map<Integer, Integer> m = new HashMap<>();
        int res = 0;
        for(int i=0;i<nums.length;i++) {
            if(i>0) nums[i] += nums[i-1];
            if(nums[i]==0) res = i+1;
            if(m.containsKey(nums[i])) res = Math.max(res, i-m.get(nums[i]));
            else m.put(nums[i], i);
        }
        return res;
    }
}
