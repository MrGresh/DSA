class TopVotedCandidate {
    private TreeMap<Integer, Integer> timeline = new TreeMap<>();

    public TopVotedCandidate(int[] persons, int[] times) {
        int n = persons.length;
        int[] count = new int[n]; // Assuming person IDs are 0 to n-1
        int leader = -1;
        int maxVotes = -1;

        for (int i = 0; i < n; i++) {
            int p = persons[i];
            int t = times[i];
            
            count[p]++;
            
            if (count[p] >= maxVotes) {
                maxVotes = count[p];
                leader = p;
            }
            
            timeline.put(t, leader);
        }
    }

    public int q(int t) {
        // Find the largest time <= t
        return timeline.floorEntry(t).getValue();
    }
}
