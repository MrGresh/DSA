class Solution {
    public static String TieBreak(String arr[]) {
        Map<String, Integer> voteCount = new HashMap<>();
        for (String name : arr) {
            voteCount.put(name, voteCount.getOrDefault(name, 0) + 1);
        }

        String winner = "";
        int maxVotes = -1;

        for (Map.Entry<String, Integer> entry : voteCount.entrySet()) {
            String candidate = entry.getKey();
            int votes = entry.getValue();

            if (votes > maxVotes) {
                maxVotes = votes;
                winner = candidate;
            } 
            // Tie-break: same votes, pick lexicographically smaller name
            else if (votes == maxVotes) {
                if (candidate.compareTo(winner) < 0) {
                    winner = candidate;
                }
            }
        }

        return winner;
    }
}
