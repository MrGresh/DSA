class Solution {
    public int minMirrorPairDistance(int[] nums) {
        Map<Integer, List<Integer>> map = new HashMap<>();
        for(int i=0;i<nums.length;i++) {
            if(!map.containsKey(nums[i])) map.put(nums[i], new ArrayList<>());
            map.get(nums[i]).add(i);
        }
        int ans = Integer.MAX_VALUE;
        for(int i=0;i<nums.length;i++) {
            int revNum = rev(nums[i]);
            if(map.containsKey(revNum)) {
                List<Integer> indices = map.get(revNum);
                int pos = Collections.binarySearch(indices, i);
                
                int insertionPoint = (pos < 0) ? -(pos + 1) : pos + 1;

                if (insertionPoint < indices.size()) {
                    int j = indices.get(insertionPoint);
                    ans = Math.min(ans, j - i);
                }
            }
        }
        return ans == Integer.MAX_VALUE ? -1 : ans;
    }
    int rev(int n) {
        int ans = 0;
        while(n>0) {
            ans = ans*10 + n%10;
            n /= 10;
        }
        return ans;
    }
}
