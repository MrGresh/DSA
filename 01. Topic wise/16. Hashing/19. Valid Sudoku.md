class Solution {
    public boolean isValidSudoku(char[][] board) {
        for(int i=0;i<9;i++) {
            Set<Character> row = new HashSet<>();
            Set<Character> col = new HashSet<>();
            for(int j=0;j<9;j++) {
                if(board[i][j]!='.') {
                    if(row.contains(board[i][j])) return false;
                    row.add(board[i][j]);
                }
                if(board[j][i]!='.') {
                    if(col.contains(board[j][i])) return false;
                    col.add(board[j][i]);
                }
            }
        }
        for(int i=1;i<=9;i++) {
            Set<Character> matrix = new HashSet<>();
            for(int[] p : getPos(i)) {
                if(board[p[0]][p[1]]!='.') {
                    if(matrix.contains(board[p[0]][p[1]])) return false;
                    matrix.add(board[p[0]][p[1]]);
                }
            }
        }
        return true;
    }
    int[][] getPos(int i) {
        if (i==1) {
            return new int[][] {{0,0},{0,1},{0,2},{1,0},{1,1},{1,2},{2,0},{2,1},{2,2}};
        }
        if (i==2) {
            return new int[][] {{0,3},{0,4},{0,5},{1,3},{1,4},{1,5},{2,3},{2,4},{2,5}};
        }
        if (i==3) {
            return new int[][] {{0,6},{0,7},{0,8},{1,6},{1,7},{1,8},{2,6},{2,7},{2,8}};
        }
        if (i==4) {
            return new int[][] {{3,0},{3,1},{3,2},{4,0},{4,1},{4,2},{5,0},{5,1},{5,2}};
        }
        if (i==5) {
            return new int[][] {{3,3},{3,4},{3,5},{4,3},{4,4},{4,5},{5,3},{5,4},{5,5}};
        }
        if (i==6) {
            return new int[][] {{3,6},{3,7},{3,8},{4,6},{4,7},{4,8},{5,6},{5,7},{5,8}};
        }
        if (i==7) {
            return new int[][] {{6,0},{6,1},{6,2},{7,0},{7,1},{7,2},{8,0},{8,1},{8,2}};
        }
        if (i==8) {
            return new int[][] {{6,3},{6,4},{6,5},{7,3},{7,4},{7,5},{8,3},{8,4},{8,5}};
        }
        if (i==9) {
            return new int[][] {{6,6},{6,7},{6,8},{7,6},{7,7},{7,8},{8,6},{8,7},{8,8}};
        }
        return new int[0][0];
    }
}
