class Solution {
    public int findSubarray(int[] arr) {
        long currentSum = 0;
        int totalCount = 0;
        Map<Long, Integer> map = new HashMap<>();
        
        for (int val : arr) {
            currentSum += val;
            if(currentSum==0) totalCount++;
            if (map.containsKey(currentSum)) {
                totalCount += map.get(currentSum);
            }
            map.put(currentSum, map.getOrDefault(currentSum, 0) + 1);
        }
        
        return totalCount;
    }
}
