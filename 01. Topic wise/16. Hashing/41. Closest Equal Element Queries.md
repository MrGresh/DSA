class Solution {
    public List<Integer> solveQueries(int[] nums, int[] queries) {
        Map<Integer, List<Integer>> data = new HashMap<>();
        for(int i=0;i<nums.length;i++) {
            if(!data.containsKey(nums[i])) data.put(nums[i], new ArrayList<>());
            data.get(nums[i]).add(i);
        }
        List<Integer> ans = new ArrayList<>();
        for(int i: queries) {
            if(data.get(nums[i]).size()==1) ans.add(-1);
            else {
                List<Integer> indices = data.get(nums[i]);
                int n = nums.length;
                int m = indices.size();
                
                int pos = Collections.binarySearch(indices, i);
                
                int minFound = Integer.MAX_VALUE;

                if (pos > 0) {
                    int leftIdx = indices.get(pos - 1);
                    minFound = i - leftIdx;
                }
                
                if (pos < m - 1) {
                    int rightIdx = indices.get(pos + 1);
                    minFound = Math.min(minFound, rightIdx - i);
                }

                int firstIdx = indices.get(0);
                int lastIdx = indices.get(m - 1);
                
                int circularDist = n - (lastIdx - firstIdx);
                if (pos == 0 || pos == m-1) minFound = Math.min(minFound, circularDist);

                ans.add(minFound);
            }
        }
        return ans;
    }
}
