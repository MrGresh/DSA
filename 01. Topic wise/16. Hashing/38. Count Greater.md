class Solution {
    public int[] countGreater(int[] arr) {
        int n = arr.length;
        int[] result = new int[n];
        
        // 1. TreeMap with Reverse Order to keep largest elements first
        TreeMap<Integer, Integer> m = new TreeMap<>(Collections.reverseOrder());
        
        // 2. Count frequencies: O(n log n)
        for (int x : arr) {
            m.put(x, m.getOrDefault(x, 0) + 1);
        }
        
        // 3. Transform frequencies into cumulative "greater than" counts: O(d log d)
        int cumFreq = 0;
        for (Map.Entry<Integer, Integer> e : m.entrySet()) {
            int temp = e.getValue(); // Current frequency
            m.put(e.getKey(), cumFreq); // Replace with count of elements seen so far (which are larger)
            cumFreq += temp; // Update cumulative sum for the next smaller key
        }
        
        // 4. Map back to result array: O(n log n)
        for (int i = 0; i < n; i++) {
            result[i] = m.get(arr[i]);
        }
        
        return result;
    }
}
