class Solution {
    public static int[] findCeilingOnRight(int[] arr) {
        int n = arr.length;
        int[] result = new int[n];
        TreeSet<Integer> set = new TreeSet<>();

        // Traverse from right to left
        for (int i = n - 1; i >= 0; i--) {
            // Find the smallest element in the set >= arr[i]
            Integer ceil = set.ceiling(arr[i]);

            if (ceil == null) {
                result[i] = -1; // No ceiling found to the right
            } else {
                result[i] = ceil;
            }

            // Add the current element to the set for elements to its left
            set.add(arr[i]);
        }

        return result;
    }
}
