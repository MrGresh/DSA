class Solution {
    public int unequalTriplets(int[] arr) {
        Map<Integer, Integer> freqMap = new HashMap<>();
        for (int x : arr) freqMap.put(x, freqMap.getOrDefault(x, 0) + 1);

        int left = 0, right = arr.length, result = 0;

        for (int mid : freqMap.values()) {
            right -= mid;
            result += left * mid * right;
            left += mid;
        }

        return result;
    }
}
