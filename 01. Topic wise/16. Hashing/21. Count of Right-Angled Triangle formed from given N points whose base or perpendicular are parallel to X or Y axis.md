public class RightTriangleCounter {
    public static long countRightTriangles(int[][] points) {
        Map<Integer, Integer> xFreq = new HashMap<>();
        Map<Integer, Integer> yFreq = new HashMap<>();

        // Step 1: Populate the frequency maps
        for (int[] point : points) {
            int x = point[0];
            int y = point[1];
            xFreq.put(x, xFreq.getOrDefault(x, 0) + 1);
            yFreq.put(y, yFreq.getOrDefault(y, 0) + 1);
        }

        long totalTriangles = 0;

        // Step 2: Iterate through points to treat each as the "pivot" (90-degree vertex)
        for (int[] point : points) {
            int x = point[0];
            int y = point[1];
            // Points sharing same X (excluding current point)
            long sameX = xFreq.get(x) - 1;
            // Points sharing same Y (excluding current point)
            long sameY = yFreq.get(y) - 1;
            // Combinations formed with this point as the corner
            totalTriangles += (sameX * sameY);
        }
        return totalTriangles;
    }
}
