class Solution {
    static int distinct(int M[][], int N) {
        Set<Integer> prev = new HashSet<>();
        for(int j=0;j<N;j++) prev.add(M[0][j]);
        for(int i=1;i<N;i++) {
            Set<Integer> curr = new HashSet<>();
            for(int j=0;j<N;j++) {
                if(prev.contains(M[i][j])) curr.add(M[i][j]);
            }
            prev = curr;
        }
        return prev.size();
    }
}
