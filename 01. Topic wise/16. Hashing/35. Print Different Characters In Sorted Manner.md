class Solution {
    public static String Sorted_Characters(String s) {
        TreeSet<Character> charSet = new TreeSet<>();
        for (char c : s.toCharArray()) {
            charSet.add(c);
        }
        StringBuilder sb = new StringBuilder();
        for (char c : charSet) {
            sb.append(c);
        }
        return sb.toString();
    }
}
