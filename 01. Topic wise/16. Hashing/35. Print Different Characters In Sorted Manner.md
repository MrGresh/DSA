class Solution {
    public static String Sorted_Characters(String s) {
        boolean[] vis = new boolean[26];
        for (char c : s.toCharArray()) {
            vis[c-'a'] = true;
        }
        StringBuilder sb = new StringBuilder();
        for (int i=0;i<26;i++) {
            if(vis[i]) sb.append((char)(i+'a'));
        }
        return sb.toString();
    }
}
