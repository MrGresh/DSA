class GetMin {
    // Push elements of array into stack
    public static Stack<Integer> _push(int arr[], int n) {
        Stack<Integer> s = new Stack<>();
        for (int x: arr) {
            s.push(x);
        }
        return s;
    }

    // Print minimum at each pop
    static void _getMinAtPop(Stack<Integer> s) {
        // Auxiliary stack to store minimums
        Stack<Integer> minStack = new Stack<>();

        // Build minStack while traversing original stack
        for (int i = 0; i < s.size(); i++) {
            if (minStack.isEmpty()) {
                minStack.push(s.get(i));
            } else {
                minStack.push(Math.min(minStack.peek(), s.get(i)));
            }
        }

        // Now pop elements and print min at each pop
        while (!s.isEmpty()) {
            s.pop();
            System.out.print(minStack.pop() + " ");
        }
        System.out.println();
    }
}
