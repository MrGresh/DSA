class Solution {
    public static int profit(int arr[]) {
        int n = arr.length;
        int totalProfit = 0;
        // Stack to store elements to the right
        Stack<Integer> stack = new Stack<>();

        // Iterate from right to left
        for (int i = n - 1; i >= 0; i--) {
            // Remove elements from stack that are smaller than current element
            // because they cannot be the "Next Greater or Equal" for elements to the left
            while (!stack.isEmpty() && stack.peek() < arr[i]) {
                stack.pop();
            }

            if (stack.isEmpty()) {
                // No element to the right is >= arr[i]
                totalProfit += arr[i];
            } else {
                // The top of the stack is the first element arr[j] >= arr[i]
                totalProfit += (stack.peek() - arr[i]);
            }

            // Push current element onto stack for the next iterations
            stack.push(arr[i]);
        }

        return totalProfit;
    }
}
