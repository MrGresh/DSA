class Solution {
    public ArrayList<Integer> reducePairs(int[] arr) {
        Stack<Integer> st = new Stack<>();
        for(int x: arr) {
            if(st.isEmpty()) st.push(x);
            else {
                boolean canInsert = true;
                while(!st.isEmpty()) {
                    if((st.peek()<0 && x>0) || (st.peek()>0 && x<0)) { // Opp Sign
                        if(Math.abs(x)<Math.abs(st.peek())) {
                            canInsert = false;
                            break;
                        }
                        else if(Math.abs(x)==Math.abs(st.peek())) {
                            st.pop();
                            canInsert = false;
                            break;
                        } else st.pop();
                    } else break; // No Opp Sign
                }
                if(canInsert) st.push(x);
            }
        }
        return new ArrayList<>(st);
    }
}
