class Solution {
    static class Data {
        int pos;
        int health;
        int originalIdx;
        char dir;
        Data(int a, int b, int i, char c) {
            pos = a;
            health = b;
            originalIdx = i;
            dir = c;
        }
    }
    public List<Integer> survivedRobotsHealths(int[] pos, int[] healths, String dirs) {
        List<Data> data = new ArrayList<>();
        for(int i=0;i<healths.length;i++) data.add(new Data(pos[i], healths[i], i, dirs.charAt(i)));
        data.sort((d1, d2) -> Integer.compare(d1.pos, d2.pos));

        List<Data> ansData = new ArrayList<>();

        int l=0, r=data.size()-1;
        while(l<data.size() && data.get(l).dir=='L') ansData.add(data.get(l++));
        while(r>=0 && data.get(r).dir=='R') ansData.add(data.get(r--));

        Stack<Data> st = new Stack<>();
        for(int i=l;i<=r;i++) {
            if(data.get(i).dir=='R') st.push(data.get(i));
            else {
                while(!st.isEmpty() && st.peek().dir=='R' && st.peek().health<data.get(i).health) {
                    st.pop();
                    data.get(i).health--;
                    if(data.get(i).health==0) break;
                }
                if(data.get(i).health==0) continue;
                if(!st.isEmpty() && st.peek().dir=='R' && st.peek().health==data.get(i).health) {
                    st.pop();
                    continue;
                }
                if(!st.isEmpty() && st.peek().dir=='R' && st.peek().health>data.get(i).health) {
                    st.peek().health--;
                    continue;
                }
                st.push(data.get(i));
            }
        }

        for(Data d: st) ansData.add(d);
        ansData.sort((d1, d2) -> Integer.compare(d1.originalIdx, d2.originalIdx));
        List<Integer> ans = new ArrayList<>();
        for(Data d: ansData) ans.add(d.health);
        return ans;
    }
}
