class Solution {
    public void deleteMid(Stack<Integer> s) {
        int size = s.size();
        int mid = (size/2)+1;
        deleteHelper(s, mid);
    }
    private void deleteHelper(Stack<Integer> s, int k) {
        if (k == 1) {
            s.pop();
            return;
        }
        int temp = s.pop();
        deleteHelper(s, k - 1);
        s.push(temp);
    }
}
