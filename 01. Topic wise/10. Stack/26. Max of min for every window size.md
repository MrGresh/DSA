class Solution {
    public ArrayList<Integer> maxOfMins(int[] arr) {
        int n = arr.length;
        int[] left = new int[n];
        int[] right = new int[n];
        Stack<Integer> s = new Stack<>();

        // 1. Find the previous smaller element for every index
        for (int i = 0; i < n; i++) {
            while (!s.isEmpty() && arr[s.peek()] >= arr[i]) s.pop();
            left[i] = s.isEmpty() ? -1 : s.peek();
            s.push(i);
        }

        s.clear();

        // 2. Find the next smaller element for every index
        for (int i = n - 1; i >= 0; i--) {
            while (!s.isEmpty() && arr[s.peek()] >= arr[i]) s.pop();
            right[i] = s.isEmpty() ? n : s.peek();
            s.push(i);
        }

        // 3. Create a result array to store max of mins for each length
        int[] res = new int[n + 1];
        for (int i = 0; i < n; i++) {
            // This element is the minimum for this specific length
            int len = right[i] - left[i] - 1;
            res[len] = Math.max(res[len], arr[i]);
        }

        // 4. Fill in entries for lengths that weren't explicitly reached
        // Some elements are minimum for a large window, but they are also
        // valid for smaller windows.
        for (int i = n - 1; i >= 1; i--) {
            res[i] = Math.max(res[i], res[i + 1]);
        }

        // Convert to ArrayList for the final answer
        ArrayList<Integer> result = new ArrayList<>();
        for (int i = 1; i <= n; i++) {
            result.add(res[i]);
        }

        return result;
    }
}
