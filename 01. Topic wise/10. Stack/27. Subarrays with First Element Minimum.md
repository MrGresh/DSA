class Solution {
    public int countSubarrays(int[] arr) {
        int n = arr.length;
        int[] nextSmaller = new int[n];
        Stack<Integer> stack = new Stack<>();

        // Fill nextSmaller array using a monotonic stack
        for (int i = n - 1; i >= 0; i--) {
            while (!stack.isEmpty() && arr[stack.peek()] >= arr[i]) {
                stack.pop();
            }
            
            if (stack.isEmpty()) {
                nextSmaller[i] = n; // No smaller element found to the right
            } else {
                nextSmaller[i] = stack.peek();
            }
            stack.push(i);
        }

        // Total count is the sum of (nextSmallerIndex - currentIndex)
        int totalCount = 0;
        for (int i = 0; i < n; i++) {
            totalCount += (nextSmaller[i] - i);
        }

        return totalCount;
    }
}
