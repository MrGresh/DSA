class Solution {
    public static String removeConsecutiveDuplicates(String s) {
        if (s == null || s.length() == 0) return "";
        
        Stack<Character> stack = new Stack<>();
        
        for (int i = 0; i < s.length(); i++) {
            char currentChar = s.charAt(i);
            
            // If stack is empty or current char is different from the top
            if (stack.isEmpty() || stack.peek() != currentChar) {
                stack.push(currentChar);
            }
        }
        
        // Construct the string from the stack
        StringBuilder sb = new StringBuilder();
        for (char c : stack) {
            sb.append(c);
        }
        
        return sb.toString();
    }
}

class Solution {
    public static String removeConsecutiveDuplicates(String s) {
        if (s == null || s.length() == 0) return "";
        
        StringBuilder sb = new StringBuilder();
        
        for (int i = 0; i < s.length(); i++) {
            char currentChar = s.charAt(i);
            
            // If sb is empty OR current char is not the same as the last added char
            if (sb.length() == 0 || sb.charAt(sb.length() - 1) != currentChar) {
                sb.append(currentChar);
            }
        }
        
        return sb.toString();
    }
}
