class TrieNode {
    TrieNode[] children = new TrieNode[26];
    int count = 0; // Tracks how many times this prefix has been visited
    int endCount = 0; // Tracks how many times this exact word has ended here
}

class Solution {
    static TrieNode root;

    public static void check(String arr[], int n) {
        root = new TrieNode();
        StringBuilder output = new StringBuilder();

        for (String s : arr) {
            output.append(getStationCode(s)).append("\n");
        }
        System.out.print(output);
    }

    private static String getStationCode(String s) {
        TrieNode curr = root;
        String prefix = "";
        boolean foundPrefix = false;

        for (int i = 0; i < s.length(); i++) {
            int idx = s.charAt(i) - 'a';
            
            // If we haven't found a unique prefix yet, keep building it
            if (!foundPrefix) {
                prefix += s.charAt(i);
            }

            if (curr.children[idx] == null) {
                curr.children[idx] = new TrieNode();
                foundPrefix = true; // This is the first time this path is taken
            }
            
            curr = curr.children[idx];
            curr.count++;
        }

        // Handle exact duplicates
        if (curr.endCount > 0) {
            curr.endCount++;
            return s + " " + curr.endCount;
        } else {
            curr.endCount = 1;
            // If the word was already a prefix of another word, return the full word
            // Otherwise return the shortest unique prefix found during traversal
            return foundPrefix ? prefix : s;
        }
    }
}
