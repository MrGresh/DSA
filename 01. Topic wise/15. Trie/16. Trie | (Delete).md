/*Complete the function below
Node is as follows:
class TrieNode
{
    char content;
    boolean isEnd;
    int count;
    LinkedList<TrieNode> childList;
    public TrieNode(char c)
    {
        childList = new LinkedList<TrieNode>();
        isEnd = false;
        content = c;
        count = 0;
    }
    public TrieNode subNode(char c)
    {
        if (childList != null)
            for (TrieNode eachChild : childList)
                if (eachChild.content == c)
                    return eachChild;
        return null;
    }
}*/
class Solution {
    public static void deleteKey(TrieNode root, char[] key) {
        deleteRecursive(root, key, 0);
    }

    private static boolean deleteRecursive(TrieNode current, char[] key, int index) {
        // Base case: we reached the end of the key
        if (index == key.length) {
            // If the word exists, unmark the end flag
            if (current.isEnd) {
                current.isEnd = false;
            }
            // If it has no other children, it's safe to delete this node
            return current.childList.isEmpty();
        }

        char c = key[index];
        TrieNode node = current.subNode(c);

        // Word not found in Trie
        if (node == null) {
            return false;
        }

        // Recursively try to delete the next character
        boolean shouldDeleteChild = deleteRecursive(node, key, index + 1);

        // If the child node is no longer needed (not an end, no more children)
        if (shouldDeleteChild) {
            current.childList.remove(node);
            
            // Return true if the current node can also be deleted:
            // 1. It is not the end of another word
            // 2. It has no more children left
            return !current.isEnd && current.childList.isEmpty();
        }

        return false;
    }
}
