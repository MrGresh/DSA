class TrieNode {
    TrieNode[] children = new TrieNode[26];
    int count = 0;
    int firstIndex = -1;
}

class Solution {
    public String mostFrequentWord(String arr[], int n) {
        TrieNode root = new TrieNode();
        int maxFreq = 0;
        String result = "";
        int lastFirstOccurrence = -1;

        for (int i = 0; i < n; i++) {
            String word = arr[i];
            TrieNode curr = root;
            
            for (char c : word.toCharArray()) {
                int idx = c - 'a';
                if (curr.children[idx] == null) {
                    curr.children[idx] = new TrieNode();
                }
                curr = curr.children[idx];
            }

            // If this is the first time we see this word, mark its index
            if (curr.firstIndex == -1) {
                curr.firstIndex = i;
            }
            curr.count++;

            // Check if this word is a candidate for the most frequent
            if (curr.count > maxFreq) {
                maxFreq = curr.count;
                result = word;
                lastFirstOccurrence = curr.firstIndex;
            } else if (curr.count == maxFreq) {
                // Tie-breaker: word whose first occurrence comes latest
                if (curr.firstIndex >= lastFirstOccurrence) {
                    lastFirstOccurrence = curr.firstIndex;
                    result = word;
                }
            }
        }

        return result;
    }
}
