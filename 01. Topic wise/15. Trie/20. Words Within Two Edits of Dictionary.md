### Brute Force with Set
```java
class Solution {
    public List<String> twoEditWords(String[] queries, String[] dictionary) {
        int n = queries[0].length();
        Set<String> dict = new HashSet<>();
        for(String w: dictionary) dict.add(w);
        List<String> res = new ArrayList<>();
        for(String w: queries) {
            if(canAdd(dict, w, n)) res.add(w);
        }
        return res;
    }
    boolean canAdd(Set<String> dict, String w, int n) {
        if(dict.contains(w)) return true;

        StringBuilder str = new StringBuilder(w);
        
        // try by changing 1 char
        for(int i=0;i<n;i++) {
            char original = str.charAt(i);
            for(char ch='a';ch<='z';ch++) {
                if(ch!=original) {
                    str.setCharAt(i, ch);
                    if(dict.contains(str.toString())) return true;
                }
            }
            str.setCharAt(i, original);
        }

        // try by changing 2 chars
        for(int i=0;i<n;i++) {
            for(int j=0;j<n;j++) {
                if(i!=j) {
                    char orgi1 = str.charAt(i);
                    char orgi2 = str.charAt(j);
                    for(char ch='a';ch<='z';ch++) {
                        if(ch!=orgi1) {
                            str.setCharAt(i, ch);
                            for(char ch2='a';ch2<='z';ch2++) {
                                if(ch2!=orgi2) {
                                    str.setCharAt(j, ch2);
                                    if(dict.contains(str.toString())) return true;
                                }
                            }
                            str.setCharAt(j, orgi2);
                        }
                    }
                    str.setCharAt(i, orgi1);
                }
            }
        }
        return false;
    }
}
```

### Trie Based
```java
class TrieNode {
    Map<Character, TrieNode> childs = new HashMap<>();
}
class Solution {
    public List<String> twoEditWords(String[] queries, String[] dictionary) {
        TrieNode root = new TrieNode();
        for(String w: dictionary) {
            TrieNode node = root;
            for(int i=0;i<w.length();i++) {
                if(!node.childs.containsKey(w.charAt(i))) {
                    node.childs.put(w.charAt(i), new TrieNode());
                }
                node = node.childs.get(w.charAt(i));
            }
        }
        List<String> res = new ArrayList<>();
        for(String w: queries) {
            if(canAdd(w, 0, 0, root)) res.add(w);
        }
        return res;
    }
    boolean canAdd(String w, int i, int count, TrieNode root) {
        if(i==w.length()) return true;
        if(root.childs.containsKey(w.charAt(i)) && canAdd(w, i+1, count, root.childs.get(w.charAt(i)))) {
            return true;
        }
        if(count<2) {
            for(TrieNode node: root.childs.values()) {
                if(canAdd(w, i+1, count+1, node)) return true;
            }
        }
        return false;
    }
}
```
