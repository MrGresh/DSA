class TrieNode {
    TrieNode[] children = new TrieNode[2];
}

class Solution {
    static TrieNode root;

    static void insert(int val) {
        TrieNode curr = root;
        // Check bits from 30 down to 0 (since max arr[i] is 10^6)
        for (int i = 30; i >= 0; i--) {
            int bit = (val >> i) & 1;
            if (curr.children[bit] == null) {
                curr.children[bit] = new TrieNode();
            }
            curr = curr.children[bit];
        }
    }

    static int getMaxXOR(int val) {
        TrieNode curr = root;
        int maxXor = 0;
        for (int i = 30; i >= 0; i--) {
            int bit = (val >> i) & 1;
            int oppositeBit = 1 - bit;
            
            // Greedy: Try to go to the opposite bit to get a '1' in XOR result
            if (curr.children[oppositeBit] != null) {
                maxXor |= (1 << i);
                curr = curr.children[oppositeBit];
            } else {
                curr = curr.children[bit];
            }
        }
        return maxXor;
    }

    static int maxSubarrayXOR(int arr[], int n) {
        root = new TrieNode();
        insert(0); // Base case: XOR of empty prefix is 0
        
        int preXor = 0;
        int maxResult = 0;
        
        for (int i = 0; i < n; i++) {
            preXor ^= arr[i];
            insert(preXor);
            maxResult = Math.max(maxResult, getMaxXOR(preXor));
        }
        
        return maxResult;
    }
}
