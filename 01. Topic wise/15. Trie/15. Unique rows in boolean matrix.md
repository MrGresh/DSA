## Linked HashMap
```java
class GfG {
    public static ArrayList<ArrayList<Integer>> uniqueRow(int a[][], int r, int c) {
        // LinkedHashSet keeps elements unique AND maintains insertion order
        LinkedHashSet<ArrayList<Integer>> set = new LinkedHashSet<>();

        for (int i = 0; i < r; i++) {
            ArrayList<Integer> currentRow = new ArrayList<>();
            for (int j = 0; j < c; j++) {
                currentRow.add(a[i][j]);
            }
            // add() returns false if the element already exists
            set.add(currentRow);
        }

        return new ArrayList<>(set);
    }
}
```

## Trie
```java
class TrieNode {
    TrieNode[] children = new TrieNode[2];
    boolean isEndOfRow = false;
}
class GfG {
    public static ArrayList<ArrayList<Integer>> uniqueRow(int a[][], int r, int c) {
        ArrayList<ArrayList<Integer>> result = new ArrayList<>();
        TrieNode root = new TrieNode();
        for (int i = 0; i < r; i++) {
            if (insert(root, a[i], c)) {
                ArrayList<Integer> rowList = new ArrayList<>();
                for (int val : a[i]) rowList.add(val);
                result.add(rowList);
            }
        }
        return result;
    }
    private static boolean insert(TrieNode root, int[] row, int c) {
        TrieNode curr = root;
        for (int j = 0; j < c; j++) {
            int val = row[j];
            if (curr.children[val] == null) {
                curr.children[val] = new TrieNode();
            }
            curr = curr.children[val];
        }
        if (!curr.isEndOfRow) {
            curr.isEndOfRow = true;
            return true;
        }
        return false;
    }
}
```
