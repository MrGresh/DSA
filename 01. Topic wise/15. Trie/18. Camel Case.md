class TrieNode {
    TrieNode[] children = new TrieNode[26];
    // List of words that have THIS exact abbreviation
    ArrayList<String> actualWords = new ArrayList<>();
}

class Solution {
    static void findAllWords(String[] dict, String pattern) {
        TrieNode root = new TrieNode();

        // 1. Build the Trie based on Uppercase letters
        for (String word : dict) {
            TrieNode curr = root;
            for (char c : word.toCharArray()) {
                if (Character.isUpperCase(c)) {
                    int idx = c - 'A';
                    if (curr.children[idx] == null) {
                        curr.children[idx] = new TrieNode();
                    }
                    curr = curr.children[idx];
                }
            }
            // Add the word only to the node representing its FULL abbreviation
            curr.actualWords.add(word);
        }

        // 2. Navigate to the end of the pattern
        TrieNode curr = root;
        for (char c : pattern.toCharArray()) {
            int idx = c - 'A';
            if (curr.children[idx] == null) {
                System.out.print("No match found");
                return;
            }
            curr = curr.children[idx];
        }

        // 3. DFS from the pattern node to collect words in lexicographical order of abbreviations
        ArrayList<String> result = new ArrayList<>();
        collectWords(curr, result);

        if (result.isEmpty()) {
            System.out.print("No match found");
        } else {
            for (int i = 0; i < result.size(); i++) {
                System.out.print(result.get(i) + (i == result.size() - 1 ? "" : " "));
            }
        }
    }

    private static void collectWords(TrieNode node, ArrayList<String> result) {
        if (node == null) return;

        // Sort words that share the same abbreviation
        Collections.sort(node.actualWords);
        result.addAll(node.actualWords);

        // Recursively visit children from A to Z to maintain lexicographical order of abbreviations
        for (int i = 0; i < 26; i++) {
            if (node.children[i] != null) {
                collectWords(node.children[i], result);
            }
        }
    }
}
