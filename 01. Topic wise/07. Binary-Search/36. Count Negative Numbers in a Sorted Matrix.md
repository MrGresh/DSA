class Solution {
    public int countNegatives(int[][] a) {
        int res = 0;
        int i=0, j=a[0].length-1;
        while(i<a.length && j>=0) {
            if(a[i][j]<0) {
                res += (a.length-i);
                j--;
            }
            else i++;
        }
        return res;
    }
}
