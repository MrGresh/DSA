class Solution {
    public long minNumberOfSeconds(int mountainHeight, int[] workerTimes) {
        long low = 0;
        long high = (long) 1e18;
        long ans = high;

        while (low <= high) {
            long mid = low + (high - low) / 2;
            if (isPossible(mountainHeight, workerTimes, mid)) {
                ans = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }
        return ans;
    }

    private boolean isPossible(int mountainHeight, int[] workerTimes, long maxTime) {
        long totalHeightRemoved = 0;
        for (int w : workerTimes) {
            long limit = (2 * maxTime) / w;
            long x = (long) ((-1 + Math.sqrt(1 + 4 * limit)) / 2);
            
            totalHeightRemoved += x;
            if (totalHeightRemoved >= mountainHeight) return true;
        }
        return totalHeightRemoved >= mountainHeight;
    }
}
