class Solution {
    public int countOnes(int[] arr) {
        int low = 0;
        int high = arr.length - 1;
        int lastOneIndex = -1;

        while (low <= high) {
            int mid = low + (high - low) / 2;

            if (arr[mid] == 1) {
                lastOneIndex = mid;
                low = mid + 1; 
            } else high = mid - 1;
        }

        return lastOneIndex + 1;
    }
}
