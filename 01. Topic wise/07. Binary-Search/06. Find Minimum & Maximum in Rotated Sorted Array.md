class Solution {
    public int findMin(int[] nums) {
        int l=0, r=nums.length-1;
        while(l<r) {
            int m = l+(r-l)/2;
            if(nums[m]<=nums[r]) r=m;
            else l=m+1;
        }
        return nums[l];
    }
}

int findMax(int[] arr) {
    int low = 0, high = arr.length - 1;

    // 1. If not rotated (or single element), last element is max
    if (arr[low] <= arr[high]) return arr[high];

    while (low <= high) {
        int mid = low + (high - low) / 2;

        // 2. Check if mid is the "peak"
        if (mid < arr.length - 1 && arr[mid] > arr[mid + 1]) {
            return arr[mid];
        }

        // 3. Standard Binary Search logic
        if (arr[mid] >= arr[low]) {
            // Left side is sorted, peak is to the right
            low = mid + 1;
        } else {
            // Right side is sorted, peak is to the left
            high = mid - 1;
        }
    }
    return arr[low];
}
