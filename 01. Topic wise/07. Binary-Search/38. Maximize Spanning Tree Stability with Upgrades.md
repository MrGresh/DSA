class Solution {
    public int maxStability(int n, int[][] edges, int k) {
        int low = 0, high = 200000;
        int ans = -1;

        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (isPossible(mid, n, edges, k)) {
                ans = mid;
                low = mid + 1;
            } else {
                high = mid - 1;
            }
        }
        return ans;
    }

    private boolean isPossible(int target, int n, int[][] edges, int k) {
        DSU dsu = new DSU(n);
        int edgesCount = 0;
        int upgradesUsed = 0;

        // 1. Mandatory Edges (Priority 1)
        for (int[] e : edges) {
            if (e[3] == 1) {
                if (e[2] < target) return false;
                if (!dsu.union(e[0], e[1])) return false;
                edgesCount++;
            }
        }

        // 2. Optional Edges that meet target WITHOUT upgrade (Priority 2)
        for (int[] e : edges) {
            if (e[3] == 0 && e[2] >= target && dsu.union(e[0], e[1])) {
                edgesCount++;
            }
        }

        // 3. Optional Edges that meet target ONLY WITH upgrade (Priority 3)
        for (int[] e : edges) {
            if (e[3] == 0 && e[2] < target && upgradesUsed < k && e[2] * 2 >= target && dsu.union(e[0], e[1])) {
                upgradesUsed++;
                edgesCount++;
            }
        }

        return edgesCount == n - 1;
    }

    class DSU {
        int[] parent;
        DSU(int n) {
            parent = new int[n];
            for (int i = 0; i < n; i++) parent[i] = i;
        }
        int find(int i) {
            if (parent[i] == i) return i;
            return parent[i] = find(parent[i]);
        }
        boolean union(int i, int j) {
            int rootI = find(i);
            int rootJ = find(j);
            if (rootI != rootJ) {
                parent[rootI] = rootJ;
                return true;
            }
            return false;
        }
    }
}
