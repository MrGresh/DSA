class Solution {
    int count = 0;
    int getCount(Node root, int l, int h) {
        count = 0;
        calc(root, l, h);
        return count;
    }
    void calc(Node root, int low, int high) {
        if(root==null) return;
        if(root.data>=low && root.data<=high) count ++;
        if(root.data>=low) calc(root.left, low, high);
        if(root.data<=high) calc(root.right, low, high);
    }
}
