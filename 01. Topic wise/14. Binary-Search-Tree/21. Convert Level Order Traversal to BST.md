class NodeDetails {
    Node node;
    int min, max;

    NodeDetails(Node node, int min, int max) {
        this.node = node;
        this.min = min;
        this.max = max;
    }
}

class GFG {
    public Node constructBST(int[] arr) {
        int n = arr.length;
        if (n == 0) return null;

        int i = 0;
        Node root = new Node(arr[i++]);
        
        // Queue to store nodes along with their valid range
        Queue<NodeDetails> queue = new LinkedList<>();
        queue.add(new NodeDetails(root, Integer.MIN_VALUE, Integer.MAX_VALUE));

        while (i < n) {
            NodeDetails current = queue.poll();

            // Check if the next element in arr can be the left child
            if (arr[i] < current.node.data && arr[i] > current.min) {
                current.node.left = new Node(arr[i]);
                queue.add(new NodeDetails(current.node.left, current.min, current.node.data));
                i++;
            }

            // Check if the next element in arr can be the right child
            if (i < n && arr[i] > current.node.data && arr[i] < current.max) {
                current.node.right = new Node(arr[i]);
                queue.add(new NodeDetails(current.node.right, current.node.data, current.max));
                i++;
            }
        }

        return root;
    }
}
