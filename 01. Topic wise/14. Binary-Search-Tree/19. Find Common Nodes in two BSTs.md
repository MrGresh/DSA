class Solution {
    public static ArrayList<Integer> findCommon(Node root1, Node root2) {
        ArrayList<Integer> result = new ArrayList<>();
        Stack<Node> s1 = new Stack<>();
        Stack<Node> s2 = new Stack<>();

        while (true) {
            // Push left nodes of BST1 to stack 1
            while (root1 != null) {
                s1.push(root1);
                root1 = root1.left;
            }
            // Push left nodes of BST2 to stack 2
            while (root2 != null) {
                s2.push(root2);
                root2 = root2.left;
            }

            // If either stack is empty, no more common nodes possible
            if (s1.isEmpty() || s2.isEmpty()) break;

            root1 = s1.peek();
            root2 = s2.peek();

            if (root1.data == root2.data) {
                // Common node found
                result.add(root1.data);
                s1.pop();
                s2.pop();
                // Move to right children for both
                root1 = root1.right;
                root2 = root2.right;
            } else if (root1.data < root2.data) {
                // BST1 value is smaller, need a larger value from BST1
                s1.pop();
                root1 = root1.right;
                root2 = null; // Keep BST2 node on stack
            } else {
                // BST2 value is smaller, need a larger value from BST2
                s2.pop();
                root2 = root2.right;
                root1 = null; // Keep BST1 node on stack
            }
        }
        return result;
    }
}
