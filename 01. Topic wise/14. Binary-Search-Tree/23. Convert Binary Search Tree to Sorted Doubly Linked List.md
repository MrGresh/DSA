class Solution {
    // To keep track of the head of the DLL and the previously processed node
    Node head = null;
    Node prev = null;

    public Node treeToDoublyList(Node root) {
        if (root == null) return null;

        // Step 1: Inorder traversal to link nodes
        inorder(root);
        
        // Note: If you need a CIRCULAR DLL, uncomment the lines below:
        // prev.right = head;
        // head.left = prev;

        return head;
    }

    private void inorder(Node curr) {
        if (curr == null) return;

        // Recursive call on left subtree
        inorder(curr.left);

        // Process current node
        if (prev == null) {
            // If prev is null, this is the smallest node (leftmost leaf)
            head = curr;
        } else {
            // Link current node with the previous node
            prev.right = curr;
            curr.left = prev;
        }
        
        // Move prev to the current node
        prev = curr;

        // Recursive call on right subtree
        inorder(curr.right);
    }
}
