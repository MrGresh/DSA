class Solution {
    static int minDiff(Node root, int K) {
        int ans = Integer.MAX_VALUE;
        while(root!=null) {
            if(root.data==K) return 0;
            if(root.data<K) {
                ans = Math.min(ans, K-root.data);
                root = root.right;
            } else {
                ans = Math.min(ans, root.data-K);
                root = root.left;
                
            }
        }
        return ans;
    }
}
