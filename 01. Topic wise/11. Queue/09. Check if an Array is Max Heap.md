class Solution {
    public boolean isMaxHeap(int[] arr) {
        if(arr.length==1) return true;
        Queue<Integer> q = new LinkedList<>();
        q.add(arr[0]);
        int idx = 1;
        while(!q.isEmpty()) {
            int x = q.poll();
            
            if(arr[idx]>x) return false;
            else q.add(arr[idx]);
            
            if(idx+1<arr.length) {
                if(arr[idx+1]>x) return false;
                else q.add(arr[idx+1]);
            }
            
            idx += 2;
            if(idx>=arr.length) return true;
        }
        return true;
    }
}
