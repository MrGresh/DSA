class Solution {
    public int[] maxSlidingWindow(int[] nums, int k) {
        int[] ans = new int[nums.length-k+1];
        int idx = -1;
        Deque<Integer> dq = new LinkedList<>();
        int start = 0;
        for(int end=0;end < nums.length;end++) {
            while(!dq.isEmpty() && nums[end] > nums[dq.peekLast()]) dq.pollLast();
            dq.addLast(end);
            if(end-start+1==k) {
                ans[++idx] = nums[dq.peekFirst()];
                if(start==dq.peekFirst()) dq.pollFirst();
                start++;
            }
        }
        return ans;
    }
}
