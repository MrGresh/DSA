class Solution {
    public int maxSubarraySum(int[] arr, int k) {
        int ans = 0, curr = 0;
        for(int i=0;i<k;i++) curr += arr[i];
        ans = curr;
        for(int i=k;i<arr.length;i++) { 
            curr -= arr[i-k];
            curr += arr[i];
            ans = Math.max(ans, curr);
        }
        return ans;
    }
}

class Solution {
    public int maxSubarrayXOR(int[] arr, int k) {
        int ans = 0;
        for(int i=0;i<k;i++) ans ^= arr[i];
        int curr = ans;
        for(int i=k;i<arr.length;i++) {
            curr ^= arr[i-k];
            curr ^= arr[i];
            ans = Math.max(ans, curr);
        }
        return ans;
    }
}
