class Solution {
    public String firstNonRepeating(String s) {
        int[] freq = new int[26];
        Queue<Character> q = new LinkedList<>();
        StringBuilder res = new StringBuilder();
        
        for(int i=0;i<s.length();i++) {
            freq[s.charAt(i)-'a']++;
            q.add(s.charAt(i));
            while(!q.isEmpty() && freq[q.peek()-'a']>1) q.poll();
            res.append(q.isEmpty() ? '#' : q.peek());
        }
        
        return res.toString();
    }
}
