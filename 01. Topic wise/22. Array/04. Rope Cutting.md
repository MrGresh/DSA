class Solution {
    public ArrayList<Integer> RopeCutting(int arr[]) {
        ArrayList<Integer> result = new ArrayList<>();
        int n = arr.length;
        Arrays.sort(arr);

        for (int i = 1; i < n; i++) {
            if (arr[i] != arr[i - 1]) {
                result.add(n - i);
            }
        }

        return result;
    }
}
