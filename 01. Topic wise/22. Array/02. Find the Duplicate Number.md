### 1 <= n <= 10^5
### nums.length == n + 1
### 1 <= nums[i] <= n
```java
class Solution {
    public int findDuplicate(int[] nums) {
        int i = 0;
        while (i < nums.length) {
            int correctIndex = nums[i] - 1;
            if (nums[i] != nums[correctIndex]) {
                int temp = nums[i];
                nums[i] = nums[correctIndex];
                nums[correctIndex] = temp;
            } else i++;
        }
        for (int j = 0; j < nums.length; j++) {
            if (nums[j] != j + 1) return nums[j];
        }
        return -1;
    }
}
```
