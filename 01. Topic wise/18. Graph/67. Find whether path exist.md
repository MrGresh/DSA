### BFS
```java
class Solution {
    public boolean is_Possible(int[][] grid) {
        int n = grid.length;
        Queue<int[]> q = new LinkedList<>();
        boolean[][] visited = new boolean[n][n];

        // 1. Find the source (1)
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    q.add(new int[]{i, j});
                    visited[i][j] = true;
                    break;
                }
            }
        }

        // Direction vectors for Up, Down, Left, Right
        int[] dr = {-1, 1, 0, 0};
        int[] dc = {0, 0, -1, 1};

        // 2. BFS
        while (!q.isEmpty()) {
            int[] curr = q.poll();
            int r = curr[0];
            int c = curr[1];

            for (int i = 0; i < 4; i++) {
                int nr = r + dr[i];
                int nc = c + dc[i];

                // Check boundaries and if cell is not a wall (0)
                if (nr >= 0 && nr < n && nc >= 0 && nc < n && 
                    grid[nr][nc] != 0 && !visited[nr][nc]) {
                    
                    // Found the destination!
                    if (grid[nr][nc] == 2) return true;

                    visited[nr][nc] = true;
                    q.add(new int[]{nr, nc});
                }
            }
        }

        return false;
    }
}
```

### DFS
```java
class Solution {
    public boolean is_Possible(int[][] grid) {
        int n = grid.length;
        boolean[][] visited = new boolean[n][n];

        // 1. Find the source (1)
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    return dfs(grid, i, j, visited, n);
                }
            }
        }
        return false;
    }

    private boolean dfs(int[][] grid, int r, int c, boolean[][] visited, int n) {
        // Base case: Out of bounds, hit a wall (0), or already visited
        if (r < 0 || r >= n || c < 0 || c >= n || grid[r][c] == 0 || visited[r][c]) {
            return false;
        }

        // Base case: Found the destination (2)!
        if (grid[r][c] == 2) return true;

        // Mark current cell as visited
        visited[r][c] = true;

        // Recursively check all 4 directions
        // If any direction returns true, we've found a path
        if (dfs(grid, r + 1, c, visited, n) || // Down
            dfs(grid, r - 1, c, visited, n) || // Up
            dfs(grid, r, c + 1, visited, n) || // Right
            dfs(grid, r, c - 1, visited, n)) { // Left
            return true;
        }

        return false;
    }
}
```
