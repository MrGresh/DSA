class Solution {
    public boolean is_Possible(int[][] grid) {
        int n = grid.length;
        Queue<int[]> q = new LinkedList<>();
        boolean[][] visited = new boolean[n][n];

        // 1. Find the source (1)
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (grid[i][j] == 1) {
                    q.add(new int[]{i, j});
                    visited[i][j] = true;
                    break;
                }
            }
        }

        // Direction vectors for Up, Down, Left, Right
        int[] dr = {-1, 1, 0, 0};
        int[] dc = {0, 0, -1, 1};

        // 2. BFS
        while (!q.isEmpty()) {
            int[] curr = q.poll();
            int r = curr[0];
            int c = curr[1];

            for (int i = 0; i < 4; i++) {
                int nr = r + dr[i];
                int nc = c + dc[i];

                // Check boundaries and if cell is not a wall (0)
                if (nr >= 0 && nr < n && nc >= 0 && nc < n && 
                    grid[nr][nc] != 0 && !visited[nr][nc]) {
                    
                    // Found the destination!
                    if (grid[nr][nc] == 2) return true;

                    visited[nr][nc] = true;
                    q.add(new int[]{nr, nc});
                }
            }
        }

        return false;
    }
}
