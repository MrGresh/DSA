class Solution {
    int nodeLevel(int V, ArrayList<ArrayList<Integer>> adj, int X) {
        Queue<Integer> q = new LinkedList<>();
        boolean[] vis = new boolean[V];
        q.add(0);
        vis[0] = true;
        int level = -1;
        while(!q.isEmpty()) {
            level++;
            int size = q.size();
            while(size-->0) {
                int node = q.poll();
                for(int nei: adj.get(node)) {
                    if(nei==X) return level+1;
                    if(!vis[nei]) {
                        vis[nei] = true;
                        q.add(nei);
                    }
                }
            }
        }
        return -1;
    }
}
