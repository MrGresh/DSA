### BFS (The Standard)
```java
import java.util.*;

class UndirectedShortestBFS {
    public int findShortestCycle(int n, List<List<Integer>> adj) {
        int minCycle = Integer.MAX_VALUE;

        for (int i = 0; i < n; i++) {
            int[] dist = new int[n];
            Arrays.fill(dist, -1);
            int[] parent = new int[n];
            Arrays.fill(parent, -1);

            Queue<Integer> q = new LinkedList<>();
            q.add(i);
            dist[i] = 0;

            while (!q.isEmpty()) {
                int u = q.poll();

                for (int v : adj.get(u)) {
                    if (dist[v] == -1) {
                        // First time seeing node v
                        dist[v] = dist[u] + 1;
                        parent[v] = u;
                        q.add(v);
                    } else if (v != parent[u]) {
                        // v is already visited AND it's not the parent of u
                        // This means we found a cycle!
                        minCycle = Math.min(minCycle, dist[u] + dist[v] + 1);
                        
                        // Optimization: Since BFS finds shortest paths, 
                        // any other cycle from this source 'i' will be >= this one.
                        // We can't return immediately because a different source
                        // might find an even smaller cycle elsewhere.
                    }
                }
            }
            if (minCycle == 3) return 3; // Absolute minimum for simple graphs
        }
        return minCycle == Integer.MAX_VALUE ? -1 : minCycle;
    }
}
```
### DFS (The Alternative)
```java
class UndirectedShortestDFS {
    int minCycle = Integer.MAX_VALUE;

    public int findShortestCycle(int n, List<List<Integer>> adj) {
        for (int i = 0; i < n; i++) {
            int[] dist = new int[n];
            Arrays.fill(dist, -1);
            dfs(i, -1, 0, dist, adj);
            if (minCycle == 3) return 3;
        }
        return minCycle == Integer.MAX_VALUE ? -1 : minCycle;
    }

    private void dfs(int u, int p, int d, int[] dist, List<List<Integer>> adj) {
        // Pruning: if current depth is already too long, stop
        if (d >= minCycle) return;

        dist[u] = d;

        for (int v : adj.get(u)) {
            if (v == p) continue; // Critical: Ignore the edge back to the parent

            if (dist[v] != -1) {
                // Cycle detected! Node v was already visited in this path
                int cycleLen = d - dist[v] + 1;
                minCycle = Math.min(minCycle, cycleLen);
            } else {
                dfs(v, u, d + 1, dist, adj);
            }
        }

        // Backtrack: Reset distance so node u can be part of 
        // a different, potentially shorter cycle from another path.
        dist[u] = -1;
    }
}
```
