class Edge {
    int src, dest, weight;
    Edge(int s, int d, int w) {
        src = s;
        dest = d;
        weight = w;
    }
}

class KruskalMST {
    public List<Edge> getMST(int V, int[][] edges) {
        List<Edge> allEdges = new ArrayList<>();
        for (int[] e : edges) {
            allEdges.add(new Edge(e[0], e[1], e[2]));
        }

        // Sort edges by weight
        Collections.sort(allEdges, Comparator.comparingInt(e -> e.weight));

        // Initialize Disjoint Set
        DisjointSet ds = new DisjointSet(V);

        List<Edge> mstEdges = new ArrayList<>();

        // Process edges in increasing order
        for (Edge e : allEdges) {
            if (ds.unionBySize(e.src, e.dest)) { // we can also use unionByRank
                mstEdges.add(e);
            }

            // Stop early if MST has V-1 edges
            if (mstEdges.size() == V - 1) break;
        }

        return mstEdges;
    }
}
