class Solution {
    public int findMotherVertex(int V, ArrayList<ArrayList<Integer>> adj) {
        boolean[] visited = new boolean[V];
        int lastFinished = 0;

        // Phase 1: Find the potential candidate
        for (int i = 0; i < V; i++) {
            if (!visited[i]) {
                dfs(i, adj, visited);
                lastFinished = i; 
            }
        }

        // Phase 2: Verify if lastFinished can actually reach all nodes
        Arrays.fill(visited, false);
        int count = dfs(lastFinished, adj, visited);

        return (count == V) ? lastFinished : -1;
    }

    private int dfs(int u, ArrayList<ArrayList<Integer>> adj, boolean[] visited) {
        visited[u] = true;
        int count = 1;
        for (int v : adj.get(u)) {
            if (!visited[v]) {
                count += dfs(v, adj, visited);
            }
        }
        return count;
    }
}
