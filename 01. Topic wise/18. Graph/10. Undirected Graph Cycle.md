```java
class Solution {
    public boolean isCycle(int V, int[][] edges) {
        // 1. Build Adjacency List
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) adj.add(new ArrayList<>());
        for (int[] edge : edges) {
            adj.get(edge[0]).add(edge[1]);
            adj.get(edge[1]).add(edge[0]);
        }

        boolean[] visited = new boolean[V];

        // 2. Check every component
        for (int i = 0; i < V; i++) {
            if (!visited[i]) {
                if (dfs(i, -1, adj, visited)) return true;
            }
        }
        return false;
    }

    private boolean dfs(int curr, int parent, List<List<Integer>> adj, boolean[] visited) {
        visited[curr] = true;

        for (int neighbor : adj.get(curr)) {
            if (!visited[neighbor]) {
                if (dfs(neighbor, curr, adj, visited)) return true;
            } 
            // If neighbor is visited and not the parent, we found a cycle
            else if (neighbor != parent) {
                return true;
            }
        }
        return false;
    }
}
```

// BFS
```java
class Solution {
    class Node {
        int node; // current node
        int parent; // parent node
        Node(int f, int s) { node = f; parent = s; }
    }

    public boolean isCycle(int V, int[][] edges) {
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) adj.add(new ArrayList<>());
        for (int[] edge : edges) {
            adj.get(edge[0]).add(edge[1]);
            adj.get(edge[1]).add(edge[0]);
        }

        boolean[] visited = new boolean[V];

        for (int i = 0; i < V; i++) {
            if (!visited[i]) {
                if (bfs(i, adj, visited)) return true;
            }
        }
        return false;
    }

    private boolean bfs(int start, List<List<Integer>> adj, boolean[] visited) {
        Queue<Node> q = new LinkedList<>();
        q.add(new Node(start, -1));
        visited[start] = true;

        while (!q.isEmpty()) {
            int curr = q.peek().node;
            int parent = q.peek().parent;
            q.remove();

            for (int neighbor : adj.get(curr)) {
                if (!visited[neighbor]) {
                    visited[neighbor] = true;
                    q.add(new Node(neighbor, curr));
                } 
                // If visited and not the parent, it's a cycle
                else if (parent != neighbor) {
                    return true;
                }
            }
        }
        return false;
    }
}
```
