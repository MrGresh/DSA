### DSU
```java
class Solution {
    public boolean containsCycle(char[][] grid) {
        int rows = grid.length, cols = grid[0].length;
        DSU dsu = new DSU(rows*cols);
        for(int i=0;i<rows;i++) {
            for(int j=0;j<cols;j++) {
                int curr = i*cols + j;
                if(j+1<cols && grid[i][j]==grid[i][j+1]) {
                    int next = (i*cols) + j + 1;
                    if(dsu.find(curr)==dsu.find(next)) return true;
                    dsu.union(curr, next);
                }
                if(i+1<rows && grid[i][j]==grid[i+1][j]) {
                    int next = ((i+1)*cols) + j;
                    if(dsu.find(curr)==dsu.find(next)) return true;
                    dsu.union(curr, next);
                }
            }
        }
        return false;
    }
    class DSU {
        int[] parent;
        DSU(int n) {
            parent = new int[n];
            for(int i=0;i<n;i++) parent[i] = i;
        }
        int find(int x) {
            if(x!=parent[x]) parent[x] = find(parent[x]);
            return parent[x];
        }
        void union(int x, int y) {
            int px = find(x), py = find(y);
            parent[px] = py;
        }
    }
}
```

### Backtracking
```java
class Solution {
    public boolean containsCycle(char[][] grid) {
        int rows = grid.length, cols = grid[0].length;
        boolean[][] vis = new boolean[rows][cols];
        for(int i=0;i<rows;i++) {
            for(int j=0;j<cols;j++) {
                if(cycle(grid[i][j], i, j, -1, -1, vis, grid)) return true;
            }
        }
        return false;
    }
    boolean cycle(char ch, int i, int j, int pi, int pj, boolean[][] vis, char[][] grid) {
        int rows = grid.length, cols = grid[0].length;
        if(vis[i][j]) return true;
        vis[i][j] = true;
        if(i-1>=0 && grid[i-1][j]==ch && (i-1!=pi) && cycle(ch, i-1, j, i, j, vis, grid)) return true;
        if(j-1>=0 && grid[i][j-1]==ch && (j-1!=pj) && cycle(ch, i, j-1, i, j, vis, grid)) return true;
        if(i+1<rows && grid[i+1][j]==ch && (i+1!=pi) && cycle(ch, i+1, j, i, j, vis, grid)) return true;
        if(j+1<cols && grid[i][j+1]==ch && (j+1!=pj) && cycle(ch, i, j+1, i, j, vis, grid)) return true;
        vis[i][j] = false;
        return false;
    }
}
```
