class Solution {
    public int[] shortestPath(int V, int[][] edges, int src) {
        List<List<Integer>> adj = new ArrayList<>();
        for(int i=0;i<V;i++) adj.add(new ArrayList<>());
        for(int[] e: edges) {
            adj.get(e[0]).add(e[1]);
            adj.get(e[1]).add(e[0]);
        }
        int[] dist = new int[V];
        Arrays.fill(dist, -1);
        Queue<Integer> q = new LinkedList<>();
        boolean[] vis = new boolean[V];
        q.add(src);
        vis[src] = true;
        int dis = -1;
        while(!q.isEmpty()) {
            ++dis;
            int size = q.size();
            for(int i=1;i<=size;i++) {
                Integer node = q.poll();
                dist[node] = dis;
                for(Integer nei: adj.get(node)) {
                    if(!vis[nei]) {
                        vis[nei] = true;
                        q.add(nei);
                    }
                }
            }
        }
        return dist;
    }
}
