### DFS (The Standard)
```java
class UndirectedLongestDFS {
    int maxCycle = -1;

    public int findLongestCycle(int n, List<List<Integer>> adj) {
        // We use an array to store the depth of the current path
        int[] dist = new int[n];
        Arrays.fill(dist, -1);

        for (int i = 0; i < n; i++) {
            // Start DFS from every node to ensure we cover all components
            dfs(i, -1, 0, dist, adj);
        }
        return maxCycle;
    }

    private void dfs(int u, int p, int d, int[] dist, List<List<Integer>> adj) {
        dist[u] = d; // Record depth in current path

        for (int v : adj.get(u)) {
            if (v == p) continue; // Ignore immediate parent

            if (dist[v] != -1) {
                // Back-edge detected! Node v is already in the current path.
                // Cycle length = Current depth - Start depth of v + 1
                maxCycle = Math.max(maxCycle, d - dist[v] + 1);
            } else {
                dfs(v, u, d + 1, dist, adj);
            }
        }

        // BACKTRACK: Reset the distance so this node can be reused 
        // in a different branch of the search.
        dist[u] = -1;
    }
}
```
