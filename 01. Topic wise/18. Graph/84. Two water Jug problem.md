class Solution {
    Integer[][] memo;
    boolean[][] visited; // To prevent StackOverflow from infinite cycles

    public int minSteps(int m, int n, int d) {
        if (d > m && d > n) return -1;
        // Optimization: Bézout's Identity (Check GCD)
        if (d % gcd(m, n) != 0) return -1;

        memo = new Integer[m + 1][n + 1];
        visited = new boolean[m + 1][n + 1];
        
        int result = rec(0, 0, m, n, d);
        return result >= 1e7 ? -1 : result;
    }

    int rec(int j1, int j2, int m, int n, int d) {
        if (j1 == d || j2 == d) return 0;
        
        if (memo[j1][j2] != null) return memo[j1][j2];
        if (visited[j1][j2]) return 10000000; // Return "Infinity" if we hit a cycle

        visited[j1][j2] = true;
        int steps = 10000000; // Using a large number instead of MAX_VALUE to avoid overflow

        // 1. Empty Jugs
        if(j1>0) steps = Math.min(steps, 1 + rec(0, j2, m, n, d));
        if(j2>0) steps = Math.min(steps, 1 + rec(j1, 0, m, n, d));
        
        // 2. Fill Jugs
        if(j1<m) steps = Math.min(steps, 1 + rec(m, j2, m, n, d));
        if(j2<n) steps = Math.min(steps, 1 + rec(j1, n, m, n, d));
        
        // 3. Pour j1 -> j2
        if(j2<n) {
            int pour1to2 = Math.min(j1, n - j2);
            steps = Math.min(steps, 1 + rec(j1 - pour1to2, j2 + pour1to2, m, n, d));
        }
        
        // 4. Pour j2 -> j1
        if(j1<m) {
            int pour2to1 = Math.min(j2, m - j1);
            steps = Math.min(steps, 1 + rec(j1 + pour2to1, j2 - pour2to1, m, n, d));
        }
        
        return memo[j1][j2] = steps;
    }

    int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
}

### BFS

class Solution {
    public int minSteps(int m, int n, int d) {
        // 1. Math Guard: d must be a multiple of GCD(m, n)
        if (d > Math.max(m, n) || d % gcd(m, n) != 0) return -1;

        // 2. Queue for BFS and a Visited array to prevent cycles
        Queue<int[]> queue = new LinkedList<>();
        boolean[][] visited = new boolean[m + 1][n + 1];

        // Start state: both jugs empty, 0 steps
        queue.offer(new int[]{0, 0, 0}); 
        visited[0][0] = true;

        while (!queue.isEmpty()) {
            int[] curr = queue.poll();
            int j1 = curr[0], j2 = curr[1], steps = curr[2];

            // Target reached!
            if (j1 == d || j2 == d) return steps;

            // 3. Define all 6 possible transitions
            int[][] nextMoves = {
                {m, j2}, // Fill J1
                {j1, n}, // Fill J2
                {0, j2}, // Empty J1
                {j1, 0}, // Empty J2
                {j1 - Math.min(j1, n - j2), j2 + Math.min(j1, n - j2)}, // Pour J1 -> J2
                {j1 + Math.min(j2, m - j1), j2 - Math.min(j2, m - j1)}  // Pour J2 -> J1
            };

            for (int[] move : nextMoves) {
                if (!visited[move[0]][move[1]]) {
                    visited[move[0]][move[1]] = true;
                    queue.offer(new int[]{move[0], move[1], steps + 1});
                }
            }
        }
        return -1;
    }

    private int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
}

class Solution {
    public int minSteps(int m, int n, int d) {
        if (d > Math.max(m, n) || d % gcd(m, n) != 0) return -1;
        
        // The answer is the minimum of starting by filling m or starting by filling n
        return Math.min(simulate(m, n, d), simulate(n, m, d));
    }

    private int simulate(int fromCap, int toCap, int d) {
        int from = fromCap; // Fill the first jug
        int to = 0;
        int steps = 1; // Step 1: Fill 'from' jug

        while (from != d && to != d) {
            // How much can we pour?
            int temp = Math.min(from, toCap - to);

            // Pour from 'from' to 'to'
            to += temp;
            from -= temp;
            steps++;

            if (from == d || to == d) break;

            // If 'from' is empty, fill it
            if (from == 0) {
                from = fromCap;
                steps++;
            }
            // If 'to' is full, empty it
            else if (to == toCap) {
                to = 0;
                steps++;
            }
        }
        return steps;
    }

    private int gcd(int a, int b) {
        return b == 0 ? a : gcd(b, a % b);
    }
}
