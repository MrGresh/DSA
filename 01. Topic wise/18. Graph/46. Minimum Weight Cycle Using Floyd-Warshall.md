### Undirected Graphs

```java
public int findMinCycleUndirected(int V, int[][] edges) {
    int[][] dist = new int[V][V];
    int[][] adj = new int[V][V]; // original weights
    for (int i = 0; i < V; i++) {
        Arrays.fill(dist[i], 1_000_000_000);
        Arrays.fill(adj[i], 1_000_000_000);
        dist[i][i] = 0;
    }

    for (int[] e : edges) {
        int u = e[0], v = e[1], w = e[2];
        dist[u][v] = dist[v][u] = adj[u][v] = adj[v][u] = Math.min(dist[u][v], w);
    }

    int minCycle = 1_000_000_000;

    for (int k = 0; k < V; k++) {
        // 1. Check for cycles involving node k as the highest-indexed node
        for (int i = 0; i < k; i++) {
            for (int j = i + 1; j < k; j++) {
                // Path i->j (using nodes < k) + edge j-k + edge k-i
                minCycle = Math.min(minCycle, dist[i][j] + adj[i][k] + adj[k][j]);
            }
        }

        // 2. Standard update
        for (int i = 0; i < V; i++) {
            for (int j = 0; j < V; j++) {
                dist[i][j] = Math.min(dist[i][j], dist[i][k] + dist[k][j]);
            }
        }
    }

    return (minCycle >= 1_000_000_000) ? -1 : minCycle;
}
```

### Directed Graphs
```java
public int findMinCycleDirected(int V, int[][] edges) {
    int[][] dist = new int[V][V];
    for (int[] row : dist) Arrays.fill(row, 1_000_000_000);

    // Initial edges
    for (int[] e : edges) {
        dist[e[0]][e[1]] = Math.min(dist[e[0]][e[1]], e[2]);
    }

    // Standard Floyd-Warshall
    for (int k = 0; k < V; k++) {
        for (int i = 0; i < V; i++) {
            for (int j = 0; j < V; j++) {
                dist[i][j] = Math.min(dist[i][j], dist[i][k] + dist[k][j]);
            }
        }
    }

    // Shortest cycle is the minimum value on the diagonal
    int minCycle = 1_000_000_000;
    for (int i = 0; i < V; i++) {
        minCycle = Math.min(minCycle, dist[i][i]);
    }

    return (minCycle >= 1_000_000_000) ? -1 : minCycle;
    // If there is no cycle, the diagonal values dist[i][i] will simply remain INF. If the final minCycle value is still INF, you know for certain the graph is a DAG (Directed Acyclic Graph).
}
```
