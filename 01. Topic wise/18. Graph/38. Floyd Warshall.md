class Solution {
    public void floydWarshall(int[][] dist) {
        int v = dist.length;
        int MAX = 100000000;
        for(int k=0;k<v;k++) {
            for(int i=0;i<v;i++) {
                for(int j=0;j<v;j++) {
                    if(dist[i][k]<MAX && dist[k][j]<MAX) {
                        dist[i][j] = Math.min(dist[i][j], dist[i][k]+dist[k][j]);
                    }
                }
            }
        }
    }
}
