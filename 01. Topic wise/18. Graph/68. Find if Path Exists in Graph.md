### BFS
```java
class Solution {
    public boolean validPath(int n, int[][] edges, int source, int destination) {
        // Base case: if source and destination are the same
        if (source == destination) return true;

        // 1. Build Adjacency List
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            adj.add(new ArrayList<>());
        }
        for (int[] edge : edges) {
            adj.get(edge[0]).add(edge[1]);
            adj.get(edge[1]).add(edge[0]);
        }

        // 2. BFS Initialization
        Queue<Integer> queue = new LinkedList<>();
        boolean[] visited = new boolean[n];

        queue.add(source);
        visited[source] = true;

        // 3. BFS Traversal
        while (!queue.isEmpty()) {
            int curr = queue.poll();
            
            // Check every neighbor of the current node
            for (int neighbor : adj.get(curr)) {
                if (neighbor == destination) return true;
                
                if (!visited[neighbor]) {
                    visited[neighbor] = true;
                    queue.add(neighbor);
                }
            }
        }

        return false;
    }
}
```

### DFS
```java
class Solution {
    public boolean validPath(int n, int[][] edges, int source, int destination) {
        if (source == destination) return true;

        // 1. Build Adjacency List
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            adj.add(new ArrayList<>());
        }
        for (int[] edge : edges) {
            adj.get(edge[0]).add(edge[1]);
            adj.get(edge[1]).add(edge[0]);
        }

        // 2. DFS Initialization
        boolean[] visited = new boolean[n];
        return dfs(source, destination, adj, visited);
    }

    private boolean dfs(int curr, int dest, List<List<Integer>> adj, boolean[] visited) {
        // Base case: Found the destination
        if (curr == dest) return true;
        
        visited[curr] = true;

        // Visit all unvisited neighbors
        for (int neighbor : adj.get(curr)) {
            if (!visited[neighbor]) {
                if (dfs(neighbor, dest, adj, visited)) {
                    return true;
                }
            }
        }

        return false;
    }
}
```
