### With adjList
#### Prim's Algo
```java
class Solution {
    class Edge {
        int to;
        int weight;
        Edge(int to, int weight) {
            this.to = to;
            this.weight = weight;
        }
    }

    public int minCost(int[][] houses) {
        int n = houses.length;
        if (n <= 1) return 0;

        List<List<Edge>> adj = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            adj.add(new ArrayList<>());
        }

        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int dist = Math.abs(houses[i][0] - houses[j][0]) + Math.abs(houses[i][1] - houses[j][1]);
                
                adj.get(i).add(new Edge(j, dist));
                adj.get(j).add(new Edge(i, dist));
            }
        }

        // Prim's Algo
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0] - b[0]);
        boolean[] vis = new boolean[n];
        int totalCost = 0;
        int edgesConnected = 0;

        // Start from house 0
        pq.offer(new int[]{0, 0}); // {cost, house_index}

        while (edgesConnected < n && !pq.isEmpty()) {
            int[] curr = pq.poll();
            int weight = curr[0];
            int u = curr[1];

            if (vis[u]) continue;
            vis[u] = true;
            
            totalCost += weight;
            edgesConnected++;

            for (Edge edge : adj.get(u)) {
                if (!vis[edge.to]) {
                    pq.offer(new int[]{edge.weight, edge.to});
                }
            }
        }

        return totalCost;
    }
}
```

### With Edge List
#### Kruskal's Algo
```java
class Solution {
    // Disjoint Set Union (DSU) helper class
    class DSU {
        int[] parent;
        public DSU(int n) {
            parent = new int[n];
            for (int i = 0; i < n; i++) parent[i] = i;
        }
        public int find(int i) {
            if (parent[i] == i) return i;
            return parent[i] = find(parent[i]); // Path compression
        }
        public boolean union(int i, int j) {
            int rootI = find(i);
            int rootJ = find(j);
            if (rootI != rootJ) {
                parent[rootI] = rootJ;
                return true;
            }
            return false;
        }
    }

    class Edge {
        int u, v, weight;
        Edge(int u, int v, int weight) {
            this.u = u;
            this.v = v;
            this.weight = weight;
        }
    }

    public int minCost(int[][] houses) {
        int n = houses.length;
        List<Edge> allEdges = new ArrayList<>();

        // 1. Generate all O(n^2) edges
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                int dist = Math.abs(houses[i][0] - houses[j][0]) + 
                           Math.abs(houses[i][1] - houses[j][1]);
                allEdges.add(new Edge(i, j, dist));
            }
        }

        // 2. Sort edges by weight: O(E log E) where E = n^2
        Collections.sort(allEdges, (a, b) -> a.weight - b.weight);

        // 3. Apply Kruskal's using DSU
        DSU dsu = new DSU(n);
        int totalCost = 0;
        int edgesCount = 0;

        for (Edge edge : allEdges) {
            if (dsu.union(edge.u, edge.v)) {
                totalCost += edge.weight;
                edgesCount++;
                if (edgesCount == n - 1) break; // MST complete
            }
        }

        return totalCost;
    }
}
```

### Efficient
**Note**: No need to prepare adjList because we know each house can be connected to any other.
#### Prim's Algo
```java
class Solution {
    public int minCost(int[][] houses) {
        int n = houses.length;
        // PriorityQueue stores [cost, house_index], sorted by cost
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0] - b[0]);
        boolean[] vis = new boolean[n];
        
        // Start with the first house (cost 0 to connect to itself)
        pq.offer(new int[]{0, 0});
        
        int totalCost = 0;
        int edgesConnected = 0;
        
        while (edgesConnected < n) {
            int[] curr = pq.poll();
            int weight = curr[0];
            int u = curr[1];
            
            if (vis[u]) continue;
            vis[u] = true;
            
            totalCost += weight;
            edgesConnected++;
            
            // Check all neighbors (every other house)
            for (int v = 0; v < n; v++) {
                if (!vis[v]) {
                    int dist = Math.abs(houses[u][0] - houses[v][0]) + Math.abs(houses[u][1] - houses[v][1]);
                    pq.offer(new int[]{dist, v});
                }
            }
        }
        
        return totalCost;
    }
}
```
