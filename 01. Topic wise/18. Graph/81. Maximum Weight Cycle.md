### DFS + Backtracking
```java
class MaxWeightCycle {
    long maxWeight = -1;

    public long findMaxWeightCycle(int n, List<List<int[]>> adj) {
        // dist[i] stores the cumulative weight from the source to node i
        // Using Double.MIN_VALUE or a special flag to represent "not in current path"
        long[] dist = new long[n];
        boolean[] inPath = new boolean[n];
        Arrays.fill(dist, -1);

        for (int i = 0; i < n; i++) {
            backtrack(i, 0, adj, dist, inPath);
        }

        return maxWeight;
    }

    private void backtrack(int u, long currentW, List<List<int[]>> adj, long[] dist, boolean[] inPath) {
        dist[u] = currentW;
        inPath[u] = true;

        for (int[] edge : adj.get(u)) {
            int v = edge[0];
            int weight = edge[1];

            if (inPath[v]) {
                // Cycle detected! 
                // Weight = (Weight reached at u) - (Weight reached at v) + edge(u,v)
                long cycleWeight = (dist[u] - dist[v]) + weight;
                maxWeight = Math.max(maxWeight, cycleWeight);
            } else if (dist[v] == -1) { 
                // If not visited at all in this specific global search
                backtrack(v, currentW + weight, adj, dist, inPath);
            }
        }

        // BACKTRACK: Reset state so node u can be part of other paths
        inPath[u] = false;
        dist[u] = -1; 
    }
}
```
