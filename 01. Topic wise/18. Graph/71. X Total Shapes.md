### Same as No. of Islands
```java
class Solution {
    public int xShape(char[][] grid) {
        int ans = 0;
        for(int i=0;i<grid.length;i++) {
            for(int j=0;j<grid[0].length;j++) {
                if(grid[i][j]=='X') {
                    ans++;
                    rec(grid, i, j);
                }
            }
        }
        return ans;
    }
    void rec(char[][] grid, int i, int j) {
        grid[i][j] = '#';
        if(i-1>=0 && grid[i-1][j]=='X') rec(grid, i-1, j);
        if(i+1<grid.length && grid[i+1][j]=='X') rec(grid, i+1, j);
        if(j-1>=0 && grid[i][j-1]=='X') rec(grid, i, j-1);
        if(j+1<grid[0].length && grid[i][j+1]=='X') rec(grid, i, j+1);
    }
}
```
