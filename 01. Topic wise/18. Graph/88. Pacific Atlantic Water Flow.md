class Solution {
    public List<List<Integer>> pacificAtlantic(int[][] heights) {
        int rows = heights.length;
        int cols = heights[0].length;
        
        boolean[][] pacific = new boolean[rows][cols];
        boolean[][] atlantic = new boolean[rows][cols];
        
        // Start DFS from the borders
        for (int i = 0; i < rows; i++) {
            dfs(heights, i, 0, pacific); // Left edge (Pacific)
            dfs(heights, i, cols - 1, atlantic); // Right edge (Atlantic)
        }
        
        for (int j = 0; j < cols; j++) {
            dfs(heights, 0, j, pacific); // Top edge (Pacific)
            dfs(heights, rows - 1, j, atlantic); // Bottom edge (Atlantic)
        }
        
        List<List<Integer>> result = new ArrayList<>();
        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                // If cell can reach both oceans
                if (pacific[i][j] && atlantic[i][j]) {
                    result.add(Arrays.asList(i, j));
                }
            }
        }
        return result;
    }
    
    private void dfs(int[][] heights, int r, int c, boolean[][] visited) {
        visited[r][c] = true;
        
        if(r+1<heights.length && !visited[r+1][c] && heights[r+1][c] >= heights[r][c]) dfs(heights, r + 1, c, visited);
        if(r-1>=0 && !visited[r-1][c] && heights[r-1][c] >= heights[r][c]) dfs(heights, r - 1, c, visited);
        if(c+1<heights[0].length && !visited[r][c+1] && heights[r][c+1] >= heights[r][c]) dfs(heights, r, c + 1, visited);
        if(c-1>=0 && !visited[r][c-1] && heights[r][c-1] >= heights[r][c]) dfs(heights, r, c - 1, visited);
    }
}
