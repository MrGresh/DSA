class Solution {
    public ArrayList<Integer> minHeightRoot(int V, int[][] edges) {
        ArrayList<Integer> res = new ArrayList<>();
        if (V <= 0) return res;
        
        // Base case: If there's only one node, it's the root
        if (V == 1) {
            res.add(0);
            return res;
        }

        // 1. Build Adjacency List and Degree Array
        List<Set<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) adj.add(new HashSet<>());
        int[] degree = new int[V];
        
        for (int[] edge : edges) {
            int u = edge[0];
            int v = edge[1];
            adj.get(u).add(v);
            adj.get(v).add(u);
            degree[u]++;
            degree[v]++;
        }

        // 2. Initialize Queue with all current leaves
        Queue<Integer> queue = new LinkedList<>();
        for (int i = 0; i < V; i++) {
            if (degree[i] == 1) {
                queue.add(i);
            }
        }

        // 3. Trim the leaves layer by layer
        int remainingNodes = V;
        while (remainingNodes > 2) {
            int leafCount = queue.size();
            remainingNodes -= leafCount;
            
            for (int i = 0; i < leafCount; i++) {
                int leaf = queue.poll();
                
                // Find the neighbor of the leaf and remove the leaf connection
                for (int neighbor : adj.get(leaf)) {
                    degree[neighbor]--;
                    if (degree[neighbor] == 1) {
                        queue.add(neighbor);
                    }
                }
            }
        }

        // The remaining nodes in the queue are the centers
        res.addAll(queue);
        return res;
    }
}
