class Solution {
    public static int primeDigits(int n) {
        Queue<Integer> q = new LinkedList<>();
        q.add(0);
        int count = -1;
        while(true) {
            count++;
            int num = q.poll();
            if(count==n) return num;
            q.add(num*10+2);
            q.add(num*10+3);
            q.add(num*10+5);
            q.add(num*10+7);
        }
    }
}
