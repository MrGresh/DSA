class Solution {
    class Edge {
        int to;
        double prob;

        Edge(int to, double prob) {
            this.to = to;
            this.prob = prob;
        }
    }

    public double maxProbability(int n, int[][] edges, double[] succProb, int start, int end) {
        List<List<Edge>> graph = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            graph.add(new ArrayList<>());
        }

        for (int i = 0; i < edges.length; i++) {
            int u = edges[i][0];
            int v = edges[i][1];
            double p = succProb[i];
            graph.get(u).add(new Edge(v, p));
            graph.get(v).add(new Edge(u, p));
        }

        // Distance array (storing max probabilities)
        double[] maxProb = new double[n];
        maxProb[start] = 1.0;

        // (Max-Heap), Not Mini-Heap
        PriorityQueue<Edge> pq = new PriorityQueue<>((a, b) -> Double.compare(b.prob, a.prob));
        pq.offer(new Edge(start, 1.0));

        while (!pq.isEmpty()) {
            Edge curr = pq.poll();
            int u = curr.to;
            double p = curr.prob;

            if (u == end) return p;

            if (p < maxProb[u]) continue;

            for (Edge e : graph.get(u)) {
                int v = e.to;
                double prob = e.prob;
                if (maxProb[u] * prob > maxProb[v]) {
                    maxProb[v] = maxProb[u] * prob;
                    pq.offer(new Edge(v, maxProb[v]));
                }
            }
        }

        return 0.0;
    }
}
