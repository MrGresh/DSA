public class codefile {
    static int solve(int n) {
        if (n == 1) return 1;
        Queue<String> q = new LinkedList<>();
        Set<Integer> vis = new HashSet<>();
        q.offer("1");
        while (!q.isEmpty()) {
            String curr = q.poll();
            int rem = mod(curr, n);
            if (rem == 0) return Integer.parseInt(curr);
            if (!vis.contains(rem)) {
                vis.add(rem);
                q.offer(curr + "0");
                q.offer(curr + "1");
            }
        }
        return -1;
    }
    static int mod(String num, int n) {
        int rem = 0;
        for (char c : num.toCharArray()) {
            rem = (rem * 10 + (c - '0')) % n;
        }
        return rem;
    }
}
