class Solution {
    int rows;
    int cols;
    public int numIslands(char[][] grid) {
        rows = grid.length;
        cols = grid[0].length;
        int ans = 0;
        for(int i=0;i<rows;i++) {
            for(int j=0;j<cols;j++) {
                if(grid[i][j] == '1') {
                    ans += 1;
                    makeAllConnectedNodesVisited(grid, i, j);
                }
            }
        }
        return ans;
    }
    void makeAllConnectedNodesVisited(char[][] grid, int i, int j) {
        grid[i][j] = '0';
        if(i-1>=0 && grid[i-1][j] == '1') makeAllConnectedNodesVisited(grid, i-1, j);
        if(i+1<rows && grid[i+1][j] == '1') makeAllConnectedNodesVisited(grid, i+1, j);
        if(j-1>=0 && grid[i][j-1] == '1') makeAllConnectedNodesVisited(grid, i, j-1);
        if(j+1<cols && grid[i][j+1] == '1') makeAllConnectedNodesVisited(grid, i, j+1);
    }
}

class Solution {
    char[][] grid;
    boolean[][] vis;
    int rows;
    int cols;
    public int numIslands(char[][] grid) {
        this.grid = grid;
        rows = grid.length;
        cols = grid[0].length;
        vis = new boolean[rows][cols];
        int ans = 0;
        for(int i=0;i<rows;i++) {
            for(int j=0;j<cols;j++) {
                if(!vis[i][j] && grid[i][j] == '1') {
                    ans += 1;
                    makeAllConnectedNodesVisited(i, j);
                }
            }
        }
        return ans;
    }
    void makeAllConnectedNodesVisited(int i, int j) {
        vis[i][j] = true;
        if(i-1>=0 && !vis[i-1][j] && grid[i-1][j] == '1') makeAllConnectedNodesVisited(i-1, j);
        if(i+1<rows && !vis[i+1][j] && grid[i+1][j] == '1') makeAllConnectedNodesVisited(i+1, j);
        if(j-1>=0 && !vis[i][j-1] && grid[i][j-1] == '1') makeAllConnectedNodesVisited(i, j-1);
        if(j+1<cols && !vis[i][j+1] && grid[i][j+1] == '1') makeAllConnectedNodesVisited(i, j+1);
    }
}
