### BFS
```java
class Solution {
    int minimumMultiplications(int[] arr, int start, int end) {
        Queue<Integer> q = new LinkedList<>();
        Set<Integer> vis = new HashSet<>();
        q.add(start);
        int dis=-1;
        while(!q.isEmpty()) {
            ++dis;
            int size = q.size();
            while(size-->0) {
                int n = q.poll();
                if(n==end) return dis;
                for(int x: arr) {
                    int newNum = (n*x)%100000;
                    if(vis.contains(newNum)) continue;
                    vis.add(newNum);
                    q.add(newNum);
                }
            }
        }
        return -1;
    }
}
```
