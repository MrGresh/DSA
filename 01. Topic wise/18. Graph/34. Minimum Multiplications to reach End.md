class Solution {
    int minimumMultiplications(int[] arr, int start, int end) {
        Queue<int[]> q = new LinkedList<>();
        Set<Integer> vis = new HashSet<>();
        q.add(new int[] {start, 0});
        while(!q.isEmpty()) {
            int[] pair = q.poll();
            if(pair[0]==end) return pair[1];
            for(int x: arr) {
                int newNum = (pair[0]*x)%100000;
                if(vis.contains(newNum)) continue;
                vis.add(newNum);
                q.add(new int[] {newNum, pair[1]+1});
            }
        }
        return -1;
    }
}
