**Note:** Each node has at most one outgoing edge.
```java
class Solution {
    public int longestCycle(int V, int[][] edges) {
        int maxCyclePath = -1;
        int[] timeVisited = new int[V]; // Stores the time step a node was visited
        boolean[] processed = new boolean[V]; // Global visited tracker
        
        // Convert edge list to adjacency array for O(1) lookup
        // Since each node has at most one outgoing edge, we use an array
        int[] adj = new int[V];
        for (int i = 0; i < V; i++) adj[i] = -1;
        for (int[] edge : edges) {
            adj[edge[0]] = edge[1];
        }

        for (int i = 0; i < V; i++) {
            if (processed[i]) continue;

            // Start a new traversal
            int curr = i;
            int step = 1;
            // Map to store (node -> timeStep) for the current path
            Map<Integer, Integer> currentPath = new HashMap<>();

            while (curr != -1 && !processed[curr]) {
                if (currentPath.containsKey(curr)) {
                    // Cycle detected!
                    maxCyclePath = Math.max(maxCyclePath, step - currentPath.get(curr));
                    break;
                }

                currentPath.put(curr, step);
                step++;
                int nextNode = adj[curr];
                
                // If we hit a node already processed in a previous outer loop, stop
                if (nextNode != -1 && processed[nextNode]) break;
                
                curr = nextNode;
            }

            // Mark all nodes in the current search path as processed
            for (int node : currentPath.keySet()) {
                processed[node] = true;
            }
        }

        return maxCyclePath;
    }
}
```
