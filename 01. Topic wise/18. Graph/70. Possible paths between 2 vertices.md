class Solution {
    public int countPaths(int V, ArrayList<ArrayList<Integer>> adj, int src, int dest) {
        return rec(adj, new boolean[V], src, dest);
    }
    int rec(ArrayList<ArrayList<Integer>> adj, boolean[] vis, int curr, int dest) {
        if(curr==dest) return 1;
        vis[curr] = true;
        int res = 0;
        for(int nei: adj.get(curr)) {
            if(!vis[nei]) res += rec(adj, vis, nei, dest);
        }
        vis[curr] = false;
        return res;
    }
}
