class Solution {
    public int minSwaps(int arr[]) {
        int n = arr.length;
        
        // Create a pair of {value, original_index}
        int[][] pair = new int[n][2];
        for (int i = 0; i < n; i++) {
            pair[i][0] = arr[i];
            pair[i][1] = i;
        }

        // Sort the pairs by value to see where elements SHOULD be
        Arrays.sort(pair, (a, b) -> Integer.compare(a[0], b[0]));

        boolean[] visited = new boolean[n];
        int totalSwaps = 0;

        for (int i = 0; i < n; i++) {
            // If already visited or already in the correct position, skip
            if (visited[i] || pair[i][1] == i) {
                continue;
            }

            // Find the size of the cycle
            int cycleSize = 0;
            int curr = i;
            
            while (!visited[curr]) {
                visited[curr] = true;
                // Move to the index where the current element was originally
                curr = pair[curr][1];
                cycleSize++;
            }

            // If a cycle exists, swaps needed is cycleSize - 1
            if (cycleSize > 0) {
                totalSwaps += (cycleSize - 1);
            }
        }

        return totalSwaps;
    }
}
