class Solution {
    int countDistinctIslands(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        Set<List<String>> distinctIslands = new HashSet<>();

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    List<String> shape = new ArrayList<>();
                    dfs(i, j, i, j, grid, shape);
                    distinctIslands.add(shape);
                }
            }
        }
        return distinctIslands.size();
    }

    private void dfs(int r, int c, int r0, int c0, int[][] grid, List<String> shape) {
        grid[r][c] = 0;
        shape.add((r - r0) + "," + (c - c0));

        int[] dr = {-1, 0, 1, 0};
        int[] dc = {0, 1, 0, -1};

        for (int i = 0; i < 4; i++) {
            int nr = r + dr[i];
            int nc = c + dc[i];

            if (nr >= 0 && nr < grid.length && nc >= 0 && nc < grid[0].length && 
                grid[nr][nc] == 1) {
                dfs(nr, nc, r0, c0, grid, shape);
            }
        }
    }
}
