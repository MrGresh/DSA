### DFS
This is the standard, most efficient way. We use a recursion stack to keep track of the current path and a dist array to record how deep each node is in that path.
```java
class DirectedLongestDFS {
    int maxCycle = -1;

    public int findLongestCycle(int V, int[][] edges) {
        // 1. Build Adjacency List
        List<List<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) adj.add(new ArrayList<>());
        for (int[] edge : edges) adj.get(edge[0]).add(edge[1]);

        boolean[] visited = new boolean[V];
        boolean[] recStack = new boolean[V];
        int[] dist = new int[V];

        // 2. Explore all components
        for (int i = 0; i < V; i++) {
            if (!visited[i]) {
                dfs(i, 0, adj, visited, recStack, dist);
            }
        }
        return maxCycle;
    }

    private void dfs(int u, int d, List<List<Integer>> adj, boolean[] visited, boolean[] recStack, int[] dist) {
        visited[u] = true;
        recStack[u] = true;
        dist[u] = d;

        for (int v : adj.get(u)) {
            if (!visited[v]) {
                dfs(v, d + 1, adj, visited, recStack, dist);
            } else if (recStack[v]) {
                // Back-edge found: Node v is already in the current path
                int currentCycleLen = dist[u] - dist[v] + 1;
                maxCycle = Math.max(maxCycle, currentCycleLen);
            }
        }

        // Backtrack: remove from path stack
        recStack[u] = false;
    }
}
```
