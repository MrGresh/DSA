**Note:** Each node has at most one outgoing edge.

```java
class Solution {
    public int longestCycle(int V, int[][] edges) {
        int maxCyclePath = -1;
        int[] timeVisited = new int[V]; // Stores the time step a node was visited
        boolean[] vis = new boolean[V]; // Global visited tracker
        
        // Convert edge list to adjacency array for O(1) lookup
        // Since each node has at most one outgoing edge, we use an array
        int[] adj = new int[V];
        for (int i = 0; i < V; i++) adj[i] = -1;
        for (int[] edge : edges) {
            adj[edge[0]] = edge[1];
        }

        for (int i = 0; i < V; i++) {
            if (!vis[i]) {
                // Start a new traversal
                int curr = i;
                int step = 1;
                // Map to store (node -> timeStep) for the current path
                Map<Integer, Integer> currentPath = new HashMap<>();
    
                while (curr != -1 && !vis[curr]) {
                    if (currentPath.containsKey(curr)) { // Cycle detected!
                        maxCyclePath = Math.max(maxCyclePath, step - currentPath.get(curr));
                        break;
                    }
                    currentPath.put(curr, step);
                    step++;
                    curr = adj[curr];
                }
    
                // Mark all nodes in the current search path as processed
                for (int node : currentPath.keySet()) vis[node] = true;
            }
        }

        return maxCyclePath;
    }
}
```

**Note:** Each node can have any number of edges.
```java
class Solution {
    public int longestCycle(int V, int[][] edges) {
        List<List<Integer>> adj = new ArrayList<>();
        for(int i=0;i<V;i++) adj.add(new ArrayList<>());
        for(int[] e: edges) adj.get(e[0]).add(e[1]);
        return find(V, adj);
    }
    int maxCycle = -1;
    public int find(int V, List<List<Integer>> adj) {
        boolean[] vis = new boolean[V];
        boolean[] recStack = new boolean[V];
        int[] dist = new int[V];
        for (int i = 0; i < V; i++) {
            if (!vis[i]) dfs(i, 0, adj, vis, recStack, dist);
        }
        return maxCycle;
    }
    private void dfs(int u, int d, List<List<Integer>> adj, boolean[] vis, boolean[] recStack, int[] dist) {
        vis[u] = true;
        recStack[u] = true;
        dist[u] = d;
        for (int v : adj.get(u)) {
            if (!vis[v]) dfs(v, d + 1, adj, vis, recStack, dist);
            else if (recStack[v]) maxCycle = Math.max(maxCycle, d - dist[v] + 1);
        }
        recStack[u] = false;
    }
}
```
