class Solution {
    public int minStepToReachTarget(int knightPos[], int targetPos[], int n) {
        // If start and target are the same
        if (knightPos[0] == targetPos[0] && knightPos[1] == targetPos[1]) {
            return 0;
        }

        // 8 possible moves for a Knight
        int[] dx = {-2, -2, -1, -1, 1, 1, 2, 2};
        int[] dy = {-1, 1, -2, 2, -2, 2, -1, 1};

        boolean[][] visited = new boolean[n + 1][n + 1];
        Queue<int[]> q = new LinkedList<>();

        // Add initial position to queue: {x, y, dist}
        int level = -1;
        q.add(new int[]{knightPos[0], knightPos[1]});
        visited[knightPos[0]][knightPos[1]] = true;

        while (!q.isEmpty()) {
            level++;
            int size = q.size();
            while(size-->0) {
                int[] curr = q.poll();
                int x = curr[0];
                int y = curr[1];
    
                // Explore all 8 moves
                for (int i = 0; i < 8; i++) {
                    int nx = x + dx[i];
                    int ny = y + dy[i];
    
                    // Check if target is reached
                    if (nx == targetPos[0] && ny == targetPos[1]) {
                        return level + 1;
                    }
    
                    // Check boundaries and if already visited
                    if (nx >= 1 && nx <= n && ny >= 1 && ny <= n && !visited[nx][ny]) {
                        visited[nx][ny] = true;
                        q.add(new int[]{nx, ny});
                    }
                }
            }
        }

        return -1; // Should not happen on a standard chessboard
    }
}
