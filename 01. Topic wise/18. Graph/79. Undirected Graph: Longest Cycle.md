### DFS
```java
class UndirectedLongest {
    int maxCycle = -1;
    public int find(int V, List<List<Integer>> adj) {
        for (int i = 0; i < V; i++) {
            int[] dist = new int[V];
            Arrays.fill(dist, -1);
            dfs(i, -1, 0, dist, adj);
        }
        return maxCycle;
    }
    private void dfs(int u, int p, int d, int[] dist, List<List<Integer>> adj) {
        dist[u] = d;
        for (int v : adj.get(u)) {
            if (v == p) continue;
            if (dist[v] != -1) maxCycle = Math.max(maxCycle, d - dist[v] + 1);
            else dfs(v, u, d + 1, dist, adj);
        }
        dist[u] = -1; // Backtrack to allow node to be part of other paths
    }
}
```
