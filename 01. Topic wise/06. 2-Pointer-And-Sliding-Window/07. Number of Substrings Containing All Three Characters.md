class Solution {
    public int numberOfSubstrings(String s) {
        int[] count = new int[3];
        int start = 0;
        int result = 0;
        int n = s.length();

        for (int end = 0; end < n; end++) {
            count[s.charAt(end) - 'a']++;
            while (count[0] > 0 && count[1] > 0 && count[2] > 0) {
                result += n - end;
                count[s.charAt(start) - 'a']--;
                start++;
            }
        }

        return result;
    }
}
