class Solution {
    public int numberOfSubarrays(int[] nums, int k) {
        return countSumLessEqual(nums, k) - countSumLessEqual(nums, k-1);
    }
    int countSumLessEqual(int[] nums, int goal) {
        if(goal<0) return 0;
        int ans=0, start=0, oddCount=0;
        for(int end=0;end<nums.length;end++) {
            oddCount += nums[end]%2!=0 ? 1 : 0;
            while(oddCount>goal) {
                oddCount -= nums[start]%2!=0 ? 1 : 0;
                start++;
            }
            // every possible contiguous subarray that ends at the current right
            ans += end-start+1;
        }
        return ans;
    }
}
