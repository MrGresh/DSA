```java
class Solution {
    public int kBitFlips(int[] nums, int k) {
        int n = nums.length;
        Queue<Integer> flipEndIndices = new LinkedList<>();
        int res = 0;
    
        for (int i = 0; i < n; i++) {
            // 1. Remove flips that no longer affect the current index 'i'
            if (!flipEndIndices.isEmpty() && flipEndIndices.peek() <= i) {
                flipEndIndices.poll();
            }
    
            // 2. Check the current state: 
            // If (original bit == number of active flips % 2), it means the current bit is 0
            if (nums[i] == (flipEndIndices.size() % 2)) {
                // Can we even flip?
                if (i + k > n) return -1;
                
                res++;
                flipEndIndices.offer(i + k); // This flip affects everything until index i + k
            }
        }
        return res;
    }
}
```

```java
class Solution {
    public int kBitFlips(int[] nums, int k) {
        int n = nums.length;
        int[] diff = new int[n + 1];
        int currentFlips = 0;
        int res = 0;
    
        for (int i = 0; i < n; i++) {
            currentFlips += diff[i];
    
            // If current state is 0, we need to flip
            // (nums[i] + currentFlips) % 2 == 0 means the bit is 0
            if ((nums[i] + currentFlips) % 2 == 0) {
                if (i + k > n) return -1;
                
                res++;
                currentFlips++;
                diff[i + k]--; // Mark where this flip's influence ends
            }
        }
        return res;
    }
}
```

```java
class Solution {
    public int kBitFlips(int[] arr, int k) {
        int n = arr.length;
        int flipped = 0; // The count of flips currently affecting the current index
        int res = 0;

        for (int i = 0; i < n; i++) {
            // If the current index is where a previous flip ends, reduce the 'flipped' count
            if (i >= k && arr[i - k] > 1) {
                flipped ^= 1;
                arr[i - k] -= 2; // Restore original value (optional, but good practice)
            }

            // If the current bit (after considering 'flipped') is 0
            // (arr[i] == flipped) is a clever way to check if it's 0 after flips
            if (arr[i] == flipped) {
                if (i + k > n) return -1;
                
                res++;
                flipped ^= 1; // Toggle the current flipping state
                arr[i] += 2;  // Use the current index to mark that a flip started here
            }
        }
        return res;
    }
}
```
