class Solution {
    public int minSwaps(int[] arr) {
        int cnt = 0;
        for(int x: arr) {
            if(x==1) cnt++;
        }
        if(cnt==0) return -1;
        if(cnt==arr.length) return 0;
        int bad = 0;
        for(int i=0;i<cnt;i++) {
            if(arr[i]==0) bad++;
        }
        int ans = bad;
        for(int i=cnt;i<arr.length;i++) {
            if(arr[i-cnt]==0) bad--;
            if(arr[i]==0) bad++;
            ans = Math.min(ans, bad);
        }
        return ans;
    }
}
