class Solution {
    public int countIncreasing(int[] arr) {
        int ans = 0, start = 0;
        for(int end=1;end<arr.length;end++) {
            if(arr[end]<=arr[end-1]) {
                int n = end-start;
                ans += ((n*(n+1))/2) - n;
                start = end;
            } else if(end==arr.length-1) {
                int n = end-start+1;
                ans += ((n*(n+1))/2) - n;
            }
        }
        return ans;
    }
}
