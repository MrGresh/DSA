class Solution {
    public int numSubarraysWithSum(int[] nums, int k) {
        Map<Integer, Integer> m = new HashMap<>();
        int res = 0;
        for(int i=0;i<nums.length;i++) {
            if(i>0) nums[i] += nums[i-1];
            if(nums[i]==k) res+=1;
            if(m.containsKey(nums[i]-k)) res += m.get(nums[i]-k);
            m.put(nums[i], m.getOrDefault(nums[i], 0)+1);
        }
        return res;
    }
}

class Solution {
    public int numSubarraysWithSum(int[] nums, int goal) {
        return countSubarraysSumLessEqual(nums, goal) - countSubarraysSumLessEqual(nums, goal-1);
    }
    int countSubarraysSumLessEqual(int[] nums, int goal) {
        if(goal<0) return 0;
        int ans=0, start=0, sum=0;
        for(int end=0;end<nums.length;end++) {
            sum += nums[end];
            while(sum>goal) {
                sum -= nums[start];
                start++;
            }
            // every possible contiguous subarray that ends at the current right
            ans += end-start+1;
        }
        return ans;
    }
}
