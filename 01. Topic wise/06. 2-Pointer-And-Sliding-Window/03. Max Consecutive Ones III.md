class Solution {
    public int longestOnes(int[] nums, int k) {
        int start = 0, end = 0, ans = 0, count = 0;
        while(end<nums.length) {
            if(nums[end]==0) count++;
            while(count > k) {
                if(nums[start]==0) {
                    count--;
                    start++;
                    break;
                } else start++;
            }
            ans = Math.max(ans, end-start+1);
            end++;
        }
        return ans;
    }
}
