class Solution {
    public int countSubstr(String s, int k) {
        return count(s, k) - count(s, k-1);
    }
    int count(String s, int k) {
        if(k<0) return 0;
        Map<Character, Integer> count = new HashMap<>();
        int start = 0, ans = 0, curr = 0;
        for(int end=0;end<s.length();end++) {
            count.put(s.charAt(end), count.getOrDefault(s.charAt(end),0)+1);
            while(count.size()>k) {
                if(count.get(s.charAt(start))==1) count.remove(s.charAt(start));
                else count.put(s.charAt(start), count.get(s.charAt(start))-1);
                start++;
            }
            ans += (end-start+1);
        }
        return ans;
    }
}
