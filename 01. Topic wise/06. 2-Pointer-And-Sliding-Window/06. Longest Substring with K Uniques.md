class Solution {
    public int longestKSubstr(String s, int k) {
        int ans = -1, start = 0;
        Map<Character, Integer> freq = new HashMap<>();
        for(int end=0;end<s.length();end++) {
            freq.put(s.charAt(end), freq.getOrDefault(s.charAt(end), 0)+1);
            while(freq.size()>k) {
                if(freq.get(s.charAt(start))==1) freq.remove(s.charAt(start));
                else freq.put(s.charAt(start), freq.get(s.charAt(start))-1);
                start++;
            }
            if(freq.size()==k) ans = Math.max(ans, end - start + 1);
        }
        return ans;
    }
}
