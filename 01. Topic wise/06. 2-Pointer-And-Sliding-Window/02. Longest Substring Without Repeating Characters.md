class Solution {
    public int lengthOfLongestSubstring(String s) {
        if(s.length()==0) return 0;
        int ans = 1;
        int f = 0, e = 0;
        Map<Character, Integer> m = new HashMap<>();
        while(e<s.length()) {
            m.put(s.charAt(e), m.getOrDefault(s.charAt(e), 0)+1);
            while(m.get(s.charAt(e))>1) {
                if( m.get(s.charAt(f))>1) m.put(s.charAt(f), m.get(s.charAt(f))-1);
                else m.remove(s.charAt(f));
                f++;
            }
            ans = Math.max(ans, e-f+1);
            e++;
        }
        return ans;
    }
}
