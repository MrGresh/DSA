### Memoization
```java
class Solution {
    int[][] memo;
    public int maximumAmount(int arr[]) {
        int n = arr.length;
        memo = new int[n][n];
        for (int[] row : memo) Arrays.fill(row, -1);
        return solve(0, n - 1, arr);
    }

    private int solve(int i, int j, int[] arr) {
        // Base case: only one or two coins left
        if (i > j) return 0;
        if (i == j) return arr[i];
        
        if (memo[i][j] != -1) return memo[i][j];

        // Option 1: You pick arr[i]. 
        // Opponent picks either arr[i+1] or arr[j], leaving you the min.
        int pickFirst = arr[i] + Math.min(solve(i + 2, j, arr), solve(i + 1, j - 1, arr));

        // Option 2: You pick arr[j]. 
        // Opponent picks either arr[i] or arr[j-1], leaving you the min.
        int pickLast = arr[j] + Math.min(solve(i + 1, j - 1, arr), solve(i, j - 2, arr));

        return memo[i][j] = Math.max(pickFirst, pickLast);
    }
}
```
### Tabulation
```java
class Solution {
    public int maximumAmount(int arr[]) {
        int n = arr.length;
        int[][] dp = new int[n][n];

        // gap represents the distance between i and j
        for (int gap = 0; gap < n; gap++) {
            for (int i = 0, j = gap; j < n; i++, j++) {
                // Base cases are implicitly handled or explicit:
                if (gap == 0) {
                    dp[i][j] = arr[i];
                } else if (gap == 1) {
                    dp[i][j] = Math.max(arr[i], arr[j]);
                } else {
                    // Minimax logic
                    int pickFirst = arr[i] + Math.min(dp[i + 2][j], dp[i + 1][j - 1]);
                    int pickLast = arr[j] + Math.min(dp[i + 1][j - 1], dp[i][j - 2]);
                    dp[i][j] = Math.max(pickFirst, pickLast);
                }
            }
        }
        return dp[0][n - 1];
    }
}
```
