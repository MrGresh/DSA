### Memoization
```java
class Solution {
    int[][] memo;
    int n;

    public int maxProfit(int[] prices) {
        n = prices.length;
        memo = new int[n][n];
        for (int[] row : memo) Arrays.fill(row, -1);
        return solve(0, n - 1, prices);
    }

    private int solve(int i, int j, int[] prices) {
        if (i > j) return 0;
        if (memo[i][j] != -1) return memo[i][j];

        int year = n - (j - i); // Current year
        
        // Option 1: Sell leftmost
        int sellLeft = (prices[i] * year) + solve(i + 1, j, prices);
        // Option 2: Sell rightmost
        int sellRight = (prices[j] * year) + solve(i, j - 1, prices);

        return memo[i][j] = Math.max(sellLeft, sellRight);
    }
}
```
### Tabulation
```java
public int maxProfit(int[] prices) {
    int n = prices.length;
    int[][] dp = new int[n][n];

    // Base case: Intervals of length 1 (Year = n)
    for (int i = 0; i < n; i++) {
        dp[i][i] = prices[i] * n;
    }

    // Fill for lengths 2 to n
    for (int len = 2; len <= n; len++) {
        for (int i = 0; i <= n - len; i++) {
            int j = i + len - 1;
            int year = n - len + 1;

            // Max of (sell left + future) or (sell right + future)
            dp[i][j] = Math.max(
                prices[i] * year + dp[i + 1][j],
                prices[j] * year + dp[i][j - 1]
            );
        }
    }
    return dp[0][n - 1];
}
```
