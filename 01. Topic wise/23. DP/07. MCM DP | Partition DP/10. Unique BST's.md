### Memoization
```java
class Solution {
    static int[] memo;

    static int numTrees(int n) {
        memo = new int[n + 1];
        Arrays.fill(memo, -1);
        return solve(n);
    }

    private static int solve(int n) {
        // Base cases: 0 nodes (null tree) or 1 node = 1 unique tree
        if (n <= 1) return 1;
        if (memo[n] != -1) return memo[n];

        int total = 0;
        // Try making each number 'i' the root
        for (int i = 1; i <= n; i++) {
            // Left subtree has i-1 nodes, Right subtree has n-i nodes
            total += solve(i - 1) * solve(n - i);
        }

        return memo[n] = total;
    }
}
```
### Tabulation
```java
class Solution {
    static int numTrees(int n) {
        int[] dp = new int[n + 1];
        
        // Base cases
        dp[0] = 1;
        dp[1] = 1;

        // i represents the total number of nodes in the current BST
        for (int i = 2; i <= n; i++) {
            // j represents the root of the BST
            for (int j = 1; j <= i; j++) {
                // dp[nodes] = sum(dp[left_nodes] * dp[right_nodes])
                dp[i] += dp[j - 1] * dp[i - j];
            }
        }
        
        return dp[n];
    }
}
```
### Space Optimized (Mathematical Formula)
```java
class Solution {
    static int numTrees(int n) {
        // Using double or long to prevent overflow during intermediate steps
        long catalan = 1;
        for (int i = 0; i < n; i++) {
            catalan = catalan * 2 * (2 * i + 1) / (i + 2);
        }
        return (int) catalan;
    }
}
```
