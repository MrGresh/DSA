class Solution {
    public static int longestBitonicSequence(int n, int[] nums) {
        int[] lis = new int[n];
        int[] lds = new int[n];
        Arrays.fill(lis, 1);
        Arrays.fill(lds, 1);

        // 1. Calculate LIS from left to right
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    lis[i] = Math.max(lis[i], 1 + lis[j]);
                }
            }
        }

        // 2. Calculate LDS from right to left (standard LIS logic but reversed)
        for (int i = n - 1; i >= 0; i--) {
            for (int j = n - 1; j > i; j--) {
                if (nums[j] < nums[i]) {
                    lds[i] = Math.max(lds[i], 1 + lds[j]);
                }
            }
        }

        int maxLen = 0;
        // 3. Find the peak where both sides are non-empty
        for (int i = 0; i < n; i++) {
            if (lis[i] > 1 && lds[i] > 1) {
                // Subtract 1 because nums[i] is counted in both lis[i] and lds[i]
                maxLen = Math.max(maxLen, lis[i] + lds[i] - 1);
            }
        }

        return maxLen;
    }
}
