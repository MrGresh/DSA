### Memoization
```java
class Solution {
    static class Box {
        int h, w, l;
        Box(int h, int w, int l) {
            this.h = h;
            // Standardize: ensure w <= l for easier comparison
            this.w = Math.min(w, l);
            this.l = Math.max(w, l);
        }
    }

    public static int maxHeight(int[] height, int[] width, int[] length) {
        int n = height.length;
        List<Box> boxes = new ArrayList<>();
        for (int i = 0; i < n; i++) {
            boxes.add(new Box(height[i], width[i], length[i]));
            boxes.add(new Box(width[i], height[i], length[i]));
            boxes.add(new Box(length[i], height[i], width[i]));
        }

        // Sort by length (and then width) descending
        boxes.sort((a, b) -> (b.l != a.l) ? b.l - a.l : b.w - a.w);
        
        int[] memo = new int[boxes.size()];
        Arrays.fill(memo, -1);
        
        int maxH = 0;
        for (int i = 0; i < boxes.size(); i++) {
            maxH = Math.max(maxH, solve(i, boxes, memo));
        }
        return maxH;
    }

    private static int solve(int i, List<Box> boxes, int[] memo) {
        if (memo[i] != -1) return memo[i];
        
        int currentMax = 0;
        Box bottom = boxes.get(i);
        
        for (int next = i + 1; next < boxes.size(); next++) {
            Box top = boxes.get(next);
            // Strict condition: both dimensions must be smaller
            if (top.w < bottom.w && top.l < bottom.l) {
                currentMax = Math.max(currentMax, solve(next, boxes, memo));
            }
        }
        
        return memo[i] = bottom.h + currentMax;
    }
}
```

### Tabulation
```java
class Solution {
    static class Box {
        int h, w, l;
        Box(int h, int w, int l) {
            this.h = h;
            // Standardize: ensure w <= l for easier comparison
            this.w = Math.min(w, l);
            this.l = Math.max(w, l);
        }
    }
    public static int maxHeight(int[] height, int[] width, int[] length) {
        int n = height.length;
        Box[] boxes = new Box[3 * n];
        for (int i = 0; i < n; i++) {
            boxes[3*i] = new Box(height[i], width[i], length[i]);
            boxes[3*i+1] = new Box(width[i], height[i], length[i]);
            boxes[3*i+2] = new Box(length[i], height[i], width[i]);
        }
    
        // Sort by base area descending to ensure we only look "backwards"
        Arrays.sort(boxes, (a, b) -> (b.l * b.w) - (a.l * a.w));
    
        int m = boxes.length;
        int[] dp = new int[m];
        int maxOverall = 0;
    
        for (int i = 0; i < m; i++) {
            dp[i] = boxes[i].h; // Initial height is just the box itself
            for (int j = 0; j < i; j++) {
                // If boxes[i] can be placed on top of boxes[j]
                if (boxes[i].w < boxes[j].w && boxes[i].l < boxes[j].l) {
                    dp[i] = Math.max(dp[i], dp[j] + boxes[i].h);
                }
            }
            maxOverall = Math.max(maxOverall, dp[i]);
        }
        return maxOverall;
    }
}
```
