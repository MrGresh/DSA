class Solution {
    public int longestStrChain(String[] words) {
        // 1. Sort by length so predecessors always come before successors
        Arrays.sort(words, (a, b) -> a.length() - b.length());
        
        int n = words.length;
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        int maxChain = 1;
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                // If words[j] can form words[i] by adding one letter
                if (isPredecessor(words[j], words[i]) && dp[j] + 1 > dp[i]) {
                    dp[i] = dp[j] + 1;
                }
            }
            maxChain = Math.max(maxChain, dp[i]);
        }
        return maxChain;
    }
    
    private boolean isPredecessor(String a, String b) {
        if (b.length() != a.length() + 1) return false;
        int i = 0, j = 0;
        while (j < b.length()) {
            if (i < a.length() && a.charAt(i) == b.charAt(j)) {
                i++;
            }
            j++;
        }
        return i == a.length();
    }
}
