### Memoization
```java
class Solution {
    int[] memo;
    public int maxEnvelopes(int[][] envelopes) {
        Arrays.sort(envelopes, (a, b) -> a[0] == b[0] ? b[1] - a[1] : a[0] - b[0]);
        memo = new int[envelopes.length];
        int maxVal = 0;
        for (int i = 0; i < envelopes.length; i++) {
            maxVal = Math.max(maxVal, solve(i, envelopes));
        }
        return maxVal;
    }

    private int solve(int i, int[][] envelopes) {
        if (memo[i] != 0) return memo[i];
        int res = 1;
        for (int j = i + 1; j < envelopes.length; j++) {
            if (envelopes[j][1] > envelopes[i][1]) {
                res = Math.max(res, 1 + solve(j, envelopes));
            }
        }
        return memo[i] = res;
    }
}
```

### Tabulation
```java
class Solution {
    public int maxEnvelopes(int[][] envelopes) {
        // Sort by width ascending, then height descending
        Arrays.sort(envelopes, (a, b) -> a[0] == b[0] ? b[1] - a[1] : a[0] - b[0]);

        int n = envelopes.length;
        int[] dp = new int[n];
        Arrays.fill(dp, 1);
        int maxLen = 1;

        for (int i = 1; i < n; i++) {
            for (int j = 0; j < i; j++) {
                if (envelopes[i][1] > envelopes[j][1]) {
                    dp[i] = Math.max(dp[i], dp[j] + 1);
                }
            }
            maxLen = Math.max(maxLen, dp[i]);
        }
        return maxLen;
    }
}
```

### Optimal
```java
class Solution {
    public int maxEnvelopes(int[][] envelopes) {
        // Step 1: Sort width ascending, height descending for same width
        Arrays.sort(envelopes, (a, b) -> {
            if (a[0] == b[0]) return b[1] - a[1];
            return a[0] - b[0];
        });

        // Step 2: Find LIS on heights using Binary Search
        int n = envelopes.length;
        int[] tails = new int[n];
        int size = 0;

        for (int[] envelope : envelopes) {
            int h = envelope[1];
            // Binary search for the index of the first element >= h
            int i = 0, j = size;
            while (i != j) {
                int mid = i + (j - i) / 2;
                if (tails[mid] < h) i = mid + 1;
                else j = mid;
            }
            tails[i] = h;
            if (i == size) size++;
        }
        return size;
    }
}
```
