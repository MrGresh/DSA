### Memoization
```java
class Solution {
    // Map of <Index, Map<Difference, Length>>
    Map<Integer, Integer>[] memo;

    public int longestArithSeqLength(int[] nums) {
        int n = nums.length;
        memo = new HashMap[n];
        for (int i = 0; i < n; i++) memo[i] = new HashMap<>();

        int maxLen = 0;
        // We check every pair as a possible start
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                maxLen = Math.max(maxLen, 2 + solve(i, nums[j] - nums[i], nums));
            }
        }
        return maxLen;
    }

    private int solve(int i, int diff, int[] nums) {
        if (memo[i].containsKey(diff)) return memo[i].get(diff);

        int res = 0;
        // Look backwards for an element that maintains the same diff
        for (int prev = i - 1; prev >= 0; prev--) {
            if (nums[i] - nums[prev] == diff) {
                res = 1 + solve(prev, diff, nums);
                break; // Only need the first one we find moving backwards
            }
        }
        memo[i].put(diff, res);
        return res;
    }
}
```

### Tabulation
```java
class Solution {
    public int longestArithSeqLength(int[] nums) {
        int n = nums.length;
        if (n <= 2) return n;

        int maxLen = 2;
        // dp[index][difference]
        // Range of nums is 0-500, so max diff is 500 and min diff is -500.
        // Total range is 1000. We add an offset of 500 to handle negative diffs.
        int[][] dp = new int[n][1001];

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < i; j++) {
                int diff = nums[i] - nums[j] + 500; // Offset for negative diff
                
                // If we found a subsequence ending at j with this diff, extend it.
                // Otherwise, the current pair (j, i) forms a sequence of length 2.
                dp[i][diff] = (dp[j][diff] == 0) ? 2 : dp[j][diff] + 1;
                
                maxLen = Math.max(maxLen, dp[i][diff]);
            }
        }
        return maxLen;
    }
}
```
