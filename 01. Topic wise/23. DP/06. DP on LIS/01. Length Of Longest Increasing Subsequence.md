### Memoization
```java
class Solution {
    Integer[] memo;
    public int lengthOfLIS(int[] nums) {
        memo = new Integer[nums.length];
        int ans = 0;
        for(int i=0;i<nums.length;i++) {
            ans = Math.max(ans, rec(nums, i));
        }
        return ans;
    }
    private int rec(int[] nums, int i) {
        if(memo[i]!=null) return memo[i];
        int val = 1;
        for(int j=i+1;j<nums.length;j++) {
            if(nums[j]>nums[i]) {
                val = Math.max(val, 1 + rec(nums, j));
            }
        }
        return memo[i] = val;
    }
}
```

### Tabulation
```java
class Solution {
    public int lengthOfLIS(int[] nums) {
        int[] dp = new int[nums.length];
        for(int i=nums.length-1;i>=0;i--) {
            int val = 1;
            for(int j=i+1;j<nums.length;j++) {
                if(nums[j]>nums[i]) val = Math.max(val, 1 + dp[j]);
            }
            dp[i] = val;
        }
        int ans = 0;
        for(int i=0;i<nums.length;i++) {
            ans = Math.max(ans, dp[i]);
        }
        return ans;
    }
}
```

### Binary Search
```java
class Solution {
    public int lengthOfLIS(int[] nums) {
        int[] tails = new int[nums.length];
        int size = 0;

        for (int x : nums) {
            int i = 0, j = size;
            // Binary search for the insertion point
            while (i != j) {
                int m = (i + j) / 2;
                if (tails[m] < x) i = m + 1;
                else j = m;
            }
            tails[i] = x;
            if (i == size) size++;
        }
        return size;
    }
}
```
