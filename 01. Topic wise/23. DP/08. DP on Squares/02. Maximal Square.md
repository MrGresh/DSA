### Memoization
```java
class Solution {
    Integer[][] memo;
    int maxSide = 0;

    public int maximalSquare(char[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        memo = new Integer[m][n];
        
        // We must check every cell because any cell could be the bottom-right of the max square
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                if (matrix[i][j] == '1') {
                    solve(i, j, matrix);
                }
            }
        }
        return maxSide * maxSide;
    }

    private int solve(int i, int j, char[][] matrix) {
        if (i < 0 || j < 0 || matrix[i][j] == '0') return 0;
        if (memo[i][j] != null) return memo[i][j];

        // The size depends on the three neighbors
        int left = solve(i, j - 1, matrix);
        int top = solve(i - 1, j, matrix);
        int topLeft = solve(i - 1, j - 1, matrix);

        int res = 1 + Math.min(topLeft, Math.min(top, left));
        maxSide = Math.max(maxSide, res);
        
        return memo[i][j] = res;
    }
}
```

### Tabulation
```java
class Solution {
    public int maximalSquare(char[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[][] dp = new int[m + 1][n + 1];
        int maxSide = 0;

        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (matrix[i - 1][j - 1] == '1') {
                    // Logic: min of top, left, and diagonal + 1
                    dp[i][j] = Math.min(Math.min(dp[i - 1][j], dp[i][j - 1]), dp[i - 1][j - 1]) + 1;
                    maxSide = Math.max(maxSide, dp[i][j]);
                }
            }
        }

        return maxSide * maxSide;
    }
}
```

### Space Optimization
```java
class Solution {
    public int maximalSquare(char[][] matrix) {
        int m = matrix.length;
        int n = matrix[0].length;
        int[] dp = new int[n + 1];
        int maxSide = 0, prev = 0;

        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                int temp = dp[j]; // Store the value before it's updated for the next cell's diagonal
                if (matrix[i - 1][j - 1] == '1') {
                    // dp[j] is top, dp[j-1] is left, prev is top-left
                    dp[j] = Math.min(Math.min(dp[j], dp[j - 1]), prev) + 1;
                    maxSide = Math.max(maxSide, dp[j]);
                } else {
                    dp[j] = 0;
                }
                prev = temp; // Current 'top' becomes 'top-left' for the next iteration
            }
        }

        return maxSide * maxSide;
    }
}
```
