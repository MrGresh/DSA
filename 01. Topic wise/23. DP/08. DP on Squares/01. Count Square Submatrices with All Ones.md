### Memoization
```java
class Solution {
    int[][] memo;

    public int countSquares(int[][] matrix) {
        int n = matrix.length;
        int m = matrix[0].length;
        memo = new int[n][m];
        for (int[] row : memo) Arrays.fill(row, -1);

        int totalSquares = 0;

        // We must call the function for every cell because any cell 
        // could be the bottom-right corner of multiple squares.
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                totalSquares += solve(i, j, matrix);
            }
        }

        return totalSquares;
    }

    private int solve(int i, int j, int[][] matrix) {
        // Base Case: Out of bounds
        if (i < 0 || j < 0) return 0;
        
        // Base Case: If the cell itself is 0, no square can end here
        if (matrix[i][j] == 0) return 0;

        // Return cached result
        if (memo[i][j] != -1) return memo[i][j];

        // The size of the square ending at (i, j) is 1 + min of 3 neighbors
        int top = solve(i - 1, j, matrix);
        int left = solve(i, j - 1, matrix);
        int topLeft = solve(i - 1, j - 1, matrix);

        return memo[i][j] = 1 + Math.min(top, Math.min(left, topLeft));
    }
}
```

### Tabulation
```java
class Solution {
    int countSquares(int[][] arr) {
        int n = arr.length, m = arr[0].length;
        // DP table to store size of largest square ending at (i,j)
        int[][] dp = new int[n][m];

        // Initialize first row of DP table
        for (int j = 0; j < m; j++) dp[0][j] = arr[0][j];

        // Initialize first column of DP table
        for (int i = 0; i < n; i++) dp[i][0] = arr[i][0];

        // Fill the rest of the DP table
        for (int i = 1; i < n; i++) {
            for (int j = 1; j < m; j++) {
                // If current cell is 0, no square ends here
                if (arr[i][j] == 0) dp[i][j] = 0;
                else { 
                    // Minimum of top, left, top-left + 1
                    dp[i][j] = 1 + Math.min(dp[i - 1][j], Math.min(dp[i - 1][j - 1], dp[i][j - 1]));
                }
            }
        }

        // Sum up all values in DP table
        int sum = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) sum += dp[i][j];
        }

        // Return total count of squares
        return sum;
    }
}
```

### Space Optimization
```java
class Solution {
    public int countSquares(int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        int[] prev = new int[m];
        int[] curr = new int[m];
        int total = 0;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (matrix[i][j] == 1) {
                    if (i == 0 || j == 0) {
                        curr[j] = 1;
                    } else {
                        curr[j] = 1 + Math.min(prev[j], Math.min(prev[j - 1], curr[j - 1]));
                    }
                    total += curr[j];
                } else {
                    curr[j] = 0;
                }
            }
            // Move curr to prev for the next row
            prev = curr.clone();
        }
        return total;
    }
}
```

### Best Space Optimization
```java
class Solution {
    public int countSquares(int[][] matrix) {
        int n = matrix.length, m = matrix[0].length;
        int[] dp = new int[m];
        int total = 0;
        int prevTopLeft = 0; // Represents dp[i-1][j-1]

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                int temp = dp[j]; // Store current dp[j] before it's updated (it will be the topLeft for next j)
                
                if (matrix[i][j] == 1) {
                    if (i == 0 || j == 0) {
                        dp[j] = 1;
                    } else {
                        // dp[j] is top, dp[j-1] is left, prevTopLeft is diagonal
                        dp[j] = 1 + Math.min(dp[j], Math.min(dp[j - 1], prevTopLeft));
                    }
                    total += dp[j];
                } else {
                    dp[j] = 0;
                }
                prevTopLeft = temp; // Update for the next iteration in the row
            }
        }
        return total;
    }
}
```
