### Memoization

class Solution {
    Long[][] memo;
    int[] left, right, num;
    int n;

    public int maxWalls(int[] robots, int[] distance, int[] walls) {
        this.n = robots.length;
        this.left = new int[n]; // If Robot i fires Left, how many walls does it hit before being blocked?
        this.right = new int[n]; // If Robot i fires Right, how many walls does it hit before being blocked?
        this.num = new int[n]; // TOTAL number of unique walls physically sitting between Robot i-1 and Robot i
        
        Map<Integer, Integer> distMap = new HashMap<>();
        for (int i = 0; i < n; i++) distMap.put(robots[i], distance[i]);
        
        Arrays.sort(robots);
        Arrays.sort(walls);

        for (int i = 0; i < n; i++) {
            int d = distMap.get(robots[i]);
            int p1 = upperBound(walls, robots[i]);
            int p2 = lowerBound(walls, robots[i]);
            
            // Standard Left reach clipped by neighbor
            int lBound = (i == 0) ? robots[i] - d : Math.max(robots[i] - d, robots[i-1] + 1);
            left[i] = p1 - lowerBound(walls, lBound);
            
            // Standard Right reach clipped by neighbor
            int rBound = (i == n-1) ? robots[i] + d : Math.min(robots[i] + d, robots[i+1] - 1);
            right[i] = upperBound(walls, rBound) - p2;
            
            // Total unique walls in the gap [Robot i-1, Robot i]
            if (i > 0) num[i] = p1 - lowerBound(walls, robots[i-1]);
        }

        // --- 2. Memoization Table ---
        memo = new Long[n][2];
        return (int) rec(0, 0);
    }

    private long rec(int i, int prevFiredRight) {
        if (i == n) return 0;
        if (memo[i][prevFiredRight] != null) return memo[i][prevFiredRight];

        // Choice A: Robot i fires LEFT
        long resL;
        if (prevFiredRight == 1 && i > 0) {
            // Overlap logic: Use the union (num[i]) minus what we already counted for i-1's Right
            resL = (Math.min(left[i] + right[i-1], num[i]) - right[i-1]) + rec(i + 1, 0);
        } else {
            resL = left[i] + rec(i + 1, 0);
        }

        // Choice B: Robot i fires RIGHT
        long resR = right[i] + rec(i + 1, 1);

        // Choice C: Robot i does NOTHING (Optional/Skip)
        long resSkip = rec(i + 1, 0);

        return memo[i][prevFiredRight] = Math.max(resL, Math.max(resR, resSkip));
    }

    // --- Helper Functions ---
    private int lowerBound(int[] arr, int target) {
        int l = 0, r = arr.length;
        while (l < r) {
            int m = l + (r - l) / 2;
            if (arr[m] < target) l = m + 1;
            else r = m;
        }
        return l;
    }

    private int upperBound(int[] arr, int target) {
        int l = 0, r = arr.length;
        while (l < r) {
            int m = l + (r - l) / 2;
            if (arr[m] <= target) l = m + 1; 
            else r = m;
        }
        return l;
    }
}
