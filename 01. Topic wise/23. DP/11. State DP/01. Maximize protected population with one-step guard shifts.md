### Memoization
```java
class Solution {
    private Long[][] memo;

    public long moveUnits(List<Integer> pop, String unit) {
        int n = pop.size();
        memo = new Long[n][2];
        return solve(n - 1, 0, pop, unit);
    }

    private long solve(int i, int isFilledByRight, List<Integer> pop, String unit) {
        if (i < 0) return 0;
        if (memo[i][isFilledByRight] != null) return memo[i][isFilledByRight];

        long res = 0;
        boolean hasUnit = (unit.charAt(i) == '1');

        if (isFilledByRight == 1) {
            res = pop.get(i) + solve(i - 1, (hasUnit && i > 0) ? 1 : 0, pop, unit);
        } else {
            if (hasUnit) {
                long stay = pop.get(i) + solve(i - 1, 0, pop, unit);
                long moveLeft = (i > 0) ? solve(i - 1, 1, pop, unit) : 0;
                res = Math.max(stay, moveLeft);
            } else res = solve(i - 1, 0, pop, unit);
        }

        return memo[i][isFilledByRight] = res;
    }
}
```

### Tabulation
```java
long moveUnits(List<Integer> pop, String unit) {
    int n = pop.size();
    long[][] dp = new long[n][2];
    for(int i=0;i<n;i++) {
        boolean hasUnit = (unit.charAt(i) == '1');
        for(int isFilledByRight = 0; isFilledByRight <= 1; isFilledByRight++) {
            long res = 0;

            if (isFilledByRight == 1) {
                res = pop.get(i) + ((i>0) ? dp[i - 1][hasUnit ? 1 : 0] : 0);
            } else {
                if (hasUnit) {
                    long stay = pop.get(i) + ((i>0) ? dp[i - 1][0] : 0);
                    long moveLeft = ((i>0) ? dp[i - 1][1] : 0);
                    res = Math.max(stay, moveLeft);
                } else res = ((i>0) ? dp[i - 1][0] : 0);
            }

            dp[i][isFilledByRight] = res;
        }
    }
    return dp[n-1][0];
}
```

### Space Optimization
```java
long moveUnits(List<Integer> pop, String unit) {
    int n = pop.size();
    long[] prev = new long[2];
    for(int i=0;i<n;i++) {
        long[] curr = new long[2];
        boolean hasUnit = (unit.charAt(i) == '1');
        for(int isFilledByRight = 0; isFilledByRight <= 1; isFilledByRight++) {
            long res = 0;

            if (isFilledByRight == 1) {
                res = pop.get(i) + ((i>0) ? prev[hasUnit ? 1 : 0] : 0);
            } else {
                if (hasUnit) {
                    long stay = pop.get(i) + ((i>0) ? prev[0] : 0);
                    long moveLeft = ((i>0) ? prev[1] : 0);
                    res = Math.max(stay, moveLeft);
                } else res = ((i>0) ? prev[0] : 0);
            }

            curr[isFilledByRight] = res;
        }
        prev = curr;
    }
    return prev[0];
}
```
