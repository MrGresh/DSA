### Memoization
```java
class Solution {
    int[] memo;
    String S;
    String[] digits;

    public int atMostNGivenDigitSet(String[] digits, int n) {
        this.S = String.valueOf(n);
        this.digits = digits;
        int L = S.length();
        int count = 0;

        // 1. Count all numbers with fewer digits than n
        for (int i = 1; i < L; i++) {
            count += Math.pow(digits.length, i);
        }

        // 2. Count numbers with exactly L digits using Digit DP
        memo = new int[L];
        Arrays.fill(memo, -1);
        return count + dp(0, true, true);
    }

    private int dp(int idx, boolean isLimit, boolean isStarting) {
        if (idx == S.length()) return 1;
        if (!isLimit && memo[idx] != -1) return memo[idx];

        int ans = 0;
        int upper = isLimit ? S.charAt(idx) - '0' : 9;

        for (String d : digits) {
            int digit = Integer.parseInt(d);
            if (digit < upper) {
                ans += dp(idx + 1, false, false);
            } else if (digit == upper) {
                ans += dp(idx + 1, isLimit, false);
            }
        }

        if (!isLimit) memo[idx] = ans;
        return ans;
    }
}
```

### Tabulation
```java
class Solution {
    public int atMostNGivenDigitSet(String[] digits, int n) {
        String S = String.valueOf(n);
        int L = S.length();
        int D = digits.length;
        int total = 0;

        // Numbers with fewer digits: D^1 + D^2 + ... + D^{L-1}
        for (int i = 1; i < L; i++) {
            total += Math.pow(D, i);
        }

        // Numbers with same digits as n
        for (int i = 0; i < L; i++) {
            boolean hasSameDigit = false;
            for (String d : digits) {
                if (d.charAt(0) < S.charAt(i)) {
                    // All remaining positions can be any of the D digits
                    total += Math.pow(D, L - i - 1);
                } else if (d.charAt(0) == S.charAt(i)) {
                    hasSameDigit = true;
                }
            }
            // If we don't have a digit that matches S.charAt(i), 
            // we can't form any more numbers with this prefix
            if (!hasSameDigit) return total;
        }

        return total + 1; // +1 for the number n itself if it's composed of 'digits'
    }
}
```
