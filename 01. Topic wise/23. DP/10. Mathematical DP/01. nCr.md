### Memoization
```java
class Solution {
    static int MOD = 1000000007;
    static Integer[][] memo;

    public int nCr(int n, int r) {
        if (r > n) return 0;
        if (r == 0 || r == n) return 1;
        
        memo = new Integer[n + 1][r + 1];
        return solve(n, r);
    }

    private int solve(int n, int r) {
        // Base Cases
        if (r == 0 || r == n) return 1;
        if (memo[n][r] != null) return memo[n][r];

        // Recursive relation: nCr = (n-1)C(r-1) + (n-1)Cr
        return memo[n][r] = (solve(n - 1, r - 1) + solve(n - 1, r)) % MOD;
    }
}
```
### Tabulation
```java
class Solution {
    int MOD = 1000000007;

    public int nCr(int n, int r) {
        if (r > n) return 0;
        if (r == 0 || r == n) return 1;

        int[][] dp = new int[n + 1][r + 1];

        for (int i = 0; i <= n; i++) {
            for (int j = 0; j <= Math.min(i, r); j++) {
                // Base cases: nC0 = 1 and nCn = 1
                if (j == 0 || j == i) {
                    dp[i][j] = 1;
                } else {
                    // Mirror of the recursive relation
                    dp[i][j] = (dp[i - 1][j - 1] + dp[i - 1][j]) % MOD;
                }
            }
        }
        return dp[n][r];
    }
}
```

### Space Optimization
```java
class Solution {
    int MOD = 1000000007;

    public int nCr(int n, int r) {
        if (r > n) return 0;
        // Optimization: nCr is the same as nC(n-r)
        if (r > n / 2) r = n - r;

        int[] dp = new int[r + 1];
        dp[0] = 1; // Base case: nC0 = 1

        for (int i = 1; i <= n; i++) {
            // Iterate backwards to use values from the previous "n"
            for (int j = Math.min(i, r); j > 0; j--) {
                dp[j] = (dp[j] + dp[j - 1]) % MOD;
            }
        }
        return dp[r];
    }
}
```
