### Memoization
```java
class Solution {
    Map<Integer, Set<Integer>> memo;
    Set<Integer> stonez;
    public boolean canCross(int[] stones) {
        if(stones[1]-stones[0]!=1) return false;
        stonez = new HashSet<>();
        for(int i=0;i<stones.length;i++) stonez.add(stones[i]);
        memo = new HashMap<>();
        for (int s : stones) memo.put(s, new HashSet<>());
        return rec(stones[stones.length-1], stones[1], 1);
    }
    boolean rec(int lastStone, int stone, int k) {
        if(stone==lastStone) return true;
        
        if(memo.get(stone).contains(k)) return false;

        if(stonez.contains(stone+k+1) && rec(lastStone, stone+k+1, k+1)) {
            return true;
        }
        if(stonez.contains(stone+k) && rec(lastStone, stone+k, k)) {
            return true;
        }
        if(k>1 && stonez.contains(stone+k-1) && rec(lastStone, stone+k-1, k-1)) {
            return true;
        }
        
        memo.get(stone).add(k);
        return false;
    }
}
```

### Tabulation
```java
class Solution {
    public boolean canCross(int[] stones) {
        int n = stones.length;
        // Mirroring your 'memo': Map stone position to a Set of jump distances 'k' 
        // that successfully landed on this stone.
        Map<Integer, Set<Integer>> dp = new HashMap<>();
        for (int s : stones) {
            dp.put(s, new HashSet<>());
        }

        // Base case: Mirroring your starting call rec(..., stones[1], 1)
        // We reached stone 1 with a jump of 1.
        if (stones[1] - stones[0] != 1) return false;
        dp.get(1).add(1);

        for (int i = 1; i < n; i++) {
            int currentStone = stones[i];
            for (int k : dp.get(currentStone)) {
                
                // Mirroring your three 'if' checks: k+1, k, k-1
                // We "push" the success forward
                for (int nextJump = k + 1; nextJump >= k - 1; nextJump--) {
                    if (nextJump <= 0) continue;
                    
                    int nextPosition = currentStone + nextJump;
                    if (dp.containsKey(nextPosition)) {
                        dp.get(nextPosition).add(nextJump);
                    }
                }
            }
        }

        // Mirroring 'stone == lastStone': check if the last stone was ever reached
        return !dp.get(stones[n - 1]).isEmpty();
    }
}
```
