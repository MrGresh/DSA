Kadane's algorithm comes into picture when we want to find the maximum possible sum in an array when summing the contiguous elements of the array.

You are given an array. Find the maximum possible sum of contiguous elements of the array ending at each position in the array. Also, return the overall maximum that we can achieve.
### Memoization
```java
class Solution {
    Long[] memo;
    public long maximumSum(int arr[], int n) {
        memo = new Long[n];
        long overallMax = Long.MIN_VALUE;
        
        // We must calculate the "ending at" sum for every index
        for (int i = 0; i < n; i++) {
            long currentEndingSum = solve(i, arr);
            System.out.print(currentEndingSum + " ");
            overallMax = Math.max(overallMax, currentEndingSum);
        }
        System.out.println();
        return overallMax;
    }

    private long solve(int i, int[] arr) {
        if (i == 0) return arr[0];
        if (memo[i] != null) return memo[i];

        // Recurrence: Either start a new subarray at i, or extend the previous one
        return memo[i] = Math.max((long)arr[i], arr[i] + solve(i - 1, arr));
    }
}
```
### Tabulation
```java
class Solution {
    public long maximumSum(int arr[], int n) {
        long[] dp = new long[n];
        
        // Base case
        dp[0] = arr[0];
        long overallMax = dp[0];
        
        // Print the first result
        System.out.print(dp[0] + " ");
        
        for (int i = 1; i < n; i++) {
            // Choice: arr[i] vs (dp[i-1] + arr[i])
            dp[i] = Math.max((long)arr[i], dp[i - 1] + arr[i]);
            
            System.out.print(dp[i] + " ");
            overallMax = Math.max(overallMax, dp[i]);
        }
        
        System.out.println(); // For the required output format
        return overallMax;
    }
}
```
### Space Optimization
```java
class Solution {
    public long maximumSum(int arr[], int n) {
        long currentEndingHere = arr[0];
        long overallMax = arr[0];
        
        System.out.print(currentEndingHere + " ");
        
        for (int i = 1; i < n; i++) {
            currentEndingHere = Math.max((long)arr[i], currentEndingHere + arr[i]);
            System.out.print(currentEndingHere + " ");
            overallMax = Math.max(overallMax, currentEndingHere);
        }
        
        System.out.println();
        return overallMax;
    }
}
```
