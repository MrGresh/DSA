### Memoization
```java
class Solution {
    Integer[][] memo;
    int offset = 1000;
    public int totalWays(int[] arr, int target) {
        memo = new Integer[arr.length+1][2001];
        return rec(arr, 0, target);
    }
    int rec(int[] arr, int i, int target) {
        if(i==arr.length) {
            return target==0 ? 1 : 0;
        }
        if(memo[i][target+offset]!=null) return memo[i][target+offset];
        int ans = 0;
        ans += rec(arr, i+1, target-arr[i]);
        ans += rec(arr, i+1, target+arr[i]);
        return memo[i][target+offset] = ans;
    }
}
```

### Tabulation
```java
class Solution {
    public int totalWays(int[] arr, int target) {
        int n = arr.length;
        int offset = 1000; 
        int[][] dp = new int[n + 1][2001];

        for (int i = n; i >= 0; i--) {
            for (int t = -1000; t <= 1000; t++) {
                int currentIdx = t + offset;
                if (i == n) dp[i][currentIdx] = (t == 0) ? 1 : 0;
                else {
                    long ways = 0;
                    if (t - arr[i] + offset >= 0 && t - arr[i] + offset <= 2000) {
                        ways += dp[i + 1][t - arr[i] + offset];
                    }
                    if (t + arr[i] + offset >= 0 && t + arr[i] + offset <= 2000) {
                        ways += dp[i + 1][t + arr[i] + offset];
                    }
                    dp[i][currentIdx] = (int) ways;
                }
            }
        }
        if (target < -1000 || target > 1000) return 0;
        return dp[0][target + offset];
    }
}
```

### Space Optimization
```java
class Solution {
    public int totalWays(int[] arr, int target) {
        int n = arr.length;
        int offset = 1000;
        int[] prev = new int[2001];

        for (int i = n; i >= 0; i--) {
            int[] curr = new int[2001];
            for (int t = -1000; t <= 1000; t++) {
                int currentIdx = t + offset;
                if (i == n) curr[currentIdx] = (t == 0) ? 1 : 0;
                else {
                    long ways = 0;
                    if (t - arr[i] + offset >= 0 && t - arr[i] + offset <= 2000) {
                        ways += prev[t - arr[i] + offset];
                    }
                    if (t + arr[i] + offset >= 0 && t + arr[i] + offset <= 2000) {
                        ways += prev[t + arr[i] + offset];
                    }
                    curr[currentIdx] = (int) ways;
                }
            }
            prev = curr;
        }
        if (target < -1000 || target > 1000) return 0;
        return prev[target + offset];
    }
}
```
