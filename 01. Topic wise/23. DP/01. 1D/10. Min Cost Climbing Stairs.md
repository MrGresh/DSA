### Memoization
```java
class Solution {
    int[] memo;
    public int minCostClimbingStairs(int[] cost) {
        int n = cost.length;
        memo = new int[n];
        Arrays.fill(memo, -1);
        // The top of the floor can be reached from either the last or second-to-last step
        return Math.min(solve(n - 1, cost), solve(n - 2, cost));
    }

    private int solve(int i, int[] cost) {
        if (i < 0) return 0;
        if (i == 0 || i == 1) return cost[i];
        if (memo[i] != -1) return memo[i];

        return memo[i] = cost[i] + Math.min(solve(i - 1, cost), solve(i - 2, cost));
    }
}
```

### Tabulation
```java
class Solution {
    public int minCostClimbingStairs(int[] cost) {
        int n = cost.length;
        int[] dp = new int[n];

        // Base cases
        dp[0] = cost[0];
        dp[1] = cost[1];

        for (int i = 2; i < n; i++) {
            dp[i] = cost[i] + Math.min(dp[i - 1], dp[i - 2]);
        }

        // The answer is the minimum of the last two steps
        return Math.min(dp[n - 1], dp[n - 2]);
    }
}
```
### Space Optimization
```java
class Solution {
    public int minCostClimbingStairs(int[] cost) {
        int n = cost.length;
        int first = cost[0];
        int second = cost[1];

        for (int i = 2; i < n; i++) {
            int current = cost[i] + Math.min(first, second);
            first = second;
            second = current;
        }

        return Math.min(first, second);
    }
}
```
