class Solution {
    public int maxJump(int[] stones) {
        int n = stones.length;
        // Base case: if only two stones, the max jump is just the distance between them
        if (n == 2) return stones[1] - stones[0];

        int maxDiff = 0;
        
        // We check every stone with the one two steps ahead.
        // This covers both the forward trip (evens) and backward trip (odds).
        for (int i = 0; i < n - 2; i++) {
            maxDiff = Math.max(maxDiff, stones[i + 2] - stones[i]);
        }
        
        return maxDiff;
    }
}
