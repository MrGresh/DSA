### Memoization
```java
class Solution {
    static int[] memo;

    static int countWays(int n) {
        memo = new int[n + 1];
        Arrays.fill(memo, -1);
        return solve(n);
    }

    private static int solve(int n) {
        // Base cases
        if (n == 0) return 1; // 1 way to stay at the start
        if (n < 0) return 0;  // Impossible to reach negative steps
        
        if (memo[n] != -1) return memo[n];

        // Sum of ways from 1, 2, and 3 steps back
        return memo[n] = solve(n - 1) + solve(n - 2) + solve(n - 3);
    }
}
```
### Tabulation
```java
class Solution {
    static int countWays(int n) {
        if (n == 0) return 1;
        if (n == 1) return 1;
        if (n == 2) return 2;
        
        int[] dp = new int[n + 1];
        dp[0] = 1; // Base: 0 steps
        dp[1] = 1; // Base: 1 step (1)
        dp[2] = 2; // Base: 2 steps (1+1, 2)
        
        for (int i = 3; i <= n; i++) {
            dp[i] = dp[i - 1] + dp[i - 2] + dp[i - 3];
        }
        
        return dp[n];
    }
}
```
### Space Optimization
```java
class Solution {
    static int countWays(int n) {
        if (n == 0 || n == 1) return 1;
        if (n == 2) return 2;
        if (n == 3) return 4;

        int a = 1; // f(n-3)
        int b = 1; // f(n-2)
        int c = 2; // f(n-1)
        int current = 0;

        for (int i = 3; i <= n; i++) {
            current = a + b + c;
            a = b;
            b = c;
            c = current;
        }

        return c;
    }
}
```
