**Note:** This is problem is exectly same as finding nth fibonacci number, so that solution can also work here.
### Memoization
```java
class Solution {
    Integer[][] memo;
    int countStrings(int n) {
        memo = new Integer[n][2];
        return rec(n-1, 0) + rec(n-1, 1);
    }
    int rec(int nLeft, int lastDigit) {
        if(nLeft==0) return 1;
        if(memo[nLeft][lastDigit]!=null) return memo[nLeft][lastDigit];
        int ans = rec(nLeft-1, 0);
        if(lastDigit==0) ans += rec(nLeft-1, 1);
        return memo[nLeft][lastDigit] = ans;
    }
}
```
### Tabulation
```java
class Solution {
    int countStrings(int n) {
        int[][] dp = new int[n][2];
        for(int nLeft=0;nLeft<n;nLeft++) {
            for(int lastDigit: new int[] {0, 1}) {
                if(nLeft==0) dp[nLeft][lastDigit] = 1;
                else {
                    int ans = dp[nLeft-1][0];
                    if(lastDigit==0) ans += dp[nLeft-1][1];
                    dp[nLeft][lastDigit] = ans;
                }
            }
        }
        return dp[n-1][0] + dp[n-1][1];
    }
}
```
### Space Optimization
```java
class Solution {
    int countStrings(int n) {
        int[] prev = new int[2];
        for(int nLeft=0;nLeft<n;nLeft++) {
            int[] curr = new int[2];
            for(int lastDigit: new int[] {0, 1}) {
                if(nLeft==0) curr[lastDigit] = 1;
                else {
                    int ans = prev[0];
                    if(lastDigit==0) ans += prev[1];
                    curr[lastDigit] = ans;
                }
            }
            prev = curr;
        }
        return prev[0] + prev[1];
    }
}
```
