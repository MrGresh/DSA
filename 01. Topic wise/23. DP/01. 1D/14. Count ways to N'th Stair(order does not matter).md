### Memoization
```java
class Solution {
    Integer[][] memo;
    int[] steps = {1, 2};

    public int countWays(int n) {
        memo = new Integer[2][n + 1];
        return solve(0, n);
    }

    private int solve(int idx, int remaining) {
        if (remaining == 0) return 1;
        if (idx == 2) return 0;
        if (memo[idx][remaining] != null) return memo[idx][remaining];

        // Skip current step size
        int skip = solve(idx + 1, remaining);
        
        // Use current step size
        int take = 0;
        if (remaining >= steps[idx]) {
            take = solve(idx, remaining - steps[idx]);
        }

        return memo[idx][remaining] = skip + take;
    }
}
```
### Tabulation
```java
class Solution {
    public int countWays(int n) {
        int[][] dp = new int[3][n + 1];
        for (int i = 0; i <= 2; i++) dp[i][0] = 1;

        int[] steps = {1, 2};
        for (int i = 1; i <= 2; i++) {
            for (int j = 1; j <= n; j++) {
                dp[i][j] = dp[i - 1][j]; // Skip
                if (j >= steps[i - 1]) {
                    dp[i][j] += dp[i][j - steps[i - 1]]; // Take
                }
            }
        }
        return dp[2][n];
    }
}
```
### Space Optimization
```java
class Solution {
    public int countWays(int n) {
        // Since we can use any number of 2's from 0 up to n/2,
        // and the rest will be 1's, the total ways is n/2 + 1.
        return (n / 2) + 1;
    }
}
```
