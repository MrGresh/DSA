### Memoization
```java
class Solution {
    long[] memo;
    public long countFriendsPairings(int n) {
        memo = new long[n + 1];
        Arrays.fill(memo, -1);
        return solve(n);
    }

    private long solve(int n) {
        // Base cases
        if (n <= 2) return n;
        
        if (memo[n] != -1) return memo[n];

        // 1. n stays single + 2. n pairs with any of (n-1) friends
        return memo[n] = solve(n - 1) + (n - 1) * solve(n - 2);
    }
}
```
### Tabulation
```java
class Solution {
    public long countFriendsPairings(int n) {
        if (n <= 2) return n;
        
        long[] dp = new long[n + 1];
        dp[1] = 1;
        dp[2] = 2;

        for (int i = 3; i <= n; i++) {
            dp[i] = dp[i - 1] + (i - 1) * dp[i - 2];
        }

        return dp[n];
    }
}
```
### Space Optimized
```java
class Solution {
    public long countFriendsPairings(int n) {
        if (n <= 2) return n;

        long a = 1; // dp[i-2]
        long b = 2; // dp[i-1]
        long res = 0;

        for (int i = 3; i <= n; i++) {
            res = b + (i - 1) * a;
            a = b;
            b = res;
        }

        return res;
    }
}
```
