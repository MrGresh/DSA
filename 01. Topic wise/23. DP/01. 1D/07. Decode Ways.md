### Memoization
```java
class Solution {
    public int numDecodings(String s) {
        int[] memo = new int[s.length()];
        Arrays.fill(memo, -1);
        return solve(0, s, memo);
    }

    private int solve(int i, String s, int[] memo) {
        if (i == s.length()) return 1; // Base case: reached the end successfully
        if (s.charAt(i) == '0') return 0; // Leading zero is invalid
        if (memo[i] != -1) return memo[i];

        // Choice 1: Take one digit
        int ways = solve(i + 1, s, memo);

        // Choice 2: Take two digits (if within bounds and valid)
        if (i + 1 < s.length()) {
            int twoDigit = Integer.parseInt(s.substring(i, i + 2));
            if (twoDigit <= 26) {
                ways += solve(i + 2, s, memo);
            }
        }

        return memo[i] = ways;
    }
}
```

### Tabulation
```java
class Solution {
    public int numDecodings(String s) {
        int n = s.length();
        if (n == 0) return 0;

        int[] dp = new int[n + 1];

        // Base Case: An empty string (reaching the end) counts as 1 valid way
        dp[n] = 1;

        for (int i = n - 1; i >= 0; i--) {
            // A leading zero cannot be decoded as a standalone or start of a pair
            if (s.charAt(i) == '0') {
                dp[i] = 0;
            } else {
                // Choice 1: Decode as a single digit
                dp[i] = dp[i + 1];

                // Choice 2: Decode as a two-digit number (if valid)
                if (i + 1 < n) {
                    // Check if the two-digit number is between 10 and 26
                    int ten = (s.charAt(i) - '0') * 10;
                    int one = (s.charAt(i + 1) - '0');
                    if (ten + one <= 26) {
                        dp[i] += dp[i + 2];
                    }
                }
            }
        }

        return dp[0];
    }
}
```

### Space Optimization
```java
class Solution {
    public int numDecodings(String s) {
        int n = s.length();
        if (n == 0 || s.charAt(0) == '0') return 0;

        // next1 represents dp[i+1], next2 represents dp[i+2]
        int next1 = 1; // Base case: dp[n] = 1
        int next2 = 0; // Base case: dp[n+1] = 0 (imaginary bound)

        for (int i = n - 1; i >= 0; i--) {
            int current = 0;
            
            if (s.charAt(i) != '0') {
                // Choice 1: Single digit
                current = next1;
                
                // Choice 2: Two digits
                if (i + 1 < n) {
                    int ten = (s.charAt(i) - '0') * 10;
                    int one = (s.charAt(i + 1) - '0');
                    if (ten + one <= 26) {
                        current += next2;
                    }
                }
            }
            
            // Shift variables for the next iteration (moving left)
            next2 = next1;
            next1 = current;
        }

        return next1;
    }
}
```
