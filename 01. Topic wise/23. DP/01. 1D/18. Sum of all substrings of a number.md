### Memoization
```java
class Solution {
    static int[] memo;
    
    public static int sumSubstrings(String s) {
        int n = s.length();
        memo = new int[n];
        Arrays.fill(memo, -1);
        
        int totalSum = 0;
        for (int i = 0; i < n; i++) {
            totalSum += solve(i, s);
        }
        return totalSum;
    }
    
    private static int solve(int i, String s) {
        if (i == 0) return s.charAt(0) - '0';
        if (memo[i] != -1) return memo[i];
        
        int digit = s.charAt(i) - '0';
        // Current sum ending at i depends on sum ending at i-1
        memo[i] = solve(i - 1, s) * 10 + digit * (i + 1);
        return memo[i];
    }
}
```
### Tabulation
```java
class Solution {
    public static int sumSubstrings(String s) {
        int n = s.length();
        int[] dp = new int[n];
        
        // Base case: first digit
        dp[0] = s.charAt(0) - '0';
        int totalSum = dp[0];
        
        for (int i = 1; i < n; i++) {
            int digit = s.charAt(i) - '0';
            // dp[i] = sum of all substrings ending at index i
            dp[i] = dp[i - 1] * 10 + digit * (i + 1);
            totalSum += dp[i];
        }
        
        return totalSum;
    }
}
```
### Space Optimization
```java
class Solution {
    public static int sumSubstrings(String s) {
        int n = s.length();
        int totalSum = 0;
        int prevEndSum = 0;
        
        for (int i = 0; i < n; i++) {
            int digit = s.charAt(i) - '0';
            // Update the sum of substrings ending at current index
            int currentEndSum = prevEndSum * 10 + digit * (i + 1);
            
            totalSum += currentEndSum;
            prevEndSum = currentEndSum;
        }
        
        return totalSum;
    }
}
```
