### Memoization
```java
class Solution {
    Integer[][] memo;
    public int maximumSum(int[] arr) {
        int n = arr.length;
        memo = new Integer[n][2];
        
        int maxTotal = arr[0];
        // We must check every possible ending position
        for (int i = 0; i < n; i++) {
            maxTotal = Math.max(maxTotal, solve(i, 0, arr));
            maxTotal = Math.max(maxTotal, solve(i, 1, arr));
        }
        return maxTotal;
    }

    private int solve(int i, int deleted, int[] arr) {
        if (i == 0) return deleted == 0 ? arr[0] : -100000000;
        if (memo[i][deleted] != null) return memo[i][deleted];

        if (deleted == 0) {
            // Standard Kadane: Start fresh or extend previous
            return memo[i][0] = Math.max(arr[i], arr[i] + solve(i - 1, 0, arr));
        } else {
            // State 1: Either we delete current arr[i], or we extend a previous deletion
            return memo[i][1] = Math.max(solve(i - 1, 0, arr), arr[i] + solve(i - 1, 1, arr));
        }
    }
}
```

### Tabulation
```java
class Solution {
    public int maximumSum(int[] arr) {
        int n = arr.length;
        if (n == 1) return arr[0];

        int[] dp0 = new int[n]; // No deletions
        int[] dp1 = new int[n]; // One deletion

        dp0[0] = arr[0];
        dp1[0] = -1000000; // Impossible to delete 1 and have a non-empty subarray at index 0

        int maxTotal = arr[0];

        for (int i = 1; i < n; i++) {
            // Standard Kadane
            dp0[i] = Math.max(arr[i], dp0[i - 1] + arr[i]);
            
            // Delete current OR keep current but use a previous deletion
            dp1[i] = Math.max(dp0[i - 1], dp1[i - 1] + arr[i]);
            
            maxTotal = Math.max(maxTotal, Math.max(dp0[i], dp1[i]));
        }

        return maxTotal;
    }
}
```

### Space Optimization
```java
class Solution {
    public int maximumSum(int[] arr) {
        int n = arr.length;
        int maxNoDeletion = arr[0];
        int maxOneDeletion = -1000000; 
        int overallMax = arr[0];

        for (int i = 1; i < n; i++) {
            // maxOneDeletion must be updated using the OLD maxNoDeletion
            maxOneDeletion = Math.max(maxNoDeletion, maxOneDeletion + arr[i]);
            maxNoDeletion = Math.max(arr[i], maxNoDeletion + arr[i]);
            
            overallMax = Math.max(overallMax, Math.max(maxNoDeletion, maxOneDeletion));
        }

        return overallMax;
    }
}
```
