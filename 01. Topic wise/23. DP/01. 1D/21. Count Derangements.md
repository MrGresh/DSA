### Memoization
```java
class Solution {
    Integer[] memo;
    public int derangeCount(int n) {
        memo = new Integer[n + 1];
        return solve(n);
    }

    private int solve(int n) {
        if (n == 1) return 0;
        if (n == 2 || n==0) return 1;

        if (memo[n] != null) return memo[n];
        
        return memo[n] = (n - 1) * (solve(n - 1) + solve(n - 2));
    }
}
```
### Tabulation
```java
class Solution {
    public int derangeCount(int N) {
        int[] dp = new int[N + 1];
        for(int n=0;n<=N;n++) {
            if (n == 1) dp[n] = 0;
            else if (n == 2 || n==0) dp[n] = 1;
            else dp[n] = (n-1)*(dp[n-1]+dp[n-2]);
        }
        return dp[N];
    }
};
```
### Space Optimization
```java
class Solution {
    public int derangeCount(int N) {
        int prev1=0, prev2=0;
        for(int n=0;n<=N;n++) {
            int curr=0;
            if (n == 1) curr = 0;
            else if (n == 2 || n==0) curr = 1;
            else curr = (n-1)*(prev1+prev2);
            prev2=prev1;
            prev1=curr;
        }
        return prev1;
    }
};
```
