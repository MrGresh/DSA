### Space Optimization
```java
class Solution {
    public int rob(int[] arr) {
        int n = arr.length;
        // Handle edge cases
        if (n == 0) return 0;
        if (n == 1) return arr[0];

        // arr1 excludes first house
        List<Integer> arr1 = new ArrayList<>();
        // arr2 excludes last house
        List<Integer> arr2 = new ArrayList<>();
        
        for (int i = 0; i < n; i++) {
            if (i != 0) arr1.add(arr[i]);
            if (i != n - 1) arr2.add(arr[i]);
        }

        // Return maximum of both cases
        return Math.max(solve(arr1), solve(arr2));
    }
    public int solve(List<Integer> nums) {
        // Handle edge case for empty input
        if (nums.size() == 0) return 0;

        // Initialize two tracking sums
        int prev2 = 0;
        int prev = nums.get(0);

        // Traverse through the array
        for (int i = 1; i < nums.size(); i++) {
            // Include current by adding it to two steps back
            int include = nums.get(i) + prev2;

            // Exclude current by taking previous best
            int exclude = prev;

            // Choose max of include and exclude
            int curr = Math.max(include, exclude);

            // Update tracking variables
            prev2 = prev;
            prev = curr;
        }

        // Final result is stored in prev
        return prev;         
    }
}
```
