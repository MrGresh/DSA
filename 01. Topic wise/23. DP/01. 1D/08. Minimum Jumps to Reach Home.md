### BFS
```java
class Solution {
    public int minimumJumps(int[] forbidden, int a, int b, int x) {
        Set<Integer> bad = new HashSet<>();
        for (int f : forbidden) bad.add(f);
        
        // Queue stores: {current_position, can_jump_backward (0: no, 1: yes)}
        Queue<int[]> queue = new LinkedList<>();
        queue.offer(new int[]{0, 1}); 
        
        // visited[position][last_was_backward]
        boolean[][] visited = new boolean[6001][2];
        visited[0][0] = true;
        
        int jumps = 0;
        while (!queue.isEmpty()) {
            int size = queue.size();
            while (size-- > 0) {
                int[] curr = queue.poll();
                int pos = curr[0];
                int canBack = curr[1];
                
                if (pos == x) return jumps;
                
                // 1. Jump Forward
                int forward = pos + a;
                if (forward <= 6000 && !bad.contains(forward) && !visited[forward][0]) {
                    visited[forward][0] = true;
                    queue.offer(new int[]{forward, 1});
                }
                
                // 2. Jump Backward
                int backward = pos - b;
                if (canBack == 1 && backward >= 0 && !bad.contains(backward) && !visited[backward][1]) {
                    visited[backward][1] = true;
                    queue.offer(new int[]{backward, 0});
                }
            }
            jumps++;
        }
        return -1;
    }
}
```

### Memoization
```java
class Solution {
    int[][] memo;
    Set<Integer> bad;
    int limit = 6000;

    public int minimumJumps(int[] forbidden, int a, int b, int x) {
        bad = new HashSet<>();
        for (int f : forbidden) bad.add(f);
        memo = new int[limit + 1][2];
        for (int[] row : memo) Arrays.fill(row, -1);
        
        int result = solve(0, 1, a, b, x);
        return result >= 1e9 ? -1 : result;
    }

    private int solve(int pos, int canBack, int a, int b, int x) {
        if (pos == x) return 0;
        if (pos < 0 || pos > limit || bad.contains(pos) || memo[pos][canBack] != -1) {
            // If already visiting this state in current recursion, return infinity to avoid cycles
            return (int)1e9; 
        }
        
        // Mark as "visiting" to prevent infinite loops in recursion
        memo[pos][canBack] = (int)1e9;
        
        // Forward jump is always an option
        int res = 1 + solve(pos + a, 1, a, b, x);
        
        // Backward jump only if allowed
        if (canBack == 1) {
            res = Math.min(res, 1 + solve(pos - b, 0, a, b, x));
        }
        
        return memo[pos][canBack] = res;
    }
}
```
