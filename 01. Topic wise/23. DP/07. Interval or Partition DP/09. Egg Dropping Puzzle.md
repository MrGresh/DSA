### Memoization
```java
class Solution {
    static int[][] memo;

    static int eggDrop(int n, int k) {
        memo = new int[n + 1][k + 1];
        for (int[] row : memo) Arrays.fill(row, -1);
        return solve(n, k);
    }

    private static int solve(int n, int k) {
        // Base Cases
        if (k == 0 || k == 1) return k; // 0 floors = 0 drops, 1 floor = 1 drop
        if (n == 1) return k;           // 1 egg = must drop sequentially (k drops)
        
        if (memo[n][k] != -1) return memo[n][k];

        int minMoves = Integer.MAX_VALUE;

        // Try dropping from every floor 'x' from 1 to k
        for (int x = 1; x <= k; x++) {
            // Max represents the worst case between breaking and surviving
            int res = 1 + Math.max(solve(n - 1, x - 1), solve(n, k - x));
            minMoves = Math.min(minMoves, res);
        }

        return memo[n][k] = minMoves;
    }
}
```

### Tabulation
```java
class Solution {
    static int eggDrop(int n, int k) {
        int[][] dp = new int[n + 1][k + 1];

        // Base cases
        for (int i = 1; i <= n; i++) {
            dp[i][1] = 1; // 1 floor
            dp[i][0] = 0; // 0 floors
        }
        for (int j = 1; j <= k; j++) {
            dp[1][j] = j; // 1 egg
        }

        for (int i = 2; i <= n; i++) {
            for (int j = 2; j <= k; j++) {
                dp[i][j] = Integer.MAX_VALUE;
                for (int x = 1; x <= j; x++) {
                    int res = 1 + Math.max(dp[i - 1][x - 1], dp[i][j - x]);
                    dp[i][j] = Math.min(dp[i][j], res);
                }
            }
        }
        return dp[n][k];
    }
}
```

### Space Optimized
```java
class Solution {
    static int eggDrop(int n, int k) {
        // dp[m][n] is the max floors we can check with m moves and n eggs
        int[][] dp = new int[k + 1][n + 1];
        int moves = 0;

        while (dp[moves][n] < k) {
            moves++;
            for (int j = 1; j <= n; j++) {
                dp[moves][j] = dp[moves - 1][j - 1] + dp[moves - 1][j] + 1;
            }
        }
        return moves;
    }
}
```
