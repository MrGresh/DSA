### Memoization
```java
class Solution {
    static int[] memo;

    public static int findCatalan(int n) {
        memo = new int[n + 1];
        Arrays.fill(memo, -1);
        return solve(n);
    }

    private static int solve(int n) {
        // Base case: C0 = 1, C1 = 1
        if (n <= 1) return 1;
        if (memo[n] != -1) return memo[n];

        int res = 0;
        for (int i = 0; i < n; i++) {
            res += solve(i) * solve(n - 1 - i);
        }
        return memo[n] = res;
    }
}
```
### Tabulation
```java
class Solution {
    public static int findCatalan(int n) {
        int[] dp = new int[n + 1];
        
        // Base cases
        dp[0] = 1;
        if (n >= 1) dp[1] = 1;

        // Fill table for values from 2 to n
        for (int i = 2; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                // dp[i] = dp[0]*dp[i-1] + dp[1]*dp[i-2] + ... + dp[i-1]*dp[0]
                dp[i] += dp[j] * dp[i - 1 - j];
            }
        }
        
        return dp[n];
    }
}
```
### Space Optimized (Mathematical Approach)
```java
class Solution {
    public static int findCatalan(int n) {
        long res = 1;
        // Using the iterative formula for Catalan numbers: 
        // C(n+1) = C(n) * (4n + 2) / (n + 2)
        for (int i = 0; i < n; i++) {
            res = res * 2 * (2 * i + 1) / (i + 2);
        }
        return (int) res;
    }
}
```
