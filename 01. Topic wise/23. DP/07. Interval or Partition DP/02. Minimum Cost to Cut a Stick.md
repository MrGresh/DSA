### Memoization
```java
class Solution {
    public int minCost(int n, int[] cuts) {
        // Add the two ends of the stick to the cuts list
        int[] newCuts = new int[cuts.length+2];
        for(int i=0;i<cuts.length;i++) newCuts[i] = cuts[i];
        newCuts[cuts.length] = n;
        newCuts[cuts.length+1] = 0;
        Arrays.sort(newCuts);

        int c = newCuts.length - 2; // number of cuts excluding ends

        // Create DP table initialized with -1
        int[][] dp = new int[c + 2][c + 2];
        for (int[] row : dp) {
            Arrays.fill(row, -1);
        }

        // Compute minimum cost to make all cuts between indices 1 and c
        return findMinimumCost(1, c, newCuts, dp);
    }

    // Recursive function with memoization to find minimum cost between indices i and j in cuts
    public int findMinimumCost(int i, int j, int[] cuts, int[][] dp) {
        // Base case: no cuts to consider
        if (i > j) {
            return 0;
        }

        // Return precomputed result if available
        if (dp[i][j] != -1) {
            return dp[i][j];
        }

        int mini = Integer.MAX_VALUE;

        // Try cutting at every position between i and j
        for (int ind = i; ind <= j; ind++) {
            // Cost of making current cut plus left and right subproblems
            int ans = cuts[j + 1] - cuts[i - 1] +
                      findMinimumCost(i, ind - 1, cuts, dp) +
                      findMinimumCost(ind + 1, j, cuts, dp);

            // Update minimum cost
            mini = Math.min(mini, ans);
        }

        // Store and return minimum cost found
        return dp[i][j] = mini;
    }
}
```

### Tabulation
```java
class Solution {
    public int minCost(int n, int[] cuts) {
        int[] newCuts = new int[cuts.length+2];
        for(int i=0;i<cuts.length;i++) newCuts[i] = cuts[i];
        newCuts[cuts.length] = n;
        newCuts[cuts.length+1] = 0;
        Arrays.sort(newCuts);

        int c = newCuts.length - 2;

        // Create DP table initialized with 0
        int[][] dp = new int[c + 2][c + 2];

        // Fill DP table for all segments
        for (int i = c; i >= 1; i--) {
            for (int j = i; j <= c; j++) {
                int mini = Integer.MAX_VALUE;

                // Try making a cut at every position between i and j
                for (int ind = i; ind <= j; ind++) {
                    // Cost of current cut plus left and right subproblems
                    int ans = newCuts[j + 1] - newCuts[i - 1] + dp[i][ind - 1] + dp[ind + 1][j];

                    // Update minimum cost
                    mini = Math.min(mini, ans);
                }

                // Store minimum cost in DP table
                dp[i][j] = mini;
            }
        }

        // Return minimum cost to cut between indices 1 and c
        return dp[1][c];
    }
}
```
