### Memoization
```java
class Solution {
    public int knapsack(int W, int val[], int wt[]) {
        int n = val.length;
        int[][] dp = new int[n][W + 1];
        for (int[] row : dp) Arrays.fill(row, -1);
        return solve(n - 1, W, val, wt, dp);
    }

    private int solve(int ind, int cap, int[] val, int[] wt, int[][] dp) {
        // Base case: No capacity or no items left
        if (ind == 0) {
            if (wt[0] <= cap) return val[0];
            else return 0;
        }

        if (dp[ind][cap] != -1) return dp[ind][cap];

        // Choice 1: Don't take the item
        int notTake = solve(ind - 1, cap, val, wt, dp);

        // Choice 2: Take the item (if weight allows)
        int take = Integer.MIN_VALUE;
        if (wt[ind] <= cap) {
            take = val[ind] + solve(ind - 1, cap - wt[ind], val, wt, dp);
        }

        return dp[ind][cap] = Math.max(take, notTake);
    }
}
```

### Tabulation
```java
class Solution {
    public int knapsack(int W, int val[], int wt[]) {
        int n = val.length;
        int[][] dp = new int[n][W + 1];

        // Base case: First item
        for (int w = wt[0]; w <= W; w++) {
            dp[0][w] = val[0];
        }

        for (int i = 1; i < n; i++) {
            for (int cap = 0; cap <= W; cap++) {
                int notTake = dp[i - 1][cap];
                
                int take = Integer.MIN_VALUE;
                if (wt[i] <= cap) {
                    take = val[i] + dp[i - 1][cap - wt[i]];
                }

                dp[i][cap] = Math.max(notTake, take);
            }
        }
        return dp[n - 1][W];
    }
}
```

### Space Optimization
```java
class Solution {
    public int knapsack(int W, int val[], int wt[]) {
        int n = val.length;
        int[] dp = new int[W + 1];

        // Base case: First item
        for (int w = wt[0]; w <= W; w++) {
            dp[w] = val[0];
        }

        // Process remaining items
        for (int i = 1; i < n; i++) {
            // Iterate backwards to ensure we use values from the previous item's row
            for (int cap = W; cap >= 0; cap--) {
                int notTake = dp[cap];
                
                int take = Integer.MIN_VALUE;
                if (wt[i] <= cap) {
                    take = val[i] + dp[cap - wt[i]];
                }

                dp[cap] = Math.max(notTake, take);
            }
        }
        return dp[W];
    }
}
```
