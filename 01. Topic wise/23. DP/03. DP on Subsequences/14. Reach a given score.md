### Memoization
```java
class Geeks {
    long[][] memo;
    int[] scores = {3, 5, 10};

    public long count(int n) {
        memo = new long[3][n + 1];
        for (long[] row : memo) Arrays.fill(row, -1);
        return solve(0, n);
    }

    private long solve(int idx, int remaining) {
        if (remaining == 0) return 1;
        if (idx == 3) return 0;
        if (memo[idx][remaining] != -1) return memo[idx][remaining];

        // Option 1: Don't use this score move, move to next
        long res = solve(idx + 1, remaining);

        // Option 2: Use this score move (and stay at same index to allow reuse)
        if (remaining >= scores[idx]) {
            res += solve(idx, remaining - scores[idx]);
        }

        return memo[idx][remaining] = res;
    }
}
```
### Tabulation
```java
class Geeks {
    public long count(int n) {
        int[] scores = {3, 5, 10};
        // dp[number_of_score_types][target_score]
        long[][] dp = new long[4][n + 1];

        // Base Case: There is 1 way to make score 0 (using no scores)
        for (int i = 0; i <= 3; i++) {
            dp[i][0] = 1;
        }

        // Mirroring the recursive logic:
        for (int i = 1; i <= 3; i++) { // i is the score type (1st, 2nd, 3rd)
            int currentScore = scores[i - 1];
            for (int j = 1; j <= n; j++) {
                
                // 1. Skip current score: Same as solve(idx + 1, remaining)
                dp[i][j] = dp[i - 1][j];

                // 2. Take current score: Same as solve(idx, remaining - currentScore)
                if (j >= currentScore) {
                    dp[i][j] += dp[i][j - currentScore];
                }
            }
        }

        return dp[3][n];
    }
}
```

### Space Optimization
```java
class Geeks {
    public long count(int n) {
        long[] dp = new long[n + 1];
        
        // Base case: 1 way to reach score 0 (empty set)
        dp[0] = 1;

        // Pick 3 first, then 5, then 10 to ensure distinct combinations
        int[] moves = {3, 5, 10};

        for (int move : moves) {
            for (int i = move; i <= n; i++) {
                dp[i] += dp[i - move];
            }
        }

        return dp[n];
    }
}
```
