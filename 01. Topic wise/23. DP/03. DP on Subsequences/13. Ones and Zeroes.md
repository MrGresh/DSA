### Memoization
```java
class Solution {
    Integer[][][] memo;
    
    public int findMaxForm(String[] strs, int m, int n) {
        int len = strs.length;
        memo = new Integer[len][m + 1][n + 1];
        
        // Pre-calculate counts of 0s and 1s for each string
        int[][] counts = new int[len][2];
        for (int i = 0; i < len; i++) {
            for (char c : strs[i].toCharArray()) {
                counts[i][c - '0']++;
            }
        }
        
        return solve(0, m, n, counts);
    }
    
    private int solve(int i, int m, int n, int[][] counts) {
        if (i == counts.length) return 0;
        if (memo[i][m][n] != null) return memo[i][m][n];
        
        // Option 1: Skip the current string
        int res = solve(i + 1, m, n, counts);
        
        // Option 2: Take the current string (if it fits)
        int zeros = counts[i][0];
        int ones = counts[i][1];
        if (m >= zeros && n >= ones) {
            res = Math.max(res, 1 + solve(i + 1, m - zeros, n - ones, counts));
        }
        
        return memo[i][m][n] = res;
    }
}
```

### Tabulation
```java
class Solution {
    public int findMaxForm(String[] strs, int m, int n) {
        int len = strs.length;
        int[][][] dp = new int[len + 1][m + 1][n + 1];
        
        for (int i = 1; i <= len; i++) {
            int zeros = 0, ones = 0;
            for (char c : strs[i-1].toCharArray()) {
                if (c == '0') zeros++; else ones++;
            }
            
            for (int j = 0; j <= m; j++) {
                for (int k = 0; k <= n; k++) {
                    // Option 1: Skip
                    dp[i][j][k] = dp[i-1][j][k];
                    // Option 2: Take
                    if (j >= zeros && k >= ones) {
                        dp[i][j][k] = Math.max(dp[i][j][k], 1 + dp[i-1][j - zeros][k - ones]);
                    }
                }
            }
        }
        return dp[len][m][n];
    }
}
```

### Space Optimization
```java
class Solution {
    public int findMaxForm(String[] strs, int m, int n) {
        // dp[i][j] = max subset size with i zeros and j ones
        int[][] dp = new int[m + 1][n + 1];
        
        for (String s : strs) {
            int zeros = 0, ones = 0;
            for (char c : s.toCharArray()) {
                if (c == '0') zeros++; else ones++;
            }
            
            // Iterate backwards to avoid using the same string twice
            for (int i = m; i >= zeros; i--) {
                for (int j = n; j >= ones; j--) {
                    dp[i][j] = Math.max(dp[i][j], 1 + dp[i - zeros][j - ones]);
                }
            }
        }
        
        return dp[m][n];
    }
}
```
