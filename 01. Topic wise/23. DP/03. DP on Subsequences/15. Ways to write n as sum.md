### Memoization
```java
class Solution {
    int MOD = 1000000007;
    Long[][] memo;

    public int countWays(int n) {
        memo = new Long[n + 1][n + 1];
        // Calculate total partitions of n using numbers 1 to n
        long total = solve(1, n, n);
        
        // Subtract 1 to exclude the partition {n} (since we need "two or more")
        return (int) ((total - 1 + MOD) % MOD);
    }

    private long solve(int i, int target, int n) {
        // Base Case: Target reached
        if (target == 0) return 1;
        // Base Case: No more numbers to use or target impossible
        if (i > n || target < 0) return 0;
        
        if (memo[i][target] != null) return memo[i][target];

        // 1. Skip i: Move to the next integer
        long skip = solve(i + 1, target, n);
        
        // 2. Take i: Subtract from target, stay at i (unbounded reuse)
        long take = solve(i, target - i, n);

        return memo[i][target] = (skip + take) % MOD;
    }
}
```
### Tabulation
```java
class Solution {
    int MOD = 1000000007;

    public int countWays(int n) {
        long[][] dp = new long[n + 1][n + 1];

        // Base case: 1 way to make sum 0 (empty set)
        for (int i = 0; i <= n; i++) dp[i][0] = 1;

        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                // The "Skip" choice (inherit from the row above)
                dp[i][j] = dp[i - 1][j];

                // The "Take" choice (look back in the current row)
                if (j >= i) {
                    dp[i][j] = (dp[i][j] + dp[i][j - i]) % MOD;
                }
            }
        }

        // Subtract 1 for the {n} case
        return (int) ((dp[n][n] - 1 + MOD) % MOD);
    }
}
```

### Space Optimization
```java
class Solution {
    int MOD = 1000000007;

    public int countWays(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 1;

        // Iterate through each "coin" (integer) from 1 to n
        for (int i = 1; i <= n; i++) {
            // Unbounded: Iterate forward to allow multiple uses of 'i'
            for (int j = i; j <= n; j++) {
                dp[j] = (dp[j] + dp[j - i]) % MOD;
            }
        }

        // Subtract 1 for the {n} case
        return (int) ((dp[n] - 1 + MOD) % MOD);
    }
}
```
