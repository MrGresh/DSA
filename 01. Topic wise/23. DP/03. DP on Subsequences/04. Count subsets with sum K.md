### Memoization
```java
class Solution {
    Integer[][] memo;
    public int perfectSum(int[] arr, int target) {
        int max = Integer.MIN_VALUE;
        for(int x: arr) max = Math.max(max, x);
        memo = new Integer[arr.length][target+1];
        return rec(arr, target, 0);
    }
    int rec(int[] arr, int k, int i) {
        if(i==arr.length) {
            return k==0 ? 1 : 0;
        }
        if(memo[i][k]!=null) return memo[i][k];
        int ans = 0;
        ans += rec(arr, k, i+1);
        if(k-arr[i]>=0) ans += rec(arr, k-arr[i], i+1);
        return memo[i][k] = ans;
    }
}
```

### Tabulation
```java
class Solution {
    public int perfectSum(int[] arr, int target) {
        int[][] dp = new int[arr.length+1][target+1];
        for(int i=arr.length;i>=0;i--) {
            for(int k=target;k>=0;k--) {
                if(i==arr.length) {
                    dp[i][k] = k==0 ? 1 : 0;
                } else {
                    int ans = 0;
                    ans += dp[i+1][k];
                    if(k-arr[i]>=0) ans += dp[i+1][k-arr[i]];
                    dp[i][k] = ans;
                }
            }
        }
        return dp[0][target];
    }
}
```

### Space Optimization
```java
class Solution {
    public int perfectSum(int[] arr, int target) {
        int[] prev = new int[target+1];
        for(int i=arr.length;i>=0;i--) {
            int[] curr = new int[target+1];
            for(int k=target;k>=0;k--) {
                if(i==arr.length) {
                    curr[k] = k==0 ? 1 : 0;
                } else {
                    int ans = 0;
                    ans += prev[k];
                    if(k-arr[i]>=0) ans += prev[k-arr[i]];
                    curr[k] = ans;
                }
            }
            prev = curr;
        }
        return prev[target];
    }
}
```
