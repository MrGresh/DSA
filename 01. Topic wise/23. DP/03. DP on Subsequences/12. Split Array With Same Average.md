### Memoization
```java
class Solution {
    byte[][][] memo;
    int[] nums;

    public boolean splitArraySameAverage(int[] nums) {
        int n = nums.length, totalSum = 0;
        for (int x : nums) totalSum += x;
        this.nums = nums;
        
        // We only need to check up to n/2 and totalSum/2
        memo = new byte[n + 1][n / 2 + 1][totalSum / 2 + 1];

        for (int k = 1; k <= n / 2; k++) {
            if ((totalSum * k) % n == 0) {
                int target = (totalSum * k) / n;
                if (solve(0, k, target)) return true;
            }
        }
        return false;
    }

    private boolean solve(int i, int k, int s) {
        if (k == 0) return s == 0;
        if (i == nums.length || s < 0) return false;
        if (memo[i][k][s] != 0) return memo[i][k][s] == 1;

        // Take or Skip logic
        boolean res = solve(i + 1, k - 1, s - nums[i]) || solve(i + 1, k, s);
        
        memo[i][k][s] = (byte) (res ? 1 : 2);
        return res;
    }
}
```

### Tabulation
```java
class Solution {
    public boolean splitArraySameAverage(int[] nums) {
        int n = nums.length, totalSum = 0;
        for (int x : nums) totalSum += x;
        int maxK = n / 2;
        int maxS = totalSum / 2;

        boolean[][][] dp = new boolean[n + 1][maxK + 1][maxS + 1];
        for (int i = 0; i <= n; i++) dp[i][0][0] = true;

        for (int i = 1; i <= n; i++) {
            int val = nums[i - 1];
            for (int k = 1; k <= maxK; k++) {
                for (int s = 0; s <= maxS; s++) {
                    // Mirror of 'Skip': Inherit from the row above
                    dp[i][k][s] = dp[i - 1][k][s];
                    // Mirror of 'Take': Check if previous sum was possible
                    if (s >= val) {
                        dp[i][k][s] = dp[i][k][s] || dp[i - 1][k - 1][s - val];
                    }
                }
            }
        }

        // Final check for the average property
        for (int k = 1; k <= maxK; k++) {
            if ((totalSum * k) % n == 0 && dp[n][k][(totalSum * k) / n]) return true;
        }
        return false;
        }
}
```

### Space Optimized
```java
class Solution {
    public boolean splitArraySameAverage(int[] nums) {
        int n = nums.length, totalSum = 0;
        for (int x : nums) totalSum += x;
        int maxK = n / 2;
        int maxS = totalSum / 2;

        // Only 2D now: dp[elements_count][current_sum]
        boolean[][] dp = new boolean[maxK + 1][maxS + 1];
        dp[0][0] = true;

        for (int val : nums) {
            for (int k = maxK; k >= 1; k--) {
                for (int s = maxS; s >= val; s--) {
                    if (dp[k - 1][s - val]) {
                        dp[k][s] = true;
                    }
                }
            }
        }

        for (int k = 1; k <= maxK; k++) {
            if ((totalSum * k) % n == 0) {
                int target = (totalSum * k) / n;
                if (target <= maxS && dp[k][target]) return true;
            }
        }
        return false;
    }
}
```
