### Memoization
```java
class Solution {
    Integer[][][] memo;
    public int maxProfit(int[] prices) {
        memo = new Integer[prices.length+1][2][3];
        return rec(prices, 0, true, 2);
    }
    int rec(int[] prices, int i, boolean canBuy, int k) {
        if(i==prices.length || k==0) return 0;
        if(memo[i][canBuy ? 1 : 0][k]!=null) return memo[i][canBuy ? 1 : 0][k];
        int val = rec(prices, i+1, canBuy, k); // do nothing
        if(canBuy) val = Math.max(val, -prices[i] + rec(prices, i+1, false, k)); // buy
        else val = Math.max(val, prices[i] + rec(prices, i+1, true, k-1)); // sell
        return memo[i][canBuy ? 1 : 0][k] = val;
    }
}
```

### Tabulation
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        // dp[day][buy_status][transactions_left]
        int[][][] dp = new int[n + 1][2][3];

        for (int i = n - 1; i >= 0; i--) {
            for (int buy = 0; buy <= 1; buy++) {
                for (int k = 1; k <= 2; k++) {
                    if (buy == 1) { // Can buy
                        dp[i][buy][k] = Math.max(
                            -prices[i] + dp[i + 1][0][k], // Buy
                            dp[i + 1][1][k]               // Skip
                        );
                    } else { // Must sell
                        dp[i][buy][k] = Math.max(
                            prices[i] + dp[i + 1][1][k - 1], // Sell
                            dp[i + 1][0][k]                  // Skip
                        );
                    }
                }
            }
        }
        return dp[0][1][2]; // Day 0, Ready to buy, 2 transactions left
    }
}
```

### Space Optimization
```java
class Solution {
    public int maxProfit(int[] prices) {
        int[][] ahead = new int[2][3];
        int[][] cur = new int[2][3];

        for (int i = prices.length - 1; i >= 0; i--) {
            for (int buy = 0; buy <= 1; buy++) {
                for (int k = 1; k <= 2; k++) {
                    if (buy == 1) {
                        cur[buy][k] = Math.max(-prices[i] + ahead[0][k], ahead[1][k]);
                    } else {
                        cur[buy][k] = Math.max(prices[i] + ahead[1][k - 1], ahead[0][k]);
                    }
                }
            }
            // Move current values to ahead for the next day iteration
            for (int r = 0; r < 2; r++) System.arraycopy(cur[r], 0, ahead[r], 0, 3);
        }
        return ahead[1][2];
    }
}
```
