### Memoization
```java
class Solution {
    Integer[][][] memo;
    public int maxProfit(int k, int[] prices) {
        memo = new Integer[prices.length+1][2][k+1];
        return rec(prices, k, 0, 1);
    }
    int rec(int[] prices, int k, int i, int canBuy) {
        if(k==0 || i==prices.length) return 0;
        if(memo[i][canBuy][k]!=null) return memo[i][canBuy][k];
        int val = rec(prices, k, i+1, canBuy); // do nothing
        if(canBuy==1) val = Math.max(val, -prices[i] + rec(prices, k, i+1, 1-canBuy));
        else val = Math.max(val, prices[i] + rec(prices, k-1, i+1, 1-canBuy));
        return memo[i][canBuy][k] = val;
    }
}
```

### Tabulation
```java
class Solution {
    public int maxProfit(int K, int[] prices) {
        int[][][] dp = new int[prices.length+1][2][K+1];
        for(int i=prices.length;i>=0;i--) {
            for(int canBuy : new int[] {1, 0}) {
                for(int k=K;k>=0;k--) {
                    if(k==0 || i==prices.length) dp[i][canBuy][k] = 0;
                    else {
                        int val = dp[i+1][canBuy][k]; // do nothing
                        if(canBuy==1) val = Math.max(val, -prices[i] + dp[i+1][1-canBuy][k]);
                        else val = Math.max(val, prices[i] + dp[i+1][1-canBuy][k-1]);
                        dp[i][canBuy][k] = val;
                    }
                }
            }
        }
        return dp[0][1][K];
    }
}
```

### Space Optimization
```java
class Solution {
    public int maxProfit(int K, int[] prices) {
        int[][] prev = new int[2][K+1];
        for(int i=prices.length;i>=0;i--) {
            int[][] curr = new int[2][K+1];
            for(int canBuy : new int[] {1, 0}) {
                for(int k=K;k>=0;k--) {
                    if(k==0 || i==prices.length) curr[canBuy][k] = 0;
                    else {
                        int val = prev[canBuy][k]; // do nothing
                        if(canBuy==1) val = Math.max(val, -prices[i] + prev[1-canBuy][k]);
                        else val = Math.max(val, prices[i] + prev[1-canBuy][k-1]);
                        curr[canBuy][k] = val;
                    }
                }
            }
            prev = curr;
        }
        return prev[1][K];
    }
}
```
