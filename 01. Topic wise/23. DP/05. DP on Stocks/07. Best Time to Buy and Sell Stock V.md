### Memoization
```java
class Solution {
    Long[][][] memo;
    public long maximumProfit(int[] prices, int k) {
        memo = new Long[prices.length+1][k+1][3];
        return rec(0, prices, k, 0);
    }
    long rec(int i, int[] prices, int k, int state) {
        if(k==0) return 0;
        if(i==prices.length) return state==2 ? Long.MIN_VALUE : 0;
        if(memo[i][k][state]!=null) return memo[i][k][state];

        long val = rec(i+1, prices, k, state); // doing nothing
        if(state==0) {
            val = Math.max(val, -prices[i] + rec(i+1, prices, k, 1)); // Normal (buy)
            val = Math.max(val, prices[i] + rec(i+1, prices, k, 2)); // Short selling (sell)
        } 
        else if(state==1) val = Math.max(val, prices[i] + rec(i+1, prices, k-1, 0)); // Normal (sell)
        else val = Math.max(val, -prices[i] + rec(i+1, prices, k-1, 0)); // Short selling (buy)

        return memo[i][k][state] = val;
    }
}
```

### Tabulation
```java
class Solution {
    public long maximumProfit(int[] prices, int K) {
        long[][][] dp = new long[prices.length+1][K+1][3];
        for(int i=prices.length;i>=0;i--) {
            for(int k=K;k>=0;k--) {
                for(int state : new int[] {0, 1, 2}) {
                    if(k==0) dp[i][k][state] = 0;
                    else if(i==prices.length) dp[i][k][state] = state==2 ? Long.MIN_VALUE : 0;
                    else {
                        long val = dp[i+1][k][state]; // doing nothing
                        if(state==0) {
                            val = Math.max(val, -prices[i] + dp[i+1][k][1]); // Normal (buy)
                            val = Math.max(val, prices[i] + dp[i+1][k][2]); // Short selling (sell)
                        } 
                        else if(state==1) val = Math.max(val, prices[i] + dp[i+1][k-1][0]); // Normal (sell)
                        else val = Math.max(val, -prices[i] + dp[i+1][k-1][0]); // Short selling (buy)

                        dp[i][k][state] = val;
                    }
                }
            }
        }
        return dp[0][K][0];
    }
}
```

### Space Optimization

```java
class Solution {
    public long maximumProfit(int[] prices, int K) {
        long[][] prev = new long[K+1][3];
        for(int i=prices.length;i>=0;i--) {
            long[][] curr = new long[K+1][3];
            for(int k=K;k>=0;k--) {
                for(int state : new int[] {0, 1, 2}) {
                    if(k==0) curr[k][state] = 0;
                    else if(i==prices.length) curr[k][state] = state==2 ? Long.MIN_VALUE : 0;
                    else {
                        long val = prev[k][state]; // doing nothing
                        if(state==0) {
                            val = Math.max(val, -prices[i] + prev[k][1]); // Normal (buy)
                            val = Math.max(val, prices[i] + prev[k][2]); // Short selling (sell)
                        } 
                        else if(state==1) val = Math.max(val, prices[i] + prev[k-1][0]); // Normal (sell)
                        else val = Math.max(val, -prices[i] + prev[k-1][0]); // Short selling (buy)

                        curr[k][state] = val;
                    }
                }
            }
            prev = curr;
        }
        return prev[K][0];
    }
}
```
