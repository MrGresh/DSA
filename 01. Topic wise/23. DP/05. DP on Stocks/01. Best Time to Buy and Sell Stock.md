### Optimized
```java
class Solution {
    public int maxProfit(int[] prices) {
        int minPrice = Integer.MAX_VALUE;
        int maxProfit = 0;
        for (int price : prices) {
            if (price < minPrice) minPrice = price;
            else maxProfit = Math.max(maxProfit, price - minPrice);
        }
        return maxProfit;
    }
}
```
### Brute Force
```java
class Solution {
    Map<String, Integer> dp;
    public int maxProfit(int[] prices) {
        dp = new HashMap<>();
        return rec(prices, 0, prices.length-1);
    }
    int rec(int[] prices, int l, int r) {
        if(l>=r) return 0;
        String key = l+","+r;
        if(dp.containsKey(key)) return dp.get(key);
        int val = Math.max(prices[r]-prices[l], Math.max(rec(prices, l+1, r), Math.max(rec(prices, l+1, r-1), rec(prices, l, r-1))));
        dp.put(key, val);
        return val;
    }
}
```
