public class codefile{
     // Greedy can work, no dp needed
    static int solve(int[] prices){
        int n = prices.length;
        int profit = 0;
        int maxFuture = 0;
        for (int i = n - 1; i >= 0; i--) {
            maxFuture = Math.max(maxFuture, prices[i]);
            profit += maxFuture - prices[i];
        }  
        return profit;         
    }
}
