```java
class Solution {
    public int maxProfit(int[] prices) {
        int minPrice = Integer.MAX_VALUE;
        int maxProfit = 0;
        for (int price : prices) {
            if (price < minPrice) minPrice = price;
            else maxProfit = Math.max(maxProfit, price - minPrice);
        }
        return maxProfit;
    }
}
```

### Memoization
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        // memo[day][canBuy]
        Integer[][] memo = new Integer[n][2];
        return dfs(0, 1, prices, memo);
    }

    private int dfs(int i, int canBuy, int[] prices, Integer[][] memo) {
        if (i == prices.length) return 0;
        if (memo[i][canBuy] != null) return memo[i][canBuy];

        // Choice: Skip today
        int idle = dfs(i + 1, canBuy, prices, memo);

        int action;
        if (canBuy == 1) {
            // Choice: Buy today (Status changes to 0)
            action = -prices[i] + dfs(i + 1, 0, prices, memo);
        } else {
            // Choice: Sell today (Transaction complete, so we add 0)
            action = prices[i]; 
        }

        return memo[i][canBuy] = Math.max(idle, action);
    }
}
```

### Tabulation
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        if (n == 0) return 0;

        // dp[day][canBuy]
        int[][] dp = new int[n + 1][2];

        // Base case: dp[n][0] and dp[n][1] are 0 (already 0 in Java)

        for (int i = n - 1; i >= 0; i--) {
            for (int canBuy = 0; canBuy <= 1; canBuy++) {
                if (canBuy == 1) {
                    // Max of (Buying today) or (Skipping)
                    dp[i][1] = Math.max(-prices[i] + dp[i + 1][0], dp[i + 1][1]);
                } else {
                    // Max of (Selling today) or (Waiting to sell later)
                    // Note: If we sell, we don't look at dp[i+1][1] because we only have 1 trade
                    dp[i][0] = Math.max(prices[i], dp[i + 1][0]);
                }
            }
        }

        return dp[0][1]; // Start at day 0 with permission to buy
    }
}
```
### Space Optimized
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        
        int nextBuy = 0;  // Represents dp[i+1][1]
        int nextSell = 0; // Represents dp[i+1][0]

        for (int i = n - 1; i >= 0; i--) {
            // curBuy = Max of (-price + nextSell, nextBuy)
            int curBuy = Math.max(-prices[i] + nextSell, nextBuy);
            
            // curSell = Max of (price, nextSell)
            int curSell = Math.max(prices[i], nextSell);

            nextBuy = curBuy;
            nextSell = curSell;
        }

        return nextBuy;
    }
}
```
