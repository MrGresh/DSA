### Memoization
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        Integer[][] memo = new Integer[n][2];
        return dfs(0, 1, prices, memo);
    }

    private int dfs(int day, int canBuy, int[] prices, Integer[][] memo) {
        if (day >= prices.length) return 0; // Notice >= because of day + 2

        if (memo[day][canBuy] != null) return memo[day][canBuy];

        // Option 1: Do nothing
        int idle = dfs(day + 1, canBuy, prices, memo);

        int action;
        if (canBuy == 1) {
            // Option 2: Buy today -> move to next day, cannot buy (0)
            action = -prices[day] + dfs(day + 1, 0, prices, memo);
        } else {
            // Option 2: Sell today -> move to day + 2 because of COOLDOWN
            action = prices[day] + dfs(day + 2, 1, prices, memo);
        }

        return memo[day][canBuy] = Math.max(idle, action);
    }
}
```
### Tabulation
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        if (n <= 1) return 0;
        
        int[][] dp = new int[n + 2][2];

        for (int i = n - 1; i >= 0; i--) {
            // Can Buy state
            dp[i][1] = Math.max(-prices[i] + dp[i + 1][0], dp[i + 1][1]);
            
            // Can Sell state (Cooldown jump to i + 2)
            dp[i][0] = Math.max(prices[i] + dp[i + 2][1], dp[i + 1][0]);
        }
        
        return dp[0][1];
    }
}
```

### Space Optimized
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        
        // Indexing: [0] is Sell state, [1] is Buy state
        int[] front2 = new int[2]; // Represents day i + 2
        int[] front1 = new int[2]; // Represents day i + 1
        int[] cur = new int[2];    // Represents day i

        for (int i = n - 1; i >= 0; i--) {
            // Buy today or skip
            cur[1] = Math.max(-prices[i] + front1[0], front1[1]);
            
            // Sell today (and use front2 for cooldown) or skip
            cur[0] = Math.max(prices[i] + front2[1], front1[0]);

            // Shift values for next iteration
            front2 = front1.clone();
            front1 = cur.clone();
        }

        return front1[1];
    }
}
```
