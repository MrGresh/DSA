### Memoization
```java
class Solution {
    Integer[][] memo;
    public int maxProfit(int[] prices, int fee) {
        memo = new Integer[prices.length+1][2];
        return rec(prices, 0, 1, fee);
    }
    int rec(int[] prices, int i, int canBuy, int fee) {
        if(i==prices.length) return 0;
        if(memo[i][canBuy]!=null) return memo[i][canBuy];
        int val = rec(prices, i+1, canBuy, fee); // do nothing
        if(canBuy==1) val = Math.max(val, -prices[i] + rec(prices, i+1, 1-canBuy, fee));
        else val = Math.max(val, prices[i] + rec(prices, i+1, 1-canBuy, fee) - fee);
        return memo[i][canBuy] = val;
    }
}
```

### Tabulation
```java
class Solution {
    public int maxProfit(int[] prices, int fee) {
        int[][] dp = new int[prices.length+1][2];
        for(int i=prices.length;i>=0;i--) {
            for(int canBuy : new int[] {1, 0}) {
                if(i==prices.length) dp[i][canBuy] = 0;
                else {
                    int val = dp[i+1][canBuy]; // do nothing
                    if(canBuy==1) val = Math.max(val, -prices[i] + dp[i+1][1-canBuy]);
                    else val = Math.max(val, prices[i] + dp[i+1][1-canBuy] - fee);
                    dp[i][canBuy] = val;
                }
            }
        }
        return dp[0][1];
    }
}
```

### Space Optimized
```java
class Solution {
    public int maxProfit(int[] prices, int fee) {
        int[] prev = new int[2];
        for(int i=prices.length;i>=0;i--) {
            int[] curr = new int[2];
            for(int canBuy : new int[] {1, 0}) {
                if(i==prices.length) curr[canBuy] = 0;
                else {
                    int val = prev[canBuy];
                    if(canBuy==1) val = Math.max(val, -prices[i] + prev[1-canBuy]);
                    else val = Math.max(val, prices[i] + prev[1-canBuy] - fee);
                    curr[canBuy] = val;
                }
            }
            prev = curr;
        }
        return prev[1];
    }
}
```
