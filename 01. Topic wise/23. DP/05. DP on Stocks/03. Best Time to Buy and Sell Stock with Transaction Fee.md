### Memoization
```java
class Solution {
    Integer[][] memo;
    public int maxProfit(int arr[], int k) {
        memo = new Integer[arr.length+1][2];
        return rec(arr, 0, 1, k);
    }
    int rec(int[] arr, int i, int canBuy, int k) {
        if(i==arr.length) return 0;
        if(memo[i][canBuy]!=null) return memo[i][canBuy];
        
        int val = rec(arr, i+1, canBuy, k); // do nothing
        if(canBuy==1) val = Math.max(val, -arr[i] + rec(arr, i+1, 1-canBuy, k));
        else val = Math.max(val, arr[i] + rec(arr, i+1, 1-canBuy, k) - k);
        
        return memo[i][canBuy] = val;
    }
}
```
### Tabulation
```java
class Solution {
    public int maxProfit(int arr[], int k) {
        int[][] dp = new int[arr.length+1][2];
        for(int i=arr.length;i>=0;i--) {
            for(int canBuy : new int[] {1, 0}) {
                if(i==arr.length) dp[i][canBuy] = 0;
                else {
                    int val = dp[i+1][canBuy]; // do nothing
                    if(canBuy==1) val = Math.max(val, -arr[i] + dp[i+1][1-canBuy]);
                    else val = Math.max(val, arr[i] + dp[i+1][1-canBuy] - k);
                    
                    dp[i][canBuy] = val;
                }
            }
        }
        return dp[0][1];
    }
}
```

### Space Optimized
```java
class Solution {
    public int maxProfit(int arr[], int k) {
        int[] prev = new int[2];
        for(int i=arr.length;i>=0;i--) {
            int[] curr = new int[2];
            for(int canBuy : new int[] {1, 0}) {
                if(i==arr.length) curr[canBuy] = 0;
                else {
                    int val = prev[canBuy]; // do nothing
                    if(canBuy==1) val = Math.max(val, -arr[i] + prev[1-canBuy]);
                    else val = Math.max(val, arr[i] + prev[1-canBuy] - k);
                    
                    curr[canBuy] = val;
                }
            }
            prev = curr;
        }
        return prev[1];
    }
}
```
