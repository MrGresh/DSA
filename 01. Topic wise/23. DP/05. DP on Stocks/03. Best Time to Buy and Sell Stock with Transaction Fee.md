### Memoization
```java
class Solution {
    public int maxProfit(int[] prices, int fee) {
        int n = prices.length;
        Integer[][] memo = new Integer[n][2];
        return dfs(0, 1, prices, fee, memo);
    }

    private int dfs(int day, int canBuy, int[] prices, int fee, Integer[][] memo) {
        if (day == prices.length) return 0;
        if (memo[day][canBuy] != null) return memo[day][canBuy];

        // Option 1: Do nothing
        int idle = dfs(day + 1, canBuy, prices, fee, memo);

        int action;
        if (canBuy == 1) {
            // Option 2: Buy today
            action = -prices[day] + dfs(day + 1, 0, prices, fee, memo);
        } else {
            // Option 2: Sell today (Subtract the fee here)
            action = prices[day] - fee + dfs(day + 1, 1, prices, fee, memo);
        }

        return memo[day][canBuy] = Math.max(idle, action);
    }
}
```

### Tabulation
```java
class Solution {
    public int maxProfit(int[] prices, int fee) {
        int n = prices.length;
        int[][] dp = new int[n + 1][2];

        for (int i = n - 1; i >= 0; i--) {
            // State: Can Buy
            dp[i][1] = Math.max(-prices[i] + dp[i + 1][0], dp[i + 1][1]);
            
            // State: Must Sell (Subtract fee on sell)
            dp[i][0] = Math.max(prices[i] - fee + dp[i + 1][1], dp[i + 1][0]);
        }

        return dp[0][1];
    }
}
```

### Space Optimized
```java
class Solution {
    public int maxProfit(int[] prices, int fee) {
        int n = prices.length;
        
        int aheadBuy = 0;  // dp[i+1][1]
        int aheadSell = 0; // dp[i+1][0]

        for (int i = n - 1; i >= 0; i--) {
            // Current day "Can Buy"
            int curBuy = Math.max(-prices[i] + aheadSell, aheadBuy);
            
            // Current day "Must Sell"
            int curSell = Math.max(prices[i] - fee + aheadBuy, aheadSell);

            aheadBuy = curBuy;
            aheadSell = curSell;
        }

        return aheadBuy;
    }
}
```
