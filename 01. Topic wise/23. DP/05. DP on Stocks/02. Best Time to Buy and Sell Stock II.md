### Optimized
```java
class Solution {
    public int maxProfit(int[] prices) {
        int ans = 0;
        for(int i=1;i<prices.length;i++) {
        if(prices[i]>prices[i-1]) ans += prices[i]-prices[i-1];
        }
        return ans;
    }
}
```
### Brute Force
```java
class Solution {
    Map<String, Integer> dp;
    public int maxProfit(int[] prices) {
        dp = new HashMap<>();
        return rec(prices, 0, true);
    }
    int rec(int[] prices, int i, boolean canBuy) {
        if(i==prices.length) return 0;
        String key = i+"."+canBuy;
        if(dp.containsKey(key)) return dp.get(key);
        int val = rec(prices, i+1, canBuy); // do nothing
        if(canBuy) val = Math.max(val, -prices[i] + rec(prices, i+1, false)); // buy
        else val = Math.max(val, prices[i] + rec(prices, i+1, true)); // sell
        dp.put(key, val);
        return val;
    }
}
```
