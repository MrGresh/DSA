```java
class Solution {
    public int maxProfit(int[] prices) {
        int ans = 0;
        for(int i=1;i<prices.length;i++) {
        if(prices[i]>prices[i-1]) ans += prices[i]-prices[i-1];
        }
        return ans;
    }
}
```

### Memoization
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        Integer[][] memo = new Integer[n][2];  // memo[i][buy] where buy = 0 or 1
        return dfs(0, 1, prices, memo);       // Start on day 0, allowed to buy
    }

    private int dfs(int day, int canBuy, int[] prices, Integer[][] memo) {
        if (day == prices.length) return 0;

        if (memo[day][canBuy] != null) return memo[day][canBuy];

        // Option 1: Do nothing today
        int idle = dfs(day + 1, canBuy, prices, memo);

        int action;
        if (canBuy == 1) {
            // Option 2: Buy today
            action = -prices[day] + dfs(day + 1, 0, prices, memo);
        } else {
            // Option 2: Sell today
            action = prices[day] + dfs(day + 1, 1, prices, memo);
        }

        memo[day][canBuy] = Math.max(idle, action);
        return memo[day][canBuy];
    }
}
```

### Tabulation
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        int[][] dp = new int[n + 1][2];
        
        // Base case: dp[n][0] and dp[n][1] are 0 (already handled by Java)

        for (int i = n - 1; i >= 0; i--) {
            // Option to Buy
            dp[i][1] = Math.max(-prices[i] + dp[i + 1][0], dp[i + 1][1]);
            // Option to Sell
            dp[i][0] = Math.max(prices[i] + dp[i + 1][1], dp[i + 1][0]);
        }
        
        return dp[0][1];
    }
}
```

### Space Optimized
```java
class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        
        // These represent dp[i+1][1] (buy) and dp[i+1][0] (sell)
        int aheadBuy = 0;
        int aheadSell = 0;

        // Iterate backwards from the last day
        for (int i = n - 1; i >= 0; i--) {
            // Current day's "Can Buy" state
            int curBuy = Math.max(-prices[i] + aheadSell, aheadBuy);
            
            // Current day's "Must Sell" state
            int curSell = Math.max(prices[i] + aheadBuy, aheadSell);

            // Update ahead values for the next iteration (day i-1)
            aheadBuy = curBuy;
            aheadSell = curSell;
        }

        return aheadBuy;
    }
}
```
