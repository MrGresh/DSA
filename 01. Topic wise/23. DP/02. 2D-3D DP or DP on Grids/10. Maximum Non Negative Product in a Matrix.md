### Memoization
```java
class Solution {
    Long[][][] memo;
    public int maxProductPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        memo = new Long[m][n][2];
        // We call the recursion which returns [minProduct, maxProduct]
        long[] result = rec(grid, 0, 0);
        
        long maxVal = result[1];
        
        // If the best we could do is negative, return -1 per instructions
        if (maxVal < 0) return -1;
        
        // Apply modulo only at the very end
        return (int) (maxVal % 1000000007);
    }

    private long[] rec(int[][] grid, int i, int j) {
        int m = grid.length, n = grid[0].length;

        // Base Case: Reached the target cell
        if (i == m - 1 && j == n - 1) {
            return new long[] { grid[i][j], grid[i][j] };
        }
        if(memo[i][j][0]!=null) return new long[] { memo[i][j][0], memo[i][j][1] };
        long currentMin = Long.MAX_VALUE;
        long currentMax = Long.MIN_VALUE;

        // Helper to check Down and Right
        int[][] dirs = {{1, 0}, {0, 1}};

        for (int[] d : dirs) {
            int nextI = i + d[0];
            int nextJ = j + d[1];

            if (nextI < m && nextJ < n) {
                long[] nextRes = rec(grid, nextI, nextJ);
                
                // Multiply current cell with the min and max of the NEXT path
                long opt1 = grid[i][j] * nextRes[0];
                long opt2 = grid[i][j] * nextRes[1];

                // Update the current cell's min and max
                currentMin = Math.min(currentMin, Math.min(opt1, opt2));
                currentMax = Math.max(currentMax, Math.max(opt1, opt2));
            }
        }
        memo[i][j][0] = currentMin;
        memo[i][j][1] = currentMax;
        return new long[] { currentMin, currentMax };
    }
}
```

### Tabulation

```java
class Solution {
    public int maxProductPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        long[][][] dp = new long[m][n][2];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                if (i == m - 1 && j == n - 1) {
                    dp[m-1][n-1][0] = (long)grid[m-1][n-1]; // base case
                    dp[m-1][n-1][1] = (long)grid[m-1][n-1]; // base case
                    continue;
                };
                int[][] dirs = {{1, 0}, {0, 1}};
                long currentMin = Long.MAX_VALUE;
                long currentMax = Long.MIN_VALUE;
                for (int[] d : dirs) {
                    int nextI = i + d[0];
                    int nextJ = j + d[1];

                    if (nextI < m && nextJ < n) {
                        long[] nextRes = dp[nextI][nextJ];
                        
                        // Multiply current cell with the min and max of the NEXT path
                        long opt1 = grid[i][j] * nextRes[0];
                        long opt2 = grid[i][j] * nextRes[1];

                        // Update the current cell's min and max
                        currentMin = Math.min(currentMin, Math.min(opt1, opt2));
                        currentMax = Math.max(currentMax, Math.max(opt1, opt2));
                    }
                }
                dp[i][j][0] = currentMin;
                dp[i][j][1] = currentMax;
            }
        }
        long maxVal = dp[0][0][1];
        if (maxVal < 0) return -1;
        return (int) (maxVal % 1000000007);
    }
}
```

```java
class Solution {
    public int maxProductPath(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        long[][] prev = new long[n][2];
        for(int i=m-1;i>=0;i--) {
            long[][] curr = new long[n][2];
            for(int j=n-1;j>=0;j--) {
                if (i == m - 1 && j == n - 1) {
                    curr[n-1][0] = (long)grid[m-1][n-1];
                    curr[n-1][1] = (long)grid[m-1][n-1];
                    continue;
                }
                int[][] dirs = {{1, 0}, {0, 1}};
                long currentMin = Long.MAX_VALUE;
                long currentMax = Long.MIN_VALUE;
                for (int[] d : dirs) {
                    int nextI = i + d[0];
                    int nextJ = j + d[1];

                    if (nextI < m && nextJ < n) {
                        long[] nextRes = nextI>i ? prev[j] : curr[nextJ];
                        
                        // Multiply current cell with the min and max of the NEXT path
                        long opt1 = grid[i][j] * nextRes[0];
                        long opt2 = grid[i][j] * nextRes[1];

                        // Update the current cell's min and max
                        currentMin = Math.min(currentMin, Math.min(opt1, opt2));
                        currentMax = Math.max(currentMax, Math.max(opt1, opt2));
                    }
                }
                curr[j][0] = currentMin;
                curr[j][1] = currentMax;
            }
            prev = curr;
        }
        long maxVal = prev[0][1];
        if (maxVal < 0) return -1;
        return (int) (maxVal % 1000000007);
    }
}
```
