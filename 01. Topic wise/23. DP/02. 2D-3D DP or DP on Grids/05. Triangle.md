### Memoization
```java
class Solution {
    Integer[][] memo;
    public int minimumTotal(List<List<Integer>> triangle) {
        int m = triangle.size();
        memo = new Integer[m][m];
        return rec(triangle, 0, 0);
    }
    int rec(List<List<Integer>> triangle, int i, int j) {
        int m = triangle.size();
        if(i==m-1) return triangle.get(i).get(j);
        if(memo[i][j]!=null) return memo[i][j];
        return memo[i][j] = triangle.get(i).get(j) + Math.min(rec(triangle, i+1, j), rec(triangle, i+1, j+1));
    }
}
```
### Tabulation
```java
class Solution {
    public int minimumTotal(List<List<Integer>> triangle) {
        int m = triangle.size();
        int[][] dp = new int[m][m];
        for(int i=m-1;i>=0;i--) {
            for(int j=triangle.get(i).size()-1;j>=0;j--) {
                if(i==m-1) dp[i][j] = triangle.get(i).get(j);
                else {
                    dp[i][j] = triangle.get(i).get(j) + Math.min(dp[i+1][j], dp[i+1][j+1]);
                }
            }
        }
        return dp[0][0];
    }
}
```
### Space Optimization
```java
class Solution {
    public int minimumTotal(List<List<Integer>> triangle) {
        int m = triangle.size();
        int[] prev = new int[triangle.get(m-1).size()];
        for(int i=m-1;i>=0;i--) {
            int[] curr = new int[triangle.get(i).size()];
            for(int j=triangle.get(i).size()-1;j>=0;j--) {
                if(i==m-1) curr[j] = triangle.get(i).get(j);
                else {
                    curr[j] = triangle.get(i).get(j) + Math.min(prev[j], prev[j+1]);
                }
            }
            prev = curr;
        }
        return prev[0];
    }
}
```
