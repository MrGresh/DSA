### Memoization
```java
class Solution {
    Integer[][] memo;
    public int minFallingPathSum(int[][] mat) {
        int val = Integer.MAX_VALUE;
        memo = new Integer[mat.length][mat[0].length];
        for(int j=0;j<mat[0].length;j++) {
            val = Math.min(val, rec(mat, 0, j));
        }
        return val;
    }
    int rec(int[][] mat, int i, int j) {
        if(i==mat.length-1) return mat[i][j];
        if(memo[i][j]!=null) return memo[i][j];
        int val = rec(mat, i+1, j);
        if(j-1>=0) val = Math.min(val, rec(mat, i+1, j-1));
        if(j<mat[0].length-1) val = Math.min(val, rec(mat, i+1, j+1));
        return memo[i][j] = val + mat[i][j];
    }
}
```

### Tabulation
```java
class Solution {
    public int minFallingPathSum(int[][] mat) {
        int m = mat.length, n = mat[0].length;
        int[][] dp = new int[m][n];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                if(i==m-1) {
                    dp[i][j] = mat[i][j];
                } else {
                    int val = dp[i+1][j];
                    if(j-1>=0) val = Math.min(val, dp[i+1][j-1]);
                    if(j+1<n) val = Math.min(val, dp[i+1][j+1]);
                    dp[i][j] = val + mat[i][j];
                }
            }
        }
        int ans = Integer.MAX_VALUE;
        for(int j=0;j<n;j++) {
            ans = Math.min(ans, dp[0][j]);
        }
        return ans;
    }
}
```

### Space Optimization
```java
class Solution {
    public int minFallingPathSum(int[][] mat) {
        int m = mat.length, n = mat[0].length;
        int[] prev = new int[n];
        for(int i=m-1;i>=0;i--) {
            int[] curr = new int[n];
            for(int j=n-1;j>=0;j--) {
                if(i==m-1) {
                    curr[j] = mat[i][j];
                } else {
                    int val = prev[j];
                    if(j-1>=0) val = Math.min(val, prev[j-1]);
                    if(j+1<n) val = Math.min(val, prev[j+1]);
                    curr[j] = val + mat[i][j];
                }
            }
            prev = curr;
        }
        int ans = Integer.MAX_VALUE;
        for(int j=0;j<n;j++) {
            ans = Math.min(ans, prev[j]);
        }
        return ans;
    }
}
```
