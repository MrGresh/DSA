### Memoization
```java
class Solution {
    Integer[][] memo;
    public int minPathSum(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        memo = new Integer[m][n];
        return rec(grid, 0, 0);
    }
    int rec(int[][] grid, int i, int j) {
        int m = grid.length, n = grid[0].length;
        if(i==m-1 && j==n-1) return grid[i][j];
        if(memo[i][j]!=null) return memo[i][j];
        int val = Integer.MAX_VALUE;
        if(i+1<m) val = rec(grid, i+1, j);
        if(j+1<n) val = Math.min(val, rec(grid, i, j+1));
        return memo[i][j] = grid[i][j]+val;
    }
}
```
### Tabulation
```java
class Solution {
    public int minPathSum(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] dp = new int[m][n];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                if(i==m-1 && j==n-1) dp[i][j] = grid[i][j];
                else {
                    int val = Integer.MAX_VALUE;
                    if(i+1<m) val = dp[i+1][j];
                    if(j+1<n) val = Math.min(val, dp[i][j+1]);
                    dp[i][j] = grid[i][j]+val;
                }
            }
        }
        return dp[0][0];
    }
}
```

### Space Optimization
```java
class Solution {
    public int minPathSum(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[] prev = new int[n];
        for(int i=m-1;i>=0;i--) {
            int[] curr = new int[n];
            for(int j=n-1;j>=0;j--) {
                if(i==m-1 && j==n-1) curr[j] = grid[i][j];
                else {
                    int val = Integer.MAX_VALUE;
                    if(i+1<m) val = prev[j];
                    if(j+1<n) val = Math.min(val, curr[j+1]);
                    curr[j] = grid[i][j]+val;
                }
            }
            prev = curr;
        }
        return prev[0];
    }
}
```
