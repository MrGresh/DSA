### Memoization
```java
class Solution {
    Integer[][][] memo;
    public int minimumDistance(String s) {
        int n = s.length();
        memo = new Integer[n+1][26][26];
        int ans = Integer.MAX_VALUE;
        for(int f1=0;f1<26;f1++) {
            for(int f2=0;f2<26;f2++) {
                if(f1!=f2) ans = Math.min(ans, rec(s, 0, f1, f2));
            }
        }
        return ans;
    }
    int rec(String s, int i, int f1, int f2) {
        if(i==s.length()) return 0;
        if(memo[i][f1][f2]!=null) return memo[i][f1][f2];
        int val1 = getDis(s.charAt(i), (char) (65+f1)) + rec(s, i+1, s.charAt(i)-'A', f2);
        int val2 = getDis(s.charAt(i), (char) (65+f2)) + rec(s, i+1, f1, s.charAt(i)-'A');
        return memo[i][f1][f2] = Math.min(val1, val2);
    }
    int getDis(char c1, char c2) {
        int[] p1 = getPos(c1);
        int[] p2 = getPos(c2);
        return Math.abs(p1[0]-p2[0]) + Math.abs(p1[1]-p2[1]);
    }
    int[] getPos(char c) {
        int index = c - 'A';
        int row = index / 6;
        int col = index % 6;
        return new int[] {row, col};
    }
}
```

### Tabulation
```java
class Solution {
    public int minimumDistance(String s) {
        int n = s.length();
        int[][][] dp = new int[n+1][26][26];
        for(int i=s.length();i>=0;i--) {
            for(int f1=0;f1<26;f1++) {
                for(int f2=0;f2<26;f2++) {
                    if(i==s.length()) dp[i][f1][f2] = 0;
                    else {
                        int val1 = getDis(s.charAt(i), (char) (65+f1)) + dp[i+1][s.charAt(i)-'A'][f2];
                        int val2 = getDis(s.charAt(i), (char) (65+f2)) + dp[i+1][f1][s.charAt(i)-'A'];
                        dp[i][f1][f2] = Math.min(val1, val2);
                    }
                }
            }
        }
        int ans = Integer.MAX_VALUE;
        for(int f1=0;f1<26;f1++) {
            for(int f2=0;f2<26;f2++) {
                if(f1!=f2) ans = Math.min(ans, dp[0][f1][f2]);
            }
        }
        return ans;
    }
    int getDis(char c1, char c2) {
        int[] p1 = getPos(c1);
        int[] p2 = getPos(c2);
        return Math.abs(p1[0]-p2[0]) + Math.abs(p1[1]-p2[1]);
    }
    int[] getPos(char c) {
        int index = c - 'A';
        int row = index / 6;
        int col = index % 6;
        return new int[] {row, col};
    }
}
```

```java
class Solution {
    public int minimumDistance(String s) {
        int n = s.length();
        int[][] prev = new int[26][26];
        for(int i=s.length();i>=0;i--) {
            int[][] curr = new int[26][26];
            for(int f1=0;f1<26;f1++) {
                for(int f2=0;f2<26;f2++) {
                    if(i==s.length()) curr[f1][f2] = 0;
                    else {
                        int val1 = getDis(s.charAt(i), (char) (65+f1)) + prev[s.charAt(i)-'A'][f2];
                        int val2 = getDis(s.charAt(i), (char) (65+f2)) + prev[f1][s.charAt(i)-'A'];
                        curr[f1][f2] = Math.min(val1, val2);
                    }
                }
            }
            prev = curr;
        }
        int ans = Integer.MAX_VALUE;
        for(int f1=0;f1<26;f1++) {
            for(int f2=0;f2<26;f2++) {
                if(f1!=f2) ans = Math.min(ans, prev[f1][f2]);
            }
        }
        return ans;
    }
    int getDis(char c1, char c2) {
        int[] p1 = getPos(c1);
        int[] p2 = getPos(c2);
        return Math.abs(p1[0]-p2[0]) + Math.abs(p1[1]-p2[1]);
    }
    int[] getPos(char c) {
        int index = c - 'A';
        int row = index / 6;
        int col = index % 6;
        return new int[] {row, col};
    }
}
```
