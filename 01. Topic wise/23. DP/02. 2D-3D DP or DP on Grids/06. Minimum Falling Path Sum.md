### Memoization
```java
class Solution {
    public int minFallingPathSum(int[][] matrix) {
        int n = matrix.length;
        int[][] memo = new int[n][n];
        for (int[] row : memo) java.util.Arrays.fill(row, Integer.MAX_VALUE);
        
        int minSum = Integer.MAX_VALUE;
        // Try starting the path from every element in the last row
        for (int j = 0; j < n; j++) {
            minSum = Math.min(minSum, solve(n - 1, j, matrix, memo));
        }
        return minSum;
    }

    private int solve(int i, int j, int[][] matrix, int[][] memo) {
        // Base Case: Out of bounds
        if (j < 0 || j >= matrix.length) return (int) 1e9;
        // Base Case: First row reached
        if (i == 0) return matrix[0][j];
        
        if (memo[i][j] != Integer.MAX_VALUE) return memo[i][j];

        int up = solve(i - 1, j, matrix, memo);
        int leftDiag = solve(i - 1, j - 1, matrix, memo);
        int rightDiag = solve(i - 1, j + 1, matrix, memo);

        return memo[i][j] = matrix[i][j] + Math.min(up, Math.min(leftDiag, rightDiag));
    }
}
```

### Tabulation
```java
class Solution {
    public int minFallingPathSum(int[][] matrix) {
        int n = matrix.length;
        int[][] dp = new int[n][n];

        // Base case: Initialize first row
        for (int j = 0; j < n; j++) dp[0][j] = matrix[0][j];

        for (int i = 1; i < n; i++) {
            for (int j = 0; j < n; j++) {
                int up = dp[i - 1][j];
                int leftDiag = (j - 1 >= 0) ? dp[i - 1][j - 1] : (int) 1e9;
                int rightDiag = (j + 1 < n) ? dp[i - 1][j + 1] : (int) 1e9;

                dp[i][j] = matrix[i][j] + Math.min(up, Math.min(leftDiag, rightDiag));
            }
        }

        int minSum = Integer.MAX_VALUE;
        for (int j = 0; j < n; j++) {
            minSum = Math.min(minSum, dp[n - 1][j]);
        }
        return minSum;
    }
}
```

### Space Optimization
```java
class Solution {
    public int minFallingPathSum(int[][] matrix) {
        int n = matrix.length;
        int[] prev = new int[n];

        // Initialize prev with the first row
        for (int j = 0; j < n; j++) prev[j] = matrix[0][j];

        for (int i = 1; i < n; i++) {
            int[] curr = new int[n];
            for (int j = 0; j < n; j++) {
                int up = prev[j];
                int leftDiag = (j - 1 >= 0) ? prev[j - 1] : (int) 1e9;
                int rightDiag = (j + 1 < n) ? prev[j + 1] : (int) 1e9;

                curr[j] = matrix[i][j] + Math.min(up, Math.min(leftDiag, rightDiag));
            }
            prev = curr;
        }

        int minSum = Integer.MAX_VALUE;
        for (int j = 0; j < n; j++) {
            minSum = Math.min(minSum, prev[j]);
        }
        return minSum;
    }
}
```
