## Brute Force Way
### Memoization
```java
class Solution {
    Integer[][][] memo;
    int countWays(int n, int k) {
        memo = new Integer[n+1][k+1][k+1];
        return rec(k, n, 0, 0);
    }
    int rec(int k, int n, int prev1, int prev2) {
        if(n==0) return 1;
        if(memo[n][prev1][prev2]!=null) return memo[n][prev1][prev2];
        int cnt = 0;
        for(int i=1;i<=k;i++) {
            if(!((prev1==prev2) && (i==prev1))) cnt += rec(k, n-1, i, prev1);
        }
        return memo[n][prev1][prev2] = cnt;
    }
}
```
### Tabulation
```java
class Solution {
    int countWays(int N, int k) {
        int[][][] dp = new int[N+1][k+1][k+1];
        for(int n=0;n<=N;n++) {
            for(int prev1=0;prev1<=k;prev1++) {
                for(int prev2=0;prev2<=k;prev2++) {
                    if(n==0) dp[n][prev1][prev2] = 1;
                    else {
                        int cnt = 0;
                        for(int i=1;i<=k;i++) {
                            if(!((prev1==prev2) && (i==prev1))) cnt += dp[n-1][i][prev1];
                        }
                        dp[n][prev1][prev2] = cnt;
                    }
                }
            }
        }
        return dp[N][0][0];
    }
}
```
### Space Optimization
```java
class Solution {
    int countWays(int N, int k) {
        int[][] prev = new int[k+1][k+1];
        for(int n=0;n<=N;n++) {
            int[][] curr = new int[k+1][k+1];
            for(int prev1=0;prev1<=k;prev1++) {
                for(int prev2=0;prev2<=k;prev2++) {
                    if(n==0) curr[prev1][prev2] = 1;
                    else {
                        int cnt = 0;
                        for(int i=1;i<=k;i++) {
                            if(!((prev1==prev2) && (i==prev1))) cnt += prev[i][prev1];
                        }
                        curr[prev1][prev2] = cnt;
                    }
                }
            }
            prev = curr;
        }
        return prev[0][0];
    }
}
```

## Optimal Way
### Memoization
```java
class Solution {
    Long[][] memo;
    public int countWays(int n, int k) {
        if (n == 0) return 0;
        if (n == 1) return k;
        memo = new Long[n + 1][2];
        return (int) (rec(n, k, 0) + rec(n, k, 1));
    }

    long rec(int i, int k, int type) {
        if (i == 2) {
            return (type == 0) ? (long) k * (k - 1) : (long) k;
        }
        if (memo[i][type] != null) return memo[i][type];
        long ans = 0;
        if (type == 0) ans = (rec(i - 1, k, 0) + rec(i - 1, k, 1)) * (k - 1);
        else ans = rec(i - 1, k, 0);
        return memo[i][type] = ans;
    }
}
```
### Tabulation
```java
class Solution {
    int countWays(int n, int k) {
        if (n == 0) return 0;
        if (n == 1) return k;
        long[][] dp = new long[n + 1][2];
        for(int i=2;i<=n;i++) {
            for(int type=0;type<=1;type++) {
                if (i == 2) {
                    dp[i][type] = (type == 0) ? (long) k * (k - 1) : (long) k;
                } else {
                    long ans = 0;
                    if (type == 0) ans = (dp[i - 1][0] + dp[i - 1][1]) * (k - 1);
                    else ans = dp[i - 1][0];
                    dp[i][type] = ans;
                }
            }
        }
        return (int) (dp[n][0] + dp[n][1]);
    }
}
```

### Space Optimization
```java
class Solution {
    int countWays(int n, int k) {
        if (n == 0) return 0;
        if (n == 1) return k;
        long[] prev = new long[2];
        for(int i=2;i<=n;i++) {
            long[] curr = new long[2];
            for(int type=0;type<=1;type++) {
                if (i == 2) {
                    curr[type] = (type == 0) ? (long) k * (k - 1) : (long) k;
                } else {
                    long ans = 0;
                    if (type == 0) ans = (prev[0] + prev[1]) * (k - 1);
                    else ans = prev[0];
                    curr[type] = ans;
                }
            }
            prev = curr;
        }
        return (int) (prev[0] + prev[1]);
    }
}
```
