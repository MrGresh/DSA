### Memoization
```java
class Solution {
    Integer[][][] memo;
    public int cherryPickup(int[][] grid) {
        memo = new Integer[grid.length][grid[0].length][grid[0].length];
        return rec(grid, 0, 0, grid[0].length-1);
    }
    int rec(int[][] grid, int i, int j1, int j2) {
        if(i==grid.length-1) {
            if(j1==j2) return grid[i][j1];
            else return grid[i][j1] + grid[i][j2];
        }
        if(memo[i][j1][j2]!=null) return memo[i][j1][j2];
        int cols = grid[0].length;
        int val = Integer.MIN_VALUE;
        int[][] dirs = {{-1, -1}, {-1, 0}, {-1, 1}, {0, -1}, {0, 0}, {0, 1}, {1, -1}, {1, 0}, {1, 1}};
        for(int[] dir: dirs) {
            int newJ1 = j1+dir[0], newJ2 = j2+dir[1];
            if(newJ1>=0 && newJ1<cols && newJ2>=0 && newJ2<cols) {
                val = Math.max(val, rec(grid, i+1, newJ1, newJ2));
            }
        }
        if(j1==j2) val += grid[i][j1];
        else val += grid[i][j1] + grid[i][j2];
        return memo[i][j1][j2] = val;
    }
}
```

### Tabulation
```java
class Solution {
    public int cherryPickup(int[][] grid) {
        int[][][] dp = new int[grid.length][grid[0].length][grid[0].length];
        int cols = grid[0].length;
        for(int i=grid.length-1;i>=0;i--) {
            for(int j1=0;j1<cols;j1++) {
                for(int j2=cols-1;j2>=0;j2--) {
                    if(i==grid.length-1) {
                        if(j1==j2) dp[i][j1][j2] = grid[i][j1];
                        else dp[i][j1][j2] = grid[i][j1] + grid[i][j2];
                    } else {
                        int val = Integer.MIN_VALUE;
                        int[][] dirs = {{-1, -1}, {-1, 0}, {-1, 1}, {0, -1}, {0, 0}, {0, 1}, {1, -1}, {1, 0}, {1, 1}};
                        for(int[] dir: dirs) {
                            int newJ1 = j1+dir[0], newJ2 = j2+dir[1];
                            if(newJ1>=0 && newJ1<cols && newJ2>=0 && newJ2<cols) {
                                val = Math.max(val, dp[i+1][newJ1][newJ2]);
                            }
                        }
                        if(j1==j2) val += grid[i][j1];
                        else val += grid[i][j1] + grid[i][j2];
                        dp[i][j1][j2] = val;
                    }
                }
            }
        }
        return dp[0][0][cols-1];
    }
}
```

```java
class Solution {
    public int cherryPickup(int[][] grid) {
        int[][] prev = new int[grid[0].length][grid[0].length];
        int cols = grid[0].length;
        for(int i=grid.length-1;i>=0;i--) {
            int[][] curr = new int[grid[0].length][grid[0].length];
            for(int j1=0;j1<cols;j1++) {
                for(int j2=cols-1;j2>=0;j2--) {
                    if(i==grid.length-1) {
                        if(j1==j2) curr[j1][j2] = grid[i][j1];
                        else curr[j1][j2] = grid[i][j1] + grid[i][j2];
                    } else {
                        int val = Integer.MIN_VALUE;
                        int[][] dirs = {{-1, -1}, {-1, 0}, {-1, 1}, {0, -1}, {0, 0}, {0, 1}, {1, -1}, {1, 0}, {1, 1}};
                        for(int[] dir: dirs) {
                            int newJ1 = j1+dir[0], newJ2 = j2+dir[1];
                            if(newJ1>=0 && newJ1<cols && newJ2>=0 && newJ2<cols) {
                                val = Math.max(val, prev[newJ1][newJ2]);
                            }
                        }
                        if(j1==j2) val += grid[i][j1];
                        else val += grid[i][j1] + grid[i][j2];
                        curr[j1][j2] = val;
                    }
                }
            }
            prev = curr;
        }
        return prev[0][cols-1];
    }
}
```
