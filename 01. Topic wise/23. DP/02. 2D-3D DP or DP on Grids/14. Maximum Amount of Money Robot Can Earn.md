### Memoization
```java
class Solution {
    Integer[][][] memo;
    public int maximumAmount(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        memo = new Integer[m][n][3];
        return rec(grid, 0, 0, 2);
    }
    int rec(int[][] grid, int i, int j, int power) {
        int m = grid.length, n = grid[0].length;
        if(i==m-1 && j==n-1) return power>0 ? Math.max(0, grid[i][j]) : grid[i][j];
        if(memo[i][j][power]!=null) return memo[i][j][power];
        if(power==0) {
            int ans = Integer.MIN_VALUE;
            if(i+1<m) ans = rec(grid, i+1, j, 0);
            if(j+1<n) ans = Math.max(ans, rec(grid, i, j+1, 0));
            return memo[i][j][power] = ans + grid[i][j];
        } else {
            int ans1 = Integer.MIN_VALUE;
            if(grid[i][j]<0) {
                if(i+1<m) ans1 = rec(grid, i+1, j, power-1);
                if(j+1<n) ans1 = Math.max(ans1, rec(grid, i, j+1, power-1));
            }

            int ans2 = Integer.MIN_VALUE;
            if(i+1<m) ans2 = rec(grid, i+1, j, power);
            if(j+1<n) ans2 = Math.max(ans2, rec(grid, i, j+1, power));
            ans2 += grid[i][j];

            return memo[i][j][power] = Math.max(ans1, ans2);
        }
    }
}
```

### Tabulation
```java
class Solution {
    public int maximumAmount(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][][] dp = new int[m][n][3];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                for(int power=2;power>=0;power--) {
                    if(i==m-1 && j==n-1) dp[i][j][power] = power>0 ? Math.max(0, grid[i][j]) : grid[i][j];
                    else {
                        if(power==0) {
                            int ans = Integer.MIN_VALUE;
                            if(i+1<m) ans = dp[i+1][j][0];
                            if(j+1<n) ans = Math.max(ans, dp[i][j+1][0]);
                            dp[i][j][power] = ans + grid[i][j];
                        } else {
                            int ans1 = Integer.MIN_VALUE;
                            if(grid[i][j]<0) {
                                if(i+1<m) ans1 = dp[i+1][j][power-1];
                                if(j+1<n) ans1 = Math.max(ans1, dp[i][j+1][power-1]);
                            }

                            int ans2 = Integer.MIN_VALUE;
                            if(i+1<m) ans2 = dp[i+1][j][power];
                            if(j+1<n) ans2 = Math.max(ans2, dp[i][j+1][power]);
                            ans2 += grid[i][j];

                            dp[i][j][power] = Math.max(ans1, ans2);
                        }
                    }
                }
            }
        }
        return dp[0][0][2];
    }
}
```

### Space Optimization
```java
class Solution {
    public int maximumAmount(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        int[][] prev = new int[n][3];
        for(int i=m-1;i>=0;i--) {
            int[][] curr = new int[n][3];
            for(int j=n-1;j>=0;j--) {
                for(int power=2;power>=0;power--) {
                    if(i==m-1 && j==n-1) curr[j][power] = power>0 ? Math.max(0, grid[i][j]) : grid[i][j];
                    else {
                        if(power==0) {
                            int ans = Integer.MIN_VALUE;
                            if(i+1<m) ans = prev[j][0];
                            if(j+1<n) ans = Math.max(ans, curr[j+1][0]);
                            curr[j][power] = ans + grid[i][j];
                        } else {
                            int ans1 = Integer.MIN_VALUE;
                            if(grid[i][j]<0) {
                                if(i+1<m) ans1 = prev[j][power-1];
                                if(j+1<n) ans1 = Math.max(ans1, curr[j+1][power-1]);
                            }

                            int ans2 = Integer.MIN_VALUE;
                            if(i+1<m) ans2 = prev[j][power];
                            if(j+1<n) ans2 = Math.max(ans2, curr[j+1][power]);
                            ans2 += grid[i][j];

                            curr[j][power] = Math.max(ans1, ans2);
                        }
                    }
                }
            }
            prev = curr;
        }
        return prev[0][2];
    }
}
```
