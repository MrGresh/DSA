### Memoization
```java
class Solution {
    Long[][] memo;
    public long minimumTotalDistance(List<Integer> robots, int[][] factories) {
        Collections.sort(robots);
        Arrays.sort(factories, (a, b) -> Integer.compare(a[0], b[0]));
        int n = robots.size();
        int m = factories.length;
        memo = new Long[m][n];
        return rec(robots, factories, 0, 0);
    }
    long rec(List<Integer> robot, int[][] factory, int f_idx, int r_idx) {
        if (r_idx == robot.size()) return 0; // Base Case: All robots are repaired
        if (f_idx == factory.length) return (long) 1e15; // Base Case: No more factories, but robots remain

        if(memo[f_idx][r_idx]!=null) return memo[f_idx][r_idx];
        
        // Option 1: Skip this factory entirely
        long res = rec(robot, factory, f_idx + 1, r_idx);

        // Option 2: This factory repairs 'k' robots (from 1 to limit)
        long currentFactoryDistance = 0;
        for (int k = 1; k <= factory[f_idx][1]; k++) {
            int robotToFix = r_idx + k - 1;
            
            // If we run out of robots, stop
            if (robotToFix >= robot.size()) break;

            // Add distance of the current robot to this factory
            currentFactoryDistance += Math.abs((long) robot.get(robotToFix) - factory[f_idx][0]);
            
            // Total cost = current group distance + subproblem for remaining robots/factories
            long fullCost = currentFactoryDistance + rec(robot, factory, f_idx + 1, r_idx + k);
            res = Math.min(res, fullCost);
        }

        return memo[f_idx][r_idx] = res;
    }
}
```
