### Memoization
```java
class Solution {
    Integer[][] memo;
    public int minFallingPathSum(int[][] mat) {
        int val = Integer.MAX_VALUE;
        memo = new Integer[mat.length][mat[0].length];
        for(int j=0;j<mat[0].length;j++) {
            val = Math.min(val, rec(mat, 0, j));
        }
        return val;
    }
    int rec(int[][] mat, int i, int j) {
        if(i==mat.length-1) return mat[i][j];
        if(memo[i][j]!=null) return memo[i][j];
        int val = Integer.MAX_VALUE;
        for(int col=0;col<mat[0].length;col++) {
            if(col!=j) val = Math.min(val, rec(mat, i+1, col));
        }
        return memo[i][j] = val + mat[i][j];
    }
}
```

### Tabulation
```java
class Solution {
    public int minFallingPathSum(int[][] mat) {
        int m = mat.length, n = mat[0].length;
        int[][] dp = new int[m][n];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                if(i==m-1) dp[i][j] = mat[i][j];
                else {
                    int val = Integer.MAX_VALUE;
                    for(int col=0;col<n;col++) {
                        if(col!=j) val = Math.min(val, dp[i+1][col]);
                    }
                    dp[i][j] = val + mat[i][j];
                }
            }
        }
        int val = Integer.MAX_VALUE;
        for(int j=0;j<mat[0].length;j++) {
            val = Math.min(val, dp[0][j]);
        }
        return val;
    }
}
```

### Space Optimization
```java
class Solution {
    public int minFallingPathSum(int[][] mat) {
        int m = mat.length, n = mat[0].length;
        int[] prev = new int[n];
        for(int i=m-1;i>=0;i--) {
            int[] curr = new int[n];
            for(int j=n-1;j>=0;j--) {
                if(i==m-1) curr[j] = mat[i][j];
                else {
                    int val = Integer.MAX_VALUE;
                    for(int col=0;col<n;col++) {
                        if(col!=j) val = Math.min(val, prev[col]);
                    }
                    curr[j] = val + mat[i][j];
                }
            }
            prev = curr;
        }
        int val = Integer.MAX_VALUE;
        for(int j=0;j<mat[0].length;j++) {
            val = Math.min(val, prev[j]);
        }
        return val;
    }
}
```
