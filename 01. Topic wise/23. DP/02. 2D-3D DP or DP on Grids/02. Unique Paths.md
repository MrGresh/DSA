### Memoization
```java
class Solution {
    Integer[][] memo;
    public int uniquePaths(int m, int n) {
        memo = new Integer[m][n];
        return rec(0, 0, m, n);
    }
    int rec(int i, int j, int m, int n) {
        if(i==m-1 && j==n-1) return 1;
        if(memo[i][j]!=null) return memo[i][j];
        int val = 0;
        if(i+1<m) val += rec(i+1, j, m, n);
        if(j+1<n) val += rec(i, j+1, m, n);
        return memo[i][j] = val;
    }
}
```
### Tabulation
```java
class Solution {
    public int uniquePaths(int m, int n) {
        int[][] dp = new int[m][n];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                if(i==m-1 && j==n-1) dp[i][j] = 1;
                else {
                    int val = 0;
                    if(i+1<m) val += dp[i+1][j];
                    if(j+1<n) val += dp[i][j+1];
                    dp[i][j] = val;
                }
            }
        }
        return dp[0][0];
    }
}
```
### Space Optimized
```java
class Solution {
    public int uniquePaths(int m, int n) {
        int[] prev = new int[n];
        for(int i=m-1;i>=0;i--) {
            int[] curr = new int[n];
            for(int j=n-1;j>=0;j--) {
                if(i==m-1 && j==n-1) curr[j] = 1;
                else {
                    int val = 0;
                    if(i+1<m) val += prev[j];
                    if(j+1<n) val += curr[j+1];
                    curr[j] = val;
                }
            }
            prev = curr;
        }
        return prev[0];
    }
}
```
