### Recursion + Visited Array
```java
class Solution {
    public int longestIncreasingPath(int[][] mat) {
        int rows = mat.length, cols = mat[0].length;
        int ans = Integer.MIN_VALUE;
        for(int i=0;i<rows;i++) {
            for(int j=0;j<cols;j++) {
                ans = Math.max(ans, rec(mat, i, j, new boolean[rows][cols]));
            }
        }
        return ans;
    }
    int rec(int[][] mat, int i, int j, boolean[][] vis) {
        int rows = mat.length, cols = mat[0].length;
        vis[i][j] = true;
        int[][] dirs = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
        int ans = Integer.MIN_VALUE;
        for(int[] d: dirs) {
            int newI = i+d[0], newJ = j+d[1];
            if(newI>=0 && newI<rows && newJ>=0 && newJ<cols && !vis[newI][newJ] && mat[newI][newJ]>mat[i][j]) {
                ans = Math.max(ans, rec(mat, newI, newJ, vis));
            }
        }
        vis[i][j] = false;
        return 1 + (ans == Integer.MIN_VALUE ? 0 : ans);
    }
}
```
**Note:** Visited array not required because **mat[newI][newJ]>mat[i][j]** we will surely not going to take any visited cell again in same path.

### Memoization
```java
class Solution {
    Integer[][] memo;
    public int longestIncreasingPath(int[][] mat) {
        int rows = mat.length, cols = mat[0].length;
        memo = new Integer[rows][cols];
        int ans = Integer.MIN_VALUE;
        for(int i=0;i<rows;i++) {
            for(int j=0;j<cols;j++) {
                ans = Math.max(ans, rec(mat, i, j));
            }
        }
        return ans;
    }
    int rec(int[][] mat, int i, int j) {
        if(memo[i][j]!=null) return memo[i][j];
        int rows = mat.length, cols = mat[0].length;
        int[][] dirs = {{1, 0}, {0, 1}, {-1, 0}, {0, -1}};
        int ans = Integer.MIN_VALUE;
        for(int[] d: dirs) {
            int newI = i+d[0], newJ = j+d[1];
            if(newI>=0 && newI<rows && newJ>=0 && newJ<cols && mat[newI][newJ]>mat[i][j]) {
                ans = Math.max(ans, rec(mat, newI, newJ));
            }
        }
        return memo[i][j] = 1 + (ans == Integer.MIN_VALUE ? 0 : ans);
    }
}
```
