### Memoization
```java
class Solution {
    int[][][] memo; // 4 tables for 4 directions
    int n, m;

    public int solve(int[][] grid) {
        n = grid.length;
        m = grid[0].length;
        memo = new int[4][n + 1][m + 1];
        for (int d = 0; d < 4; d++) 
            for (int i = 0; i <= n; i++) Arrays.fill(memo[d][i], -1);

        int maxCalories = 0;
        // Meeting point (i, j) cannot be on the boundary
        for (int i = 1; i < n - 1; i++) {
            for (int j = 1; j < m - 1; j++) {
                // Scenario 1: Iahub moves Right, Iahubina moves Up
                int s1 = get(0, i, j-1, grid) + get(1, i, j+1, grid) + 
                         get(2, i+1, j, grid) + get(3, i-1, j, grid);
                
                // Scenario 2: Iahub moves Down, Iahubina moves Right
                int s2 = get(0, i-1, j, grid) + get(1, i+1, j, grid) + 
                         get(2, i, j-1, grid) + get(3, i, j+1, grid);
                
                maxCalories = Math.max(maxCalories, Math.max(s1, s2));
            }
        }
        return maxCalories;
    }

    private int get(int dir, int r, int c, int[][] grid) {
        if (r < 0 || r >= n || c < 0 || c >= m) return -100000000; // Large negative
        if (memo[dir][r][c] != -1) return memo[dir][r][c];

        int res = grid[r][c];
        if (dir == 0) { // Top-Left to (r,c)
            if (r == 0 && c == 0) return grid[r][c];
            res += Math.max(get(0, r-1, c, grid), get(0, r, c-1, grid));
        } else if (dir == 1) { // Bottom-Right to (r,c)
            if (r == n-1 && c == m-1) return grid[r][c];
            res += Math.max(get(1, r+1, c, grid), get(1, r, c+1, grid));
        } else if (dir == 2) { // Bottom-Left to (r,c)
            if (r == n-1 && c == 0) return grid[r][c];
            res += Math.max(get(2, r+1, c, grid), get(2, r, c-1, grid));
        } else { // Top-Right to (r,c)
            if (r == 0 && c == m-1) return grid[r][c];
            res += Math.max(get(3, r-1, c, grid), get(3, r, c+1, grid));
        }
        return memo[dir][r][c] = res;
    }
}
```

### Tabulation
```java
public int tabulation(int[][] a) {
    int n = a.length, m = a[0].length;
    int[][] tl = new int[n+2][m+2]; // Top-Left
    int[][] br = new int[n+2][m+2]; // Bottom-Right
    int[][] bl = new int[n+2][m+2]; // Bottom-Left
    int[][] tr = new int[n+2][m+2]; // Top-Right

    for(int i=1; i<=n; i++)
        for(int j=1; j<=m; j++)
            tl[i][j] = a[i-1][j-1] + Math.max(tl[i-1][j], tl[i][j-1]);

    for(int i=n; i>=1; i--)
        for(int j=m; j>=1; j--)
            br[i][j] = a[i-1][j-1] + Math.max(br[i+1][j], br[i][j+1]);

    for(int i=n; i>=1; i--)
        for(int j=1; j<=m; j++)
            bl[i][j] = a[i-1][j-1] + Math.max(bl[i+1][j], bl[i][j-1]);

    for(int i=1; i<=n; i++)
        for(int j=m; j>=1; j--)
            tr[i][j] = a[i-1][j-1] + Math.max(tr[i-1][j], tr[i][j+1]);

    int ans = 0;
    for(int i=2; i<n; i++) {
        for(int j=2; j<m; j++) {
            ans = Math.max(ans, tl[i-1][j] + br[i+1][j] + bl[i][j-1] + tr[i][j+1]);
            ans = Math.max(ans, tl[i][j-1] + br[i][j+1] + bl[i+1][j] + tr[i-1][j]);
        }
    }
    return ans;
}
```
