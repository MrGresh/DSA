### Memoization
```java
class Solution {
    Integer[][] memo;
    public int maximumPoints(int mat[][]) {
        memo = new Integer[mat.length][3];
        int val = Integer.MIN_VALUE;
        for(int col=0;col<3;col++) {
            val = Math.max(val, rec(mat, 0, col));
        }
        return val;
    }
    int rec(int[][] mat, int i, int j) {
        if(i==mat.length-1) return mat[i][j];
        if(memo[i][j]!=null) return memo[i][j];
        int val = Integer.MIN_VALUE;
        for(int col=0;col<3;col++) {
            if(col!=j) val = Math.max(val, rec(mat, i+1, col));
        }
        return memo[i][j] = mat[i][j]+val;
    }
}
```

### Tabulation
```java
class Solution {
    public int maximumPoints(int mat[][]) {
        int[][] dp = new int[mat.length][3];
        for(int i=mat.length-1;i>=0;i--) {
            for(int j=0;j<3;j++) {
                if(i==mat.length-1) dp[i][j] = mat[i][j];
                else {
                    int val = Integer.MIN_VALUE;
                    for(int col=0;col<3;col++) {
                        if(col!=j) val = Math.max(val, dp[i+1][col]);
                    }
                    dp[i][j] = mat[i][j]+val;
                }
            }
        }
        int val = Integer.MIN_VALUE;
        for(int col=0;col<3;col++) {
            val = Math.max(val, dp[0][col]);
        }
        return val;
    }
}
```

### Space Optimizatiom
```java
class Solution {
    public int maximumPoints(int mat[][]) {
        int[] prev = new int[3];
        for(int i=mat.length-1;i>=0;i--) {
            int[] curr = new int[3];
            for(int j=0;j<3;j++) {
                if(i==mat.length-1) curr[j] = mat[i][j];
                else {
                    int val = Integer.MIN_VALUE;
                    for(int col=0;col<3;col++) {
                        if(col!=j) val = Math.max(val, prev[col]);
                    }
                    curr[j] = mat[i][j]+val;
                }
            }
            prev = curr;
        }
        int val = Integer.MIN_VALUE;
        for(int col=0;col<3;col++) {
            val = Math.max(val, prev[col]);
        }
        return val;
    }
}
```
