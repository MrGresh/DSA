### Memoization
```java
class Solution {
    Integer[][][] memo;
    public int maxPathScore(int[][] grid, int k) {
        int m = grid.length, n = grid[0].length;
        memo = new Integer[m][n][k+1];
        int ans = rec(0, 0, k, m, n, grid);
        return ans == Integer.MIN_VALUE ? -1 : ans;
    }
    int rec(int i, int j, int k, int m, int n, int[][] grid) {
        if(i==m-1 && j==n-1) return grid[i][j];
        if(memo[i][j][k]!=null) return memo[i][j][k];

        int max = Integer.MIN_VALUE;
        if(i+1<m && k-getCost(grid[i+1][j])>=0) {
            max = rec(i+1, j, k-getCost(grid[i+1][j]), m, n, grid);
        }
        if(j+1<n && k-getCost(grid[i][j+1])>=0) {
            max = Math.max(max, rec(i, j+1, k-getCost(grid[i][j+1]), m, n, grid));
        }
        
        return memo[i][j][k] = (max == Integer.MIN_VALUE ? Integer.MIN_VALUE : grid[i][j] + max);
    }
    int getCost(int x) {
        if(x==0) return 0;
        else return 1;
    }
}
```

### Tabulation
```java
class Solution {
    public int maxPathScore(int[][] grid, int K) {
        int m = grid.length, n = grid[0].length;
        int[][][] dp = new int[m][n][K+1];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                for(int k=0;k<=K;k++) {
                    if(i==m-1 && j==n-1) dp[i][j][k] = grid[i][j];
                    else {
                        int max = Integer.MIN_VALUE;
                        if(i+1<m && k-getCost(grid[i+1][j])>=0) {
                            max = dp[i+1][j][k-getCost(grid[i+1][j])];
                        }
                        if(j+1<n && k-getCost(grid[i][j+1])>=0) {
                            max = Math.max(max, dp[i][j+1][k-getCost(grid[i][j+1])]);
                        }
                        
                        dp[i][j][k] = (max == Integer.MIN_VALUE ? Integer.MIN_VALUE : grid[i][j] + max);
                    }
                }
            }
        }
        int ans = dp[0][0][K];
        return ans == Integer.MIN_VALUE ? -1 : ans;
    }
    int getCost(int x) {
        if(x==0) return 0;
        else return 1;
    }
}
```

### Space Optimization
```java
class Solution {
    public int maxPathScore(int[][] grid, int K) {
        int m = grid.length, n = grid[0].length;
        int[][] prev = new int[n][K+1];
        for(int i=m-1;i>=0;i--) {
            int[][] curr = new int[n][K+1];
            for(int j=n-1;j>=0;j--) {
                for(int k=0;k<=K;k++) {
                    if(i==m-1 && j==n-1) curr[j][k] = grid[i][j];
                    else {
                        int max = Integer.MIN_VALUE;
                        if(i+1<m && k-getCost(grid[i+1][j])>=0) {
                            max = prev[j][k-getCost(grid[i+1][j])];
                        }
                        if(j+1<n && k-getCost(grid[i][j+1])>=0) {
                            max = Math.max(max, curr[j+1][k-getCost(grid[i][j+1])]);
                        }
                        
                        curr[j][k] = (max == Integer.MIN_VALUE ? Integer.MIN_VALUE : grid[i][j] + max);
                    }
                }
            }
            prev = curr;
        }
        int ans = prev[0][K];
        return ans == Integer.MIN_VALUE ? -1 : ans;
    }
    int getCost(int x) {
        if(x==0) return 0;
        else return 1;
    }
}
```
