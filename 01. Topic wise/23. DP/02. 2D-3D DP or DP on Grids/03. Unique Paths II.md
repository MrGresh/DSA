### Memoization
```java
class Solution {
    Integer[][] memo;
    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length, n = obstacleGrid[0].length;
        if(obstacleGrid[0][0]==1 || obstacleGrid[m-1][n-1]==1) return 0;
        memo = new Integer[m][n];
        return rec(obstacleGrid, m, n, 0, 0);
    }
    int rec(int[][] obstacleGrid, int m, int n, int i, int j) {
        if(i==m-1 && j==n-1) return 1;
        if(obstacleGrid[i][j]==1) return 0;
        if(memo[i][j]!=null) return memo[i][j];
        int val = 0;
        if(i+1<m) val += rec(obstacleGrid, m, n, i+1, j);
        if(j+1<n) val += rec(obstacleGrid, m, n, i, j+1);
        return memo[i][j] = val;
    }
}
```

### Tabulation
```java
class Solution {
    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length, n = obstacleGrid[0].length;
        if(obstacleGrid[0][0]==1 || obstacleGrid[m-1][n-1]==1) return 0;
        int[][] dp = new int[m][n];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                if(i==m-1 && j==n-1) dp[i][j] = 1;
                else if(obstacleGrid[i][j]==1) dp[i][j] = 0;
                else {
                    int val = 0;
                    if(i+1<m) val += dp[i+1][j];
                    if(j+1<n) val += dp[i][j+1];
                    dp[i][j] = val;
                }
            }
        }
        return dp[0][0];
    }
}
```

### Space Optimized
```java
class Solution {
    public int uniquePathsWithObstacles(int[][] obstacleGrid) {
        int m = obstacleGrid.length, n = obstacleGrid[0].length;
        if(obstacleGrid[0][0]==1 || obstacleGrid[m-1][n-1]==1) return 0;
        int[] prev = new int[n];
        for(int i=m-1;i>=0;i--) {
            int[] curr = new int[n];
            for(int j=n-1;j>=0;j--) {
                if(i==m-1 && j==n-1) curr[j] = 1;
                else if(obstacleGrid[i][j]==1) curr[j] = 0;
                else {
                    int val = 0;
                    if(i+1<m) val += prev[j];
                    if(j+1<n) val += curr[j+1];
                    curr[j] = val;
                }
            }
            prev = curr;
        }
        return prev[0];
    }
}
```
