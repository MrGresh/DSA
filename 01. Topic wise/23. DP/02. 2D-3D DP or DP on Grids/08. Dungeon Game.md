### Memoization
```java
class Solution {
    Integer[][] memo;

    public int calculateMinimumHP(int[][] dungeon) {
        int m = dungeon.length;
        int n = dungeon[0].length;
        memo = new Integer[m][n];
        return solve(dungeon, 0, 0);
    }

    int solve(int[][] dungeon, int i, int j) {
        int m = dungeon.length;
        int n = dungeon[0].length;

        // Base Case: Reached the Princess
        if (i == m - 1 && j == n - 1) {
            return Math.max(1, 1 - dungeon[i][j]);
        }

        if (memo[i][j] != null) return memo[i][j];

        int res = Integer.MAX_VALUE;

        // Check Down
        if (i + 1 < m) {
            res = Math.min(res, solve(dungeon, i + 1, j));
        }

        // Check Right
        if (j + 1 < n) {
            res = Math.min(res, solve(dungeon, i, j + 1));
        }

        // Minimum health needed to ENTER this cell
        memo[i][j] = Math.max(1, res - dungeon[i][j]);
        
        return memo[i][j];
    }
}
```

### Tabulation
```java
class Solution {
    public int calculateMinimumHP(int[][] dungeon) {
        int m = dungeon.length;
        int n = dungeon[0].length;
        int[][] dp = new int[m][n];
        for(int i=m-1;i>=0;i--) {
            for(int j=n-1;j>=0;j--) {
                if (i == m - 1 && j == n - 1) { // Base Case
                    dp[i][j] = Math.max(1, 1 - dungeon[i][j]);
                } else {
                    int res = Integer.MAX_VALUE;
                    if (i + 1 < m) res = Math.min(res, dp[i + 1][j]);
                    if (j + 1 < n) res = Math.min(res, dp[i][j+1]);
                    dp[i][j] = Math.max(1, res - dungeon[i][j]);
                }
            }
        }
        return dp[0][0];
    }
}
```

### Space Optimization
```java
class Solution {
    public int calculateMinimumHP(int[][] dungeon) {
        int m = dungeon.length;
        int n = dungeon[0].length;
        int[] prev = new int[n];
        for(int i=m-1;i>=0;i--) {
            int[] curr = new int[n];
            for(int j=n-1;j>=0;j--) {
                if (i == m - 1 && j == n - 1) { // Base Case
                    curr[j] = Math.max(1, 1 - dungeon[i][j]);
                } else {
                    int res = Integer.MAX_VALUE;
                    if (i + 1 < m) res = Math.min(res, prev[j]);
                    if (j + 1 < n) res = Math.min(res, curr[j+1]);
                    curr[j] = Math.max(1, res - dungeon[i][j]);
                }
            }
            prev = curr;
        }
        return prev[0];
    }
}
```
