### Memoization
```java
class Solution {
    Integer[][] memo;
    public int calculateMinimumHP(int[][] dungeon) {
        int m = dungeon.length, n = dungeon[0].length;
        memo = new Integer[m][n];
        return solve(0, 0, dungeon, m, n);
    }

    private int solve(int r, int c, int[][] dungeon, int m, int n) {
        // Base Case: Reached the Princess
        if (r == m - 1 && c == n - 1) {
            return Math.max(1, 1 - dungeon[r][c]);
        }
        if (memo[r][c] != null) return memo[r][c];

        int res = Integer.MAX_VALUE;
        // Try going Down
        if (r + 1 < m) res = Math.min(res, solve(r + 1, c, dungeon, m, n));
        // Try going Right
        if (c + 1 < n) res = Math.min(res, solve(r, c + 1, dungeon, m, n));

        // Health needed before entering this room
        int healthNeeded = Math.max(1, res - dungeon[r][c]);
        return memo[r][c] = healthNeeded;
    }
}
```

### Tabulation
```java
class Solution {
    public int calculateMinimumHP(int[][] dungeon) {
        int m = dungeon.length;
        int n = dungeon[0].length;
        int[][] dp = new int[m + 1][n + 1];

        // Fill with infinity as the "requirement" for out-of-bounds
        for (int i = 0; i <= m; i++) Arrays.fill(dp[i], Integer.MAX_VALUE);
        
        // Base case: To survive after the princess room, you need at least 1 HP
        dp[m][n - 1] = 1;
        dp[m - 1][n] = 1;

        for (int i = m - 1; i >= 0; i--) {
            for (int j = n - 1; j >= 0; j--) {
                int minNextHealth = Math.min(dp[i + 1][j], dp[i][j + 1]);
                dp[i][j] = Math.max(1, minNextHealth - dungeon[i][j]);
            }
        }

        return dp[0][0];
    }
}
```

### Space Optimization
```java
class Solution {
    public int calculateMinimumHP(int[][] dungeon) {
        int n = dungeon[0].length;
        int[] dp = new int[n + 1];
        Arrays.fill(dp, Integer.MAX_VALUE);
        
        // Logic for the bottom-right corner
        dp[n - 1] = 1; 

        for (int i = dungeon.length - 1; i >= 0; i--) {
            for (int j = n - 1; j >= 0; j--) {
                // If we are at the bottom-right corner, min health needed after is 1
                int next = (i == dungeon.length - 1 && j == n - 1) ? 1 : Math.min(dp[j], dp[j + 1]);
                dp[j] = Math.max(1, next - dungeon[i][j]);
            }
            dp[n] = Integer.MAX_VALUE; // Reset boundary for the next row
        }
        return dp[0];
    }
}
```
