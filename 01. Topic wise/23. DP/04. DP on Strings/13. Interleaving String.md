### Memoization
```java
class Solution {
    Boolean[][] memo;
    public boolean isInterleave(String s1, String s2, String s3) {
        if (s1.length() + s2.length() != s3.length()) return false;
        memo = new Boolean[s1.length() + 1][s2.length() + 1];
        return solve(0, 0, s1, s2, s3);
    }

    private boolean solve(int i, int j, String s1, String s2, String s3) {
        if (i == s1.length() && j == s2.length()) return true;
        if (memo[i][j] != null) return memo[i][j];

        boolean res = false;
        // Try taking from s1
        if (i < s1.length() && s1.charAt(i) == s3.charAt(i + j)) {
            res = solve(i + 1, j, s1, s2, s3);
        }
        // Try taking from s2 (only if s1 didn't already find a path)
        if (!res && j < s2.length() && s2.charAt(j) == s3.charAt(i + j)) {
            res = solve(i, j + 1, s1, s2, s3);
        }

        return memo[i][j] = res;
    }
}
```

### Tabulation
```java
class Solution {
    public boolean isInterleave(String s1, String s2, String s3) {
        int m = s1.length(), n = s2.length();
        if (m + n != s3.length()) return false;

        boolean[][] dp = new boolean[m + 1][n + 1];
        dp[0][0] = true;

        // Base case: filling the first row (s1 is empty)
        for (int j = 1; j <= n; j++) {
            dp[0][j] = dp[0][j - 1] && s2.charAt(j - 1) == s3.charAt(j - 1);
        }

        // Base case: filling the first column (s2 is empty)
        for (int i = 1; i <= m; i++) {
            dp[i][0] = dp[i - 1][0] && s1.charAt(i - 1) == s3.charAt(i - 1);
        }

        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                // Can we reach this by taking a char from s1?
                boolean fromS1 = dp[i - 1][j] && s1.charAt(i - 1) == s3.charAt(i + j - 1);
                // Can we reach this by taking a char from s2?
                boolean fromS2 = dp[i][j - 1] && s2.charAt(j - 1) == s3.charAt(i + j - 1);
                
                dp[i][j] = fromS1 || fromS2;
            }
        }

        return dp[m][n];
    }
}
```

### Space Optimization
```java
class Solution {
    public boolean isInterleave(String s1, String s2, String s3) {
        int m = s1.length(), n = s2.length();
        if (m + n != s3.length()) return false;

        boolean[] dp = new boolean[n + 1];
        dp[0] = true;

        // Initialize first row
        for (int j = 1; j <= n; j++) {
            dp[j] = dp[j - 1] && s2.charAt(j - 1) == s3.charAt(j - 1);
        }

        for (int i = 1; i <= m; i++) {
            // dp[0] represents the first column (s2 is empty)
            dp[0] = dp[0] && s1.charAt(i - 1) == s3.charAt(i - 1);
            for (int j = 1; j <= n; j++) {
                // dp[j] (before update) is dp[i-1][j]
                // dp[j-1] (after update) is dp[i][j-1]
                dp[j] = (dp[j] && s1.charAt(i - 1) == s3.charAt(i + j - 1)) ||
                        (dp[j - 1] && s2.charAt(j - 1) == s3.charAt(i + j - 1));
            }
        }

        return dp[n];
    }
}
```
