### Memoization
```java
class Solution {
    public int numDistinct(String s, String t) {
        // Initialize dp array with -1
        int[][] dp = new int[s.length()][t.length()];
        for (int[] row : dp) Arrays.fill(row, -1);

        // Start recursion
        return helper(0, 0, s, t, dp);
    }
    int helper(int i, int j, String s, String t, int[][] dp) {
        // If we have matched all characters of t
        if (j == t.length()) return 1;

        // If we have exhausted s but not t
        if (i == s.length()) return 0;

        // If already computed
        if (dp[i][j] != -1) return dp[i][j];

        // If characters match
        if (s.charAt(i) == t.charAt(j)) {
            // Option 1: take current character
            int take = helper(i + 1, j + 1, s, t, dp);

            // Option 2: skip current character of s
            int notTake = helper(i + 1, j, s, t, dp);

            // Store result
            dp[i][j] = take + notTake;
        } else {
            // If mismatch, skip character of s
            dp[i][j] = helper(i + 1, j, s, t, dp);
        }
        return dp[i][j];
    }
}
```

### Tabulation
```java
class Solution {
    public int numDistinct(String s, String t) {
        // Get lengths of both strings
        int n = s.length();
        int m = t.length();

        // Create dp table of size (n+1) x (m+1)
        long[][] dp = new long[n + 1][m + 1];

        // Base case: empty t can be formed from any suffix of s
        for (int i = 0; i <= n; i++) {
            dp[i][m] = 1;
        }

        // Fill dp table from bottom to top
        for (int i = n - 1; i >= 0; i--) {
            for (int j = m - 1; j >= 0; j--) {
                // If characters match, we have two options:
                // 1. Take this character -> dp[i+1][j+1]
                // 2. Skip this character -> dp[i+1][j]
                if (s.charAt(i) == t.charAt(j)) {
                    dp[i][j] = dp[i + 1][j + 1] + dp[i + 1][j];
                } else {
                    // If characters don't match, we can only skip
                    dp[i][j] = dp[i + 1][j];
                }
            }
        }

        // Final answer is stored at dp[0][0]
        return (int) dp[0][0];
    }
}
```

### Space Optimization
```java
class Solution {
    public int numDistinct(String s, String t) {
        // Length of source string
        int n = s.length();
        // Length of target string
        int m = t.length();

        // Create dp array of size m+1
        long[] dp = new long[m + 1];

        // Base case: Empty target string can always be formed
        dp[m] = 1;

        // Traverse source string from end to start
        for (int i = n - 1; i >= 0; i--) {
            // Create copy for current iteration
            long[] curr = Arrays.copyOf(dp, m + 1);

            // Traverse target string from end to start
            for (int j = m - 1; j >= 0; j--) {
                // If characters match, take it or skip it
                if (s.charAt(i) == t.charAt(j)) {
                    curr[j] = dp[j + 1] + dp[j];
                } 
                // Else only skip option
                else {
                    curr[j] = dp[j];
                }
            }

            // Update dp
            dp = curr;
        }

        // Return answer
        return (int)dp[0];
    }
}
```
