int minOperations(String s1, String s2) {
    int n = s1.length();
    int m = s2.length();
    
    // Step 1: Find LCS Length
    int lcsLength = longestCommonSubsequence(s1, s2);
    
    // Step 2: Calculate Deletions and Insertions
    int deletions = n - lcsLength;
    int insertions = m - lcsLength;
    
    return deletions + insertions;
}
