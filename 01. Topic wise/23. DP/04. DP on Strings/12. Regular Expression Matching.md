### Memoization
```java
class Solution {
    Boolean[][] memo;

    public boolean isMatch(String s, String p) {
        memo = new Boolean[s.length() + 1][p.length() + 1];
        return solve(0, 0, s, p);
    }

    private boolean solve(int i, int j, String s, String p) {
        if (memo[i][j] != null) return memo[i][j];
        
        // Base Case: If we reach the end of the pattern
        if (j == p.length()) return i == s.length();

        // Check if current characters match
        boolean firstMatch = (i < s.length() && 
                             (p.charAt(j) == s.charAt(i) || p.charAt(j) == '.'));

        boolean ans;
        // Handle '*' wildcard
        if (j + 1 < p.length() && p.charAt(j + 1) == '*') {
            // Choice 1: Skip '*' (zero occurrences)
            // Choice 2: Use '*' (if first chars match, move s pointer)
            ans = solve(i, j + 2, s, p) || (firstMatch && solve(i + 1, j, s, p));
        } else {
            // Standard match: move both pointers
            ans = firstMatch && solve(i + 1, j + 1, s, p);
        }

        return memo[i][j] = ans;
    }
}
```

### Tabulation
```java
class Solution {
    public boolean isMatch(String s, String p) {
        int m = s.length(), n = p.length();
        boolean[][] dp = new boolean[m + 1][n + 1];
        dp[0][0] = true;

        // Handle patterns like a*, a*b*, a*b*c* matching empty string
        for (int j = 2; j <= n; j++) {
            if (p.charAt(j - 1) == '*') {
                dp[0][j] = dp[0][j - 2];
            }
        }

        for (int i = 1; i <= m; i++) {
            for (int j = 1; j <= n; j++) {
                if (p.charAt(j - 1) == '*') {
                    // Zero occurrences
                    dp[i][j] = dp[i][j - 2];
                    // One or more occurrences
                    if (p.charAt(j - 2) == s.charAt(i - 1) || p.charAt(j - 2) == '.') {
                        dp[i][j] = dp[i][j] || dp[i - 1][j];
                    }
                } else {
                    // Regular match or '.'
                    if (p.charAt(j - 1) == s.charAt(i - 1) || p.charAt(j - 1) == '.') {
                        dp[i][j] = dp[i - 1][j - 1];
                    }
                }
            }
        }
        return dp[m][n];
    }
}
```

### Space Optimized
```java
class Solution {
    public boolean isMatch(String s, String p) {
        int m = s.length(), n = p.length();
        boolean[] prev = new boolean[n + 1];
        prev[0] = true;

        for (int j = 2; j <= n; j++) {
            if (p.charAt(j - 1) == '*') prev[j] = prev[j - 2];
        }

        for (int i = 1; i <= m; i++) {
            boolean[] curr = new boolean[n + 1];
            for (int j = 1; j <= n; j++) {
                if (p.charAt(j - 1) == '*') {
                    curr[j] = curr[j - 2] || 
                             ((p.charAt(j - 2) == s.charAt(i - 1) || p.charAt(j - 2) == '.') && prev[j]);
                } else {
                    if (p.charAt(j - 1) == s.charAt(i - 1) || p.charAt(j - 1) == '.') {
                        curr[j] = prev[j - 1];
                    }
                }
            }
            prev = curr;
        }
        return prev[n];
    }
}
```
