### Memoization
```java
class Solution {
    public int longCommSubstr(String str1, String str2) {
        int n = str1.length();
        int m = str2.length();
        
        // Memoization table
        int[][] dp = new int[n + 1][m + 1];
        for (int[] row : dp) Arrays.fill(row, -1);
        
        // Use an array of size 1 to store max length by reference
        int[] maxLen = new int[1];
        
        // We need to trigger the recursion for all possible ending pairs
        // because the substring could end anywhere.
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                solve(i, j, str1, str2, dp, maxLen);
            }
        }
        
        return maxLen[0];
    }

    private int solve(int i, int j, String s1, String s2, int[][] dp, int[] maxLen) {
        // Base case: if either string is exhausted
        if (i == 0 || j == 0) {
            return 0;
        }

        // Return memoized value
        if (dp[i][j] != -1) {
            return dp[i][j];
        }

        // Standard substring logic:
        // We only care about the diagonal (i-1, j-1)
        int count = 0;
        if (s1.charAt(i - 1) == s2.charAt(j - 1)) {
            count = 1 + solve(i - 1, j - 1, s1, s2, dp, maxLen);
        } else {
            count = 0;
        }

        // Update global maximum
        maxLen[0] = Math.max(maxLen[0], count);
        
        return dp[i][j] = count;
    }
}
```

### Tabulation
```java
class Solution {
    public int longCommSubstr(String str1, String str2) {
        int n = str1.length();
        int m = str2.length();
    
        // DP table: dp[i][j] stores length of common substring ending at str1[i-1] and str2[j-1]
        int[][] dp = new int[n + 1][m + 1];

        // Variable to store the maximum substring length found so far
        int ans = 0;

        // Fill the DP table
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                // If characters match, extend the substring
                if (str1.charAt(i - 1) == str2.charAt(j - 1)) {
                    dp[i][j] = 1 + dp[i - 1][j - 1];
                    ans = Math.max(ans, dp[i][j]); // Update maximum length
                } else {
                    dp[i][j] = 0; // Reset length when characters don't match
                }
            }
        }
        return ans; // Return the longest length found
    }
}
```

### Space Optimized
```java
class Solution {
    public int longCommSubstr(String str1, String str2) {
        int n = str1.length();
        int m = str2.length();
    
        // Arrays to store previous and current row values
        int[] prev = new int[m + 1];
        int[] cur = new int[m + 1];
        
        // Variable to store the maximum LCS length found
        int ans = 0; 
    
        // Loop through both strings
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                
                // If characters match, extend the common substring
                if (str1.charAt(i - 1) == str2.charAt(j - 1)) {
                    int val = 1 + prev[j - 1];
                    cur[j] = val;
                    
                    // Update maximum substring length found so far
                    ans = Math.max(ans, val);
                } 
                // If characters don't match, reset to 0
                else {
                    cur[j] = 0;
                }
            }
            
            // Move current row to previous for next iteration
            System.arraycopy(cur, 0, prev, 0, m + 1);
        }
        
        // Return the maximum length found
        return ans;
    }
}
```
