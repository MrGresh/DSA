### Memoization
```java
class Solution {
    // Main function to start the matching process
    public boolean isMatch(String s, String p) {
        int n = p.length();
        int m = s.length();

        // DP array initialized to -1 (uncomputed)
        // Rows = Pattern index, Cols = String index
        int[][] dp = new int[n][m];
        for (int[] row : dp) Arrays.fill(row, -1);

        return wildcardMatchingUtil(p, s, n - 1, m - 1, dp);
    }

    // Recursive function with memoization to check wildcard matching
    public boolean wildcardMatchingUtil(String S1, String S2, int i, int j, int[][] dp) {
        // Base Case 1: Both pattern and text are fully matched
        if (i < 0 && j < 0) return true;

        // Base Case 2: Pattern exhausted but text still remains
        if (i < 0 && j >= 0) return false;

        // Base Case 3: Text exhausted but pattern may have '*' left
        if (j < 0 && i >= 0) return isAllStars(S1, i);

        // If already computed, return stored result
        if (dp[i][j] != -1) return dp[i][j] == 1;

        // If characters match or pattern has '?'
        if (S1.charAt(i) == S2.charAt(j) || S1.charAt(i) == '?')
            dp[i][j] = wildcardMatchingUtil(S1, S2, i - 1, j - 1, dp) ? 1 : 0;

        // If pattern has '*', check both possibilities
        else if (S1.charAt(i) == '*')
            dp[i][j] = (wildcardMatchingUtil(S1, S2, i - 1, j, dp) ||
                        wildcardMatchingUtil(S1, S2, i, j - 1, dp)) ? 1 : 0;

        // If characters don't match
        else dp[i][j] = 0;

        return dp[i][j] == 1;
    }

    // Function to check if substring S1[0..i] contains only '*'
    public boolean isAllStars(String S1, int i) {
        for (int j = 0; j <= i; j++) {
            if (S1.charAt(j) != '*')
                return false;
        }
        return true;
    }
}
```

### Tabulation
```java
class Solution {
    public boolean isMatch(String S1, String S2) {
        return rec(S2, S1);
    }
    // Function to check if the pattern S1 matches the string S2
    public boolean rec(String S1, String S2) {
        // Get lengths of pattern and text
        int n = S1.length();
        int m = S2.length();

        // Create a DP table where dp[i][j] tells if pattern[0...i-1] matches text[0...j-1]
        boolean[][] dp = new boolean[n + 1][m + 1];

        // Base case: empty pattern matches empty string
        dp[0][0] = true;

        // Base case: empty pattern cannot match non-empty string
        for (int j = 1; j <= m; j++) {
            dp[0][j] = false;
        }

        // Base case: pattern can match empty string only if it contains all '*'
        for (int i = 1; i <= n; i++) {
            dp[i][0] = isAllStars(S1, i);
        }

        // Fill the DP table row by row
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {

                // Case 1: Characters match exactly or pattern has '?'
                if (S1.charAt(i - 1) == S2.charAt(j - 1) || S1.charAt(i - 1) == '?') {
                    dp[i][j] = dp[i - 1][j - 1];
                }

                // Case 2: Pattern has '*'
                else if (S1.charAt(i - 1) == '*') {
                    // '*' matches zero characters (dp[i-1][j]) OR
                    // '*' matches one/more characters (dp[i][j-1])
                    dp[i][j] = dp[i - 1][j] || dp[i][j - 1];
                }

                // Case 3: No match
                else {
                    dp[i][j] = false;
                }
            }
        }

        // Final answer is stored at dp[n][m]
        return dp[n][m];
    }
    // Helper function to check if the first i characters of the pattern are all '*'
    public boolean isAllStars(String S1, int i) {
        // Loop through the first i characters
        for (int j = 1; j <= i; j++) {
            // If any character is not '*', return false
            if (S1.charAt(j - 1) != '*')
                return false;
        }
        // All were '*', so return true
        return true;
    }
}
```

### Space Optimization
```java
class Solution {
    public boolean isMatch(String s, String p) {
        return wildcardMatching(p, s);
    }

    // Function to perform wildcard matching
    public boolean wildcardMatching(String S1, String S2) {
        int n = S1.length();
        int m = S2.length();

        // Previous row results
        boolean[] prev = new boolean[m + 1];
        // Current row results
        boolean[] cur = new boolean[m + 1];

        // Empty pattern matches empty string
        prev[0] = true;

        // Iterate over each character of pattern
        for (int i = 1; i <= n; i++) {
            // Check if first column should be true
            cur[0] = isAllStars(S1, i);

            // Iterate over each character of text
            for (int j = 1; j <= m; j++) {
                char pChar = S1.charAt(i - 1);
                char tChar = S2.charAt(j - 1);

                if (pChar == tChar || pChar == '?') {
                    cur[j] = prev[j - 1];
                } else if (pChar == '*') {
                    cur[j] = prev[j] || cur[j - 1];
                } else {
                    cur[j] = false;
                }
            }

            // Move current row to previous
            prev = Arrays.copyOf(cur, m + 1);
        }

        return prev[m];
    }
    
    // Helper function to check if all characters in S1[0...i-1] are '*'
    public boolean isAllStars(String S1, int i) {
        for (int j = 1; j <= i; j++) {
            if (S1.charAt(j - 1) != '*') return false;
        }
        return true;
    }
}
```
