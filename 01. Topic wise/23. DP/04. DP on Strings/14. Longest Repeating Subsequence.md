### Memoization
```java
class Solution {
    Integer[][] memo;
    public int LongestRepeatingSubsequence(String s) {
        int n = s.length();
        memo = new Integer[n][n];
        return solve(0, 0, s, n);
    }

    private int solve(int i, int j, String s, int n) {
        if (i == n || j == n) return 0;
        if (memo[i][j] != null) return memo[i][j];

        // If characters match AND indices are different
        if (s.charAt(i) == s.charAt(j) && i != j) {
            return memo[i][j] = 1 + solve(i + 1, j + 1, s, n);
        } else {
            // Standard LCS logic: move one pointer at a time
            return memo[i][j] = Math.max(solve(i + 1, j, s, n), solve(i, j + 1, s, n));
        }
    }
}
```
### Tabulation
```java
class Solution {
    public int LongestRepeatingSubsequence(String s) {
        int n = s.length();
        int[][] dp = new int[n + 1][n + 1];

        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                // Key constraint: characters match but indices i-1 and j-1 are different
                if (s.charAt(i - 1) == s.charAt(j - 1) && i != j) {
                    dp[i][j] = 1 + dp[i - 1][j - 1];
                } else {
                    dp[i][j] = Math.max(dp[i - 1][j], dp[i][j - 1]);
                }
            }
        }
        return dp[n][n];
    }
}
```

### Space Optimization
```java
class Solution {
    public int LongestRepeatingSubsequence(String s) {
        int n = s.length();
        int[] dp = new int[n + 1];

        for (int i = 1; i <= n; i++) {
            int prev = 0; // Represents dp[i-1][j-1]
            for (int j = 1; j <= n; j++) {
                int temp = dp[j]; // Store current dp[j] to use as 'prev' in next j iteration
                if (s.charAt(i - 1) == s.charAt(j - 1) && i != j) {
                    dp[j] = 1 + prev;
                } else {
                    dp[j] = Math.max(dp[j], dp[j - 1]);
                }
                prev = temp;
            }
        }
        return dp[n];
    }
}
```
