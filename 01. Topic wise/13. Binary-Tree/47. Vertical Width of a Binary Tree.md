class Solution {
    int minDist = 0;
    int maxDist = 0;
    public int verticalWidth(Node root) {
        if (root == null) return 0;
        minDist = 0;
        maxDist = 0;
        findBounds(root, 0);
        return maxDist - minDist + 1;
    }
    private void findBounds(Node node, int hd) {
        if (node == null) return;
        if (hd < minDist) minDist = hd;
        if (hd > maxDist) maxDist = hd;
        findBounds(node.left, hd - 1);
        findBounds(node.right, hd + 1);
    }
}
