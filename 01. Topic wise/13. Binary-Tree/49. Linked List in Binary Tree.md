class Solution {
    public boolean isSubPath(ListNode head, TreeNode root) {
        if(root==null) return false;
        if(isExist(head, root)) return true;
        return isSubPath(head, root.left) || isSubPath(head, root.right);
    }
    boolean isExist(ListNode head, TreeNode root) {
        if(head==null) return true;
        if(root==null) return false;
        return head.val==root.val && (isExist(head.next, root.left) || isExist(head.next, root.right));
    }
}
