class Solution {
    public ArrayList<Integer> verticalSum(Node root) {
        // TreeMap to store sum for each horizontal distance (HD)
        // Keys are HD, Values are the running sum for that HD
        TreeMap<Integer, Integer> map = new TreeMap<>();
        
        // Helper function to traverse the tree
        calculateSum(root, 0, map);
        
        // Prepare the final result from the sorted map
        ArrayList<Integer> result = new ArrayList<>();
        for (int sum : map.values()) {
            result.add(sum);
        }
        
        return result;
    }
    
    private void calculateSum(Node node, int hd, TreeMap<Integer, Integer> map) {
        if (node == null) return;
        
        // Update the sum for the current horizontal distance
        map.put(hd, map.getOrDefault(hd, 0) + node.data);
        
        // Move left: HD decreases by 1
        calculateSum(node.left, hd - 1, map);
        
        // Move right: HD increases by 1
        calculateSum(node.right, hd + 1, map);
    }
}
