class Solution {
    TreeMap<Integer, Integer> map;
    public ArrayList<Integer> verticalSum(Node root) {
        map = new TreeMap<>();
        
        calculateSum(root, 0);
        
        ArrayList<Integer> result = new ArrayList<>();
        for (int sum : map.values()) result.add(sum);
        
        return result;
    }
    
    private void calculateSum(Node node, int hd) {
        if (node == null) return;
        
        map.put(hd, map.getOrDefault(hd, 0) + node.data);
        
        calculateSum(node.left, hd - 1);
        
        calculateSum(node.right, hd + 1);
    }
}
