class Solution {
    Node head = null;
    Node prev = null;
    Node bToDLL(Node root) {
        if (root == null) return null;
        head = null;
        prev = null;
        convertToDLL(root);
        return head;
    }
    private void convertToDLL(Node root) {
        if (root == null) return;
        convertToDLL(root.left);
        if (prev == null) head = root;
        else {
            root.left = prev;
            prev.right = root;
        }
        prev = root;
        convertToDLL(root.right);
    }
}
