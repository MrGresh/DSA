## Top-Down
```java
class Solution {
    int ans;
    int maxDiff(Node root) {
        ans = Integer.MIN_VALUE;
        rec(root, root.data);
        return ans;
    }
    void rec(Node root, int curr) {
        if(root==null) return;
        int max = Math.max(curr, root.data);
        if(root.left!=null) ans = Math.max(ans, max-root.left.data);
        if(root.right!=null) ans = Math.max(ans, max-root.right.data);
        rec(root.left, max);
        rec(root.right, max);
    }
}
```
## Bottom-Up
```java
class Solution {
    int ans;
    int maxDiff(Node root) {
        ans = Integer.MIN_VALUE;
        rec(root);
        return ans;
    }
    int rec(Node root) {
        if(root.left==null && root.right==null) return root.data;
        int minVal = Integer.MAX_VALUE;
        if(root.left!=null) minVal = Math.min(minVal, rec(root.left));
        if(root.right!=null) minVal = Math.min(minVal, rec(root.right));
        ans = Math.max(ans, root.data-minVal);
        return Math.min(root.data, minVal);
    }
}
```
