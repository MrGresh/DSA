```java
class Solution {
    List<Integer> pre, in, post;
    public List<List<Integer>> allTraversal(TreeNode root) {
        pre = new ArrayList<>();
        in = new ArrayList<>();
        post = new ArrayList<>();
        rec(root);
        List<List<Integer>> result = new ArrayList<>();
        result.add(pre);
        result.add(in);
        result.add(post);
        return result;
    }
    private void rec(TreeNode root) {
        if (root == null) return;
        pre.add(root.val);
        rec(root.left);
        in.add(root.val);
        rec(root.right);
        post.add(root.val);
    }
}
```

// Iterative
```java
class Pair {
    TreeNode node;
    int state;
    Pair(TreeNode node, int state) {
        this.node = node;
        this.state = state;
    }
}
class Solution {
    public List<List<Integer>> allTraversal(TreeNode root) {
        List<Integer> pre = new ArrayList<>();
        List<Integer> in = new ArrayList<>();
        List<Integer> post = new ArrayList<>();
        
        if (root == null) return Arrays.asList(pre, in, post);

        Stack<Pair> st = new Stack<>();
        st.push(new Pair(root, 1));

        while (!st.isEmpty()) {
            Pair it = st.peek();

            if (it.state == 1) {
                pre.add(it.node.val);
                it.state++;
                if (it.node.left != null) {
                    st.push(new Pair(it.node.left, 1));
                }
            } 
            else if (it.state == 2) {
                in.add(it.node.val);
                it.state++;
                if (it.node.right != null) {
                    st.push(new Pair(it.node.right, 1));
                }
            } 
            else {
                post.add(it.node.val);
                st.pop();
            }
        }

        return Arrays.asList(pre, in, post);
    }
}
```
