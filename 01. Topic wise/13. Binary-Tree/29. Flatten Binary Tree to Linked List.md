class Solution {
    List<TreeNode> preorder;
    public void flatten(TreeNode root) {
        if(root==null) return;
        preorder = new ArrayList<>();
        rec(root);
        for(int i=0;i<preorder.size()-1;i++) {
            preorder.get(i).right = preorder.get(i+1);
            preorder.get(i).left = null;
        }
        preorder.get(preorder.size()-1).left=null;
    }
    void rec(TreeNode root) {
        if(root==null) return;
        preorder.add(root);
        rec(root.left);
        rec(root.right);
    }
}

class Solution {
    private TreeNode prev = null;
    public void flatten(TreeNode root) {
        if (root == null) return;

        flatten(root.right);
        flatten(root.left);

        root.right = prev;
        root.left = null;
        
        prev = root;
    }
}

public void flatten(TreeNode root) {
    TreeNode curr = root;
    while (curr != null) {
        if (curr.left != null) {
            // Find the rightmost node of the left subtree
            TreeNode last = curr.left;
            while (last.right != null) last = last.right;
            
            // Connection logic
            last.right = curr.right; // Move original right subtree
            curr.right = curr.left;  // Move left subtree to the right
            curr.left = null;        // Clean up left
        }
        curr = curr.right; // Move to the next node in the "linked list"
    }
}
