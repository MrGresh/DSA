class Solution {
    ArrayList<Integer> res = new ArrayList<>();

    ArrayList<Integer> boundaryTraversal(Node root) {
        if (root == null) return res;
        
        if (!isLeaf(root)) res.add(root.data);
        
        addLeftBoundary(root.left);
        
        addLeaves(root);
        
        addRightBoundary(root.right);
        
        return res;
    }

    boolean isLeaf(Node node) {
        return node.left == null && node.right == null;
    }

    void addLeftBoundary(Node node) {
        if (node == null || isLeaf(node)) return;
        res.add(node.data);
        if (node.left != null) addLeftBoundary(node.left);
        else addLeftBoundary(node.right);
    }

    void addLeaves(Node node) {
        if (node == null) return;
        if (isLeaf(node)) {
            res.add(node.data);
            return;
        }
        addLeaves(node.left);
        addLeaves(node.right);
    }

    void addRightBoundary(Node node) {
        if (node == null || isLeaf(node)) return;
        if (node.right != null) addRightBoundary(node.right);
        else addRightBoundary(node.left);
        res.add(node.data);
    }
}
