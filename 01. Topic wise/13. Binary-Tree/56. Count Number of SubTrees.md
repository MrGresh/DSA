class Solution {
    int count;
    int countSubtrees(Node root, int x) {
        count = 0;
        rec(root, x);
        return count;
    }
    int rec(Node root, int x) {
        if(root==null) return 0;
        int sum = root.data + rec(root.left, x) + rec(root.right, x);
        if(sum==x) count++;
        return sum;
    }
}
