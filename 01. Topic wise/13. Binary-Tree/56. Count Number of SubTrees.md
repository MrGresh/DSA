```java
class Solution {
    int count;
    int countSubtrees(Node root, int x) {
        count = 0;
        rec(root, x);
        return count;
    }
    int rec(Node root, int x) {
        if(root==null) return 0;
        int lSum = rec(root.left, x);
        int rSum = rec(root.right, x);
        int sum = root.data+lSum+rSum;
        if(sum==x) count++;
        return sum;
    }
}
```
