class Solution {
    static class Node {
        int freq;
        int firstIdx;
        Node left, right;

        Node(int freq, int firstIdx) {
            this.freq = freq;
            this.firstIdx = firstIdx;
        }
    }
    ArrayList<String> result;
    public ArrayList<String> huffmanCodes(String s, int[] f) {

        PriorityQueue<Node> pq = new PriorityQueue<>((a, b) -> {
            return (a.freq != b.freq) ? (a.freq - b.freq) : (a.firstIdx - b.firstIdx);
        });

        for (int i = 0; i < s.length(); i++) pq.offer(new Node(f[i], i));

        while (pq.size() > 1) {
            Node left  = pq.poll();
            Node right = pq.poll();

            Node merged = new Node(
                left.freq + right.freq,
                Math.min(left.firstIdx, right.firstIdx)
            );
            merged.left  = left;
            merged.right = right;

            pq.offer(merged);
        }

        Node root = pq.poll();
        result = new ArrayList<>();

        if (root.left == null && root.right == null) {
            result.add("0");
            return result;
        }

        preorder(root, "");
        return result;
    }

    void preorder(Node node, String code) {
        if (node == null) return;
        if (node.left == null && node.right == null) {
            result.add(code);
            return;
        }
        preorder(node.left,  code + "0");
        preorder(node.right, code + "1");
    }
}
