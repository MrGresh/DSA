class Pair {
    int include;
    int exclude;
    Pair(int inc, int exc) {
        this.include = inc;
        this.exclude = exc;
    }
}
class Solution {
    public int getMaxSum(Node root) {
        Pair res = solve(root);
        return Math.max(res.include, res.exclude);
    }
    private Pair solve(Node root) {
        if (root == null) {
            return new Pair(0, 0);
        }

        Pair left = solve(root.left);
        Pair right = solve(root.right);

        int inc = root.data + left.exclude + right.exclude;
        int exc = Math.max(left.include, left.exclude) + Math.max(right.include, right.exclude);
        return new Pair(inc, exc);
    }
}
