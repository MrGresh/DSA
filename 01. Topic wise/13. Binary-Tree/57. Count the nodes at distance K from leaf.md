class Solution {
    int cnt;
    Set<Node> specialNodes;
    Node[] path;
    int k;
    int kthFromLeaf(Node root, int K) {
        cnt = 0;
        path = new Node[100001];
        specialNodes = new HashSet<>();
        k = K;
        solve(root, 0);
        return specialNodes.size();
    }
    void solve(Node node, int level) {
        if (node == null) return;
        path[level] = node;

        // Check if current node is a leaf
        if (node.left == null && node.right == null) {
            // If it's a leaf, look back k distance in the path
            if (level >= k) {
                specialNodes.add(path[level - k]);
            }
            return;
        }
        
        solve(node.left, level + 1);
        solve(node.right, level + 1);
    }
}
