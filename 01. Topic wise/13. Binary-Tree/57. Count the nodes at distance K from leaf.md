class Solution {
    int cnt;
    int kthFromLeaf(Node root, int k) {
        cnt = 0;
        Node[] path = new Node[100001];
        Set<Node> specialNodes = new HashSet<>();
        solve(root, k, path, 0, specialNodes);
        return specialNodes.size();
    }
    void solve(Node node, int k, Node[] path, int level, Set<Node> result) {
        if (node == null) return;
        path[level] = node;

        // Check if current node is a leaf
        if (node.left == null && node.right == null) {
            // If it's a leaf, look back k distance in the path
            if (level >= k) {
                result.add(path[level - k]);
            }
            return;
        }
        
        solve(node.left, k, path, level + 1, result);
        solve(node.right, k, path, level + 1, result);
    }
}
