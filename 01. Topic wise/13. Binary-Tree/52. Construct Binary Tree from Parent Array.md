class Solution {
    public Node createTree(int[] parent) {
        int n = parent.length;
        Node[] nodes = new Node[n];
        Node root = null;

        // Step 1: Create all nodes first
        for (int i = 0; i < n; i++) {
            nodes[i] = new Node(i);
        }

        // Step 2: Establish parent-child relationships
        for (int i = 0; i < n; i++) {
            if (parent[i] == -1) {
                // This is the root of the tree
                root = nodes[i];
            } else {
                Node p = nodes[parent[i]];
                
                // First appearance goes to the left child
                if (p.left == null) {
                    p.left = nodes[i];
                } else {
                    // Second appearance goes to the right child
                    p.right = nodes[i];
                }
            }
        }

        return root;
    }
}
