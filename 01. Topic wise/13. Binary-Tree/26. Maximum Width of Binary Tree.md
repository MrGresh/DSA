class AnnotatedNode {
    TreeNode node;
    int index;

    AnnotatedNode(TreeNode node, int index) {
        this.node = node;
        this.index = index;
    }
}

class Solution {
    public int widthOfBinaryTree(TreeNode root) {
        if (root == null) return 0;

        int maxWidth = 0;
        Queue<AnnotatedNode> queue = new LinkedList<>();
        queue.offer(new AnnotatedNode(root, 0));

        while (!queue.isEmpty()) {
            int size = queue.size();
            int start = 0;
            int end = 0;

            for (int i = 0; i < size; i++) {
                AnnotatedNode current = queue.poll();
                TreeNode node = current.node;
                int index = current.index;

                if (i == 0) start = index;
                if (i == size - 1) end = index;

                if (node.left != null) {
                    queue.offer(new AnnotatedNode(node.left, 2 * (index - start)));
                }
                if (node.right != null) {
                    queue.offer(new AnnotatedNode(node.right, 2 * (index - start) + 1));
                }
            }
            maxWidth = Math.max(maxWidth, end - start + 1);
        }

        return maxWidth;
    }
}
