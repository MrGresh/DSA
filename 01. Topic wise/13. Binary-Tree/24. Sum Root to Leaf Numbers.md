class Solution {
    int res = 0;
    public int sumNumbers(TreeNode root) {
        if(root==null) return 0;
        rec(root, 0);
        return res;
    }
    void rec(TreeNode root, int sum) {
        if(root==null) return;
        sum = sum*10 + root.val;
        if(root.left==null && root.right==null) {
            res += sum;
            return;
        }
        rec(root.left, sum);
        rec(root.right, sum);
    }
}
