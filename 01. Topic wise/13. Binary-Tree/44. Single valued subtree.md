class Solution {
    int ans = 0;
    public int singlevalued(Node root) {
        ans = 0;
        rec(root);
        return ans;
    }
    boolean rec(Node root) {
        if(root.left==null && root.right==null) {
            ans++;
            return true;
        }
        boolean flag1 = true, flag2 = true, flag = true;
        if(root.left!=null) flag1 = rec(root.left);
        if(root.right!=null) flag2 = rec(root.right);
        
        flag = flag1 && flag2;
        if(flag && root.left!=null && root.data != root.left.data) flag = false;
        if(flag && root.right!=null && root.data != root.right.data) flag = false;
        if(flag) ans++;
        
        return flag;
    }
}
