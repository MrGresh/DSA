class Pair {
    Node node;
    int hd;
    Pair(Node node, int hd) {
        this.node = node;
        this.hd = hd;
    }
}
class Solution {
    static void topAndBottomView(Node root) {
        Map<Integer, Integer> top = new TreeMap<>();
        Map<Integer, Integer> bottom = new TreeMap<>();
        Queue<Pair> q = new LinkedList<>();
        if (root != null) q.offer(new Pair(root, 0));
    
        while (!q.isEmpty()) {
            Pair p = q.poll();
            Node node = p.node;
            int hd = p.hd;
    
            if(!top.containsKey(hd)) top.put(hd, node.data);
            bottom.put(hd, node.data); // overwrite each time
    
            if (node.left != null) q.offer(new Pair(node.left, hd - 1));
            if (node.right != null) q.offer(new Pair(node.right, hd + 1));
        }
    }
}
