class Solution {
    // These need to persist across recursive calls
    Node head = null;
    Node prev = null;

    Node bTreeToCList(Node root) {
        if (root == null) return null;

        // 1. Convert to Doubly Linked List (Inorder)
        convertToDLL(root);

        // 2. Make it Circular
        // At this point, 'prev' is the last node and 'head' is the first
        if (head != null && prev != null) {
            prev.right = head;
            head.left = prev;
        }

        return head;
    }

    private void convertToDLL(Node curr) {
        if (curr == null) return;

        // Traverse Left
        convertToDLL(curr.left);

        // Process Current Node
        if (prev == null) {
            // This is the leftmost node, so it becomes the head
            head = curr;
        } else {
            // Link current node with previous node
            curr.left = prev;
            prev.right = curr;
        }
        // Move prev to current
        prev = curr;

        // Traverse Right
        convertToDLL(curr.right);
    }
}
