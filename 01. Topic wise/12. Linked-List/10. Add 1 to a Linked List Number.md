class Solution {
    public Node addOne(Node head) {
        head = reverseList(head);
        Node curr = head;
        int carry = 1;
        while(true) {
            int val = curr.data + carry;
            if(val>9) {
                val -= 10;
                carry = 1;
                curr.data = val;
            } else {
                carry=0;
                curr.data = val;
                break;
            }
            if(curr.next!=null) curr = curr.next;
            else break;
        }
        if(carry==1) curr.next = new Node(1);
        return reverseList(head);
    }
    public Node reverseList(Node head) {
        Node curr = head, nxt = null, prev = null;
        while(curr!=null) {
            nxt = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nxt;
        }
        return prev;
    }
}

class Solution {
    public Node addOne(Node head) {
        int carry = rec(head);
        if(carry==0) return head;
        else {
            Node newHead = new Node(carry);
            newHead.next = head;
            return newHead;
        }
    }
    int rec(Node head) {
        if(head==null) return 1;
        int carry = rec(head.next);
        if(carry==0) return 0;
        int val = head.data + carry;
        if(val > 9) {
            val -= 10;
            carry = 1;
        } else carry = 0;
        head.data = val;
        return carry;
    }
}
