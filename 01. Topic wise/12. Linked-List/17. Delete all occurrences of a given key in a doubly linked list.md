class Solution {
    static Node deleteAllOccurOfX(Node head, int x) {
        while(head!=null && head.data==x) head = head.next;
        if(head==null) return null;
        head.prev = null;
        Node curr = head.next;
        while(curr!=null) {
            if(curr.data==x) {
                curr.prev.next = curr.next;
                if(curr.next!=null) curr.next.prev = curr.prev;
            }
            curr = curr.next;
        }
        return head;
    }
}
