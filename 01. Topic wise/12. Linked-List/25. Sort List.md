ListNode sortList(ListNode head) {
    if(head==null || head.next==null) return head;
    PriorityQueue<Integer> pq = new PriorityQueue<>();
    while(head!=null) {
        pq.add(head.val);
        head = head.next;
    }
    ListNode newHead = new ListNode(pq.poll());
    ListNode curr = newHead;
    while(!pq.isEmpty()) {
        curr.next = new ListNode(pq.poll());
        curr = curr.next;
    }
    return newHead;
}

ListNode sortList(ListNode head) {
    if(head==null || head.next==null) return head;
    int len = 0;
    ListNode curr = head;
    while(curr!=null) {
        len++;
        curr = curr.next;
    }
    for(int pass=1;pass<=len;pass++) { // Bubble Sort: TLE
        curr = head;
        boolean flag = true;
        for(int j=1;j<=len-pass;j++) {
            if(curr.val>curr.next.val) {
                flag = false;
                int t = curr.val;
                curr.val = curr.next.val;
                curr.next.val = t;
            }
            curr = curr.next;
        }
        if(flag) break;
    } 
    return head;
}
