class Solution {
    public static ArrayList<ArrayList<Integer>> findPairsWithGivenSum(int target, Node head) {
        ArrayList<ArrayList<Integer>> res = new ArrayList<>();
        if(head.next==null) return res;
        Node tail;
        Node t = head;
        while(t.next!=null) t = t.next;
        tail = t;
        Node l = head, r = tail;
        while(l.data<r.data) {
            int val = l.data + r.data;
            if(val==target) {
                res.add(new ArrayList<>(Arrays.asList(l.data, r.data)));
                l = l.next;
                r = r.prev;
            }
            else if(val>target) r = r.prev;
            else l = l.next;
        }
        return res;
    }
}
