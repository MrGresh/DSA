class Solution {
    public static int fractional_node(Node head, int k) {
        int len = 0;
        Node curr = head;
        while(curr!=null) {
            len++;
            curr=curr.next;
        }
        int target = (len + k - 1) / k; // OR (double)len/k
        curr = head;
        for(int i=1;i<target;i++) {
            curr = curr.next;
        }
        return curr.data;
    }
}
