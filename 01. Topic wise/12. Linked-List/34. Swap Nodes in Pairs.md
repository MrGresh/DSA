class Solution {
    public ListNode swapPairs(ListNode head) {
        if(head==null || head.next==null) return head;
        ListNode t = head;
        while(t!=null && t.next!=null) {
            int temp = t.val;
            t.val = t.next.val;
            t.next.val = temp;
            t = t.next.next;
        }
        return head;
    }
}
