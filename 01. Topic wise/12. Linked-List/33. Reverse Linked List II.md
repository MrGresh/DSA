class Solution {
    public ListNode reverseBetween(ListNode head, int left, int right) {
        ListNode curr = head, nxt = null, prev = null, leftBefore = null, firstLeft = null;
        int i = 1;
        while(i++ < left) {
            leftBefore = curr;
            curr = curr.next;
        }
        firstLeft = curr;
        i--;
        while(i++ <= (right)) {
            nxt = curr.next;
            curr.next = prev;
            prev = curr;
            curr = nxt;
        }
        firstLeft.next = curr;
        if(leftBefore!= null) {
            leftBefore.next = prev;
            return head;
        } else return prev;
    }
}
