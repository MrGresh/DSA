class Solution {
    public int maxDistance(int[] nums1, int[] nums2) {
        int i=0, j=0, ans = Integer.MIN_VALUE;
        while(i<nums1.length && j<nums2.length) {
            if(nums1[i] <= nums2[j]) {
                if(i<=j) ans = Math.max(ans, j-i);
                j++;
            } else i++;
        }
        return ans == Integer.MIN_VALUE ? 0 : ans;
    }
}
