class Solution {
    public boolean canPartitionGrid(int[][] grid) {
        int m = grid.length;
        int n = grid[0].length;
        long totalSum = 0;
        long[] rowSums = new long[m];
        long[] colSums = new long[n];

        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                rowSums[i] += grid[i][j];
                colSums[j] += grid[i][j];
                totalSum += grid[i][j];
            }
        }

        return checkCuts(grid, rowSums, totalSum, true) || 
               checkCuts(grid, colSums, totalSum, false);
    }

    private boolean checkCuts(int[][] grid, long[] lineSums, long totalSum, boolean isHorizontal) {
        int m = grid.length, n = grid[0].length;
        int numLines = lineSums.length;
        long prefixSum = 0;

        // Maps to track value counts in each section
        Map<Integer, Integer> sec1Values = new HashMap<>();
        Map<Integer, Integer> sec2Values = new HashMap<>();

        // Initialize Section 2 with all values
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                int val = grid[i][j];
                sec2Values.put(val, sec2Values.getOrDefault(val, 0) + 1);
            }
        }

        for (int k = 0; k < numLines - 1; k++) {
            // Move the current row/column from Section 2 to Section 1
            if (isHorizontal) {
                for (int j = 0; j < n; j++) move(grid[k][j], sec2Values, sec1Values);
            } else {
                for (int i = 0; i < m; i++) move(grid[i][k], sec2Values, sec1Values);
            }

            prefixSum += lineSums[k];
            long s1 = prefixSum;
            long s2 = totalSum - s1;

            if (s1 == s2) return true;

            // Check if discounting one cell from S1 makes it equal to S2
            if (canDiscount(s1, s2, sec1Values, isHorizontal, k + 1, isHorizontal ? n : m, grid, true, k)) return true;
            // Check if discounting one cell from S2 makes it equal to S1
            if (canDiscount(s2, s1, sec2Values, isHorizontal, numLines - (k + 1), isHorizontal ? n : m, grid, false, k)) return true;
        }
        return false;
    }

    private void move(int val, Map<Integer, Integer> from, Map<Integer, Integer> to) {
        from.put(val, from.get(val) - 1);
        if (from.get(val) == 0) from.remove(val);
        to.put(val, to.getOrDefault(val, 0) + 1);
    }

    private boolean canDiscount(long sSource, long sTarget, Map<Integer, Integer> vals, 
                                boolean isHoriz, int lines, int lineLen, int[][] grid, boolean isSec1, int cutIdx) {
        long diff = sSource - sTarget;
        if (diff <= 0 || diff > 100000) return false;
        int targetVal = (int) diff;

        if (!vals.containsKey(targetVal)) return false;

        // 2D Section: Always connected
        if (lines > 1 && lineLen > 1) return true;

        // 1D Section: Only endpoints are allowed
        // If it's a single row, only first/last column values
        // If it's a single column, only first/last row values
        if (isHoriz) {
            if (lines == 1) { // Section is a single row
                int r = isSec1 ? cutIdx : cutIdx + 1;
                return (grid[r][0] == targetVal || grid[r][lineLen - 1] == targetVal);
            } else { // Section is 1 cell wide (lineLen == 1)
                int startR = isSec1 ? 0 : cutIdx + 1;
                int endR = isSec1 ? cutIdx : grid.length - 1;
                return (grid[startR][0] == targetVal || grid[endR][0] == targetVal);
            }
        } else {
            if (lines == 1) { // Section is a single column
                int c = isSec1 ? cutIdx : cutIdx + 1;
                return (grid[0][c] == targetVal || grid[grid.length - 1][c] == targetVal);
            } else { // Section is 1 cell high (lineLen == 1)
                int startC = isSec1 ? 0 : cutIdx + 1;
                int endC = isSec1 ? cutIdx : grid[0].length - 1;
                return (grid[0][startC] == targetVal || grid[0][endC] == targetVal);
            }
        }
    }
}
