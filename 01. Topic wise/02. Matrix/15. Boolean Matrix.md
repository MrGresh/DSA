class Solution {
    void booleanMatrix(int mat[][]) {
        int R = mat.length;
        int C = mat[0].length;
        
        // Step 1: Auxiliary arrays to keep track of rows and columns to be updated
        int[] rowMarker = new int[R];
        int[] colMarker = new int[C];

        // Step 2: Store the rows and columns that need to be changed to 1
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (mat[i][j] == 1) {
                    rowMarker[i] = 1;
                    colMarker[j] = 1;
                }
            }
        }

        // Step 3: Update the matrix based on the markers
        for (int i = 0; i < R; i++) {
            for (int j = 0; j < C; j++) {
                if (rowMarker[i] == 1 || colMarker[j] == 1) {
                    mat[i][j] = 1;
                }
            }
        }
    }
}
