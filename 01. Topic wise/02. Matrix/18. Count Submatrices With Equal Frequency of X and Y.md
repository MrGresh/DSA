class Solution {
    public int numberOfSubmatrices(char[][] grid) {
        int rows = grid.length;
        int cols = grid[0].length;
        
        // Use the same dimensions as the grid
        int[][] countX = new int[rows][cols];
        int[][] countY = new int[rows][cols];
        int result = 0;

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                // Get values from neighbors, handling boundaries with 0
                int upX = (i > 0) ? countX[i - 1][j] : 0;
                int leftX = (j > 0) ? countX[i][j - 1] : 0;
                int diagX = (i > 0 && j > 0) ? countX[i - 1][j - 1] : 0;
                
                int upY = (i > 0) ? countY[i - 1][j] : 0;
                int leftY = (j > 0) ? countY[i][j - 1] : 0;
                int diagY = (i > 0 && j > 0) ? countY[i - 1][j - 1] : 0;

                // Standard 2D Prefix Sum formula
                countX[i][j] = upX + leftX - diagX + (grid[i][j] == 'X' ? 1 : 0);
                countY[i][j] = upY + leftY - diagY + (grid[i][j] == 'Y' ? 1 : 0);

                // Check criteria
                if (countX[i][j] > 0 && countX[i][j] == countY[i][j]) result++;
            }
        }

        return result;
    }
}
