class Solution {
    public int numberOfSubmatrices(char[][] grid) {
        int rows = grid.length;
        int cols = grid[0].length;
        
        // 2D arrays to store prefix counts of 'X' and 'Y'
        int[][] countX = new int[rows + 1][cols + 1];
        int[][] countY = new int[rows + 1][cols + 1];
        int result = 0;

        for (int i = 0; i < rows; i++) {
            for (int j = 0; j < cols; j++) {
                countX[i + 1][j + 1] = countX[i][j + 1] + countX[i + 1][j] - countX[i][j] + (grid[i][j] == 'X' ? 1 : 0);
                
                countY[i + 1][j + 1] = countY[i][j + 1] + countY[i + 1][j] - countY[i][j] + (grid[i][j] == 'Y' ? 1 : 0);

                if (countX[i + 1][j + 1] > 0 && countX[i + 1][j + 1] == countY[i + 1][j + 1]) result++;
            }
        }

        return result;
    }
}
