class Solution {
    static ArrayList<Integer> diagView(int mat[][]) {
        ArrayList<Integer> res = new ArrayList<>();
        int m = mat.length, n=mat[0].length;
        for(int col=0;col<n;col++) {
            int i=0, j=col;
            while(i<m && j>=0) res.add(mat[i++][j--]);
        }
        for(int row=1;row<m;row++) {
            int i=row, j=n-1;
            while(i<m && j>=0) res.add(mat[i++][j--]);
        }
        return res;
    }
}
