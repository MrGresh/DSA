class Solution {
    // Function for finding determinant of matrix.
    static int determinantOfMatrix(int matrix[][], int n) {
        // Base Case 1: 1x1 matrix
        if (n == 1) {
            return matrix[0][0];
        }
        
        // Base Case 2: 2x2 matrix (optimization)
        if (n == 2) {
            return (matrix[0][0] * matrix[1][1]) - (matrix[0][1] * matrix[1][0]);
        }

        int det = 0;
        int sign = 1;

        for (int j = 0; j < n; j++) {
            // Get the minor submatrix by excluding row 0 and column j
            int[][] submatrix = getSubmatrix(matrix, n, j);
            
            // Recursive call
            det += sign * matrix[0][j] * determinantOfMatrix(submatrix, n - 1);
            
            // Alternate signs: +, -, +, -
            sign = -sign;
        }

        return det;
    }

    // Helper function to create a submatrix
    private static int[][] getSubmatrix(int[][] matrix, int n, int colToRemove) {
        int[][] sub = new int[n - 1][n - 1];
        int subRow = 0;
        
        // Start from row 1 (skipping row 0)
        for (int i = 1; i < n; i++) {
            int subCol = 0;
            for (int j = 0; j < n; j++) {
                if (j == colToRemove) continue;
                sub[subRow][subCol] = matrix[i][j];
                subCol++;
            }
            subRow++;
        }
        return sub;
    }
}
