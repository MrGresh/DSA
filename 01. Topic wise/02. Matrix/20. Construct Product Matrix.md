class Solution {
    public int[][] constructProductMatrix(int[][] grid) {
        int n = grid.length;
        int m = grid[0].length;
        int[][] p = new int[n][m];
        long currentProduct = 1;
        int mod = 12345;

        // Step 1: Pass forward (Prefix Product)
        // Store the product of all elements seen so far in p[i][j]
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                p[i][j] = (int) currentProduct;
                currentProduct = (currentProduct * (grid[i][j] % mod)) % mod;
            }
        }

        // Step 2: Pass backward (Suffix Product)
        // Multiply the existing prefix product by the suffix product
        currentProduct = 1;
        for (int i = n - 1; i >= 0; i--) {
            for (int j = m - 1; j >= 0; j--) {
                p[i][j] = (int) ((p[i][j] * currentProduct) % mod);
                currentProduct = (currentProduct * (grid[i][j] % mod)) % mod;
            }
        }

        return p;
    }
}
