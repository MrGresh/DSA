class Solution {
    public static int balanceSums(int[][] mat) {
        int n = mat.length;
        int[] rowSum = new int[n];
        int[] colSum = new int[n];
        
        int maxSum = 0;
        long totalCurrentSum = 0;

        // Calculate all row and column sums
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                rowSum[i] += mat[i][j];
                colSum[j] += mat[i][j];
                totalCurrentSum += mat[i][j];
            }
        }

        // Find the maximum sum among all rows and columns
        for (int i = 0; i < n; i++) {
            maxSum = Math.max(maxSum, rowSum[i]);
            maxSum = Math.max(maxSum, colSum[i]);
        }

        // The target sum for the whole matrix is n * maxSum
        // Operations = (Target Total Sum) - (Current Total Sum)
        return (int)((long)n * maxSum - totalCurrentSum);
    }
}
