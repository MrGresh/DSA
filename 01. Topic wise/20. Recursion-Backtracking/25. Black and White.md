class BlackAndWhite {
    static long count = 0;
    static int[] dx = {1, 1, 2, 2, -1, -1, -2, -2};
    static int[] dy = {2, -2, 1, -1, 2, -2, 1, -1};

    static long numOfWays(int N, int M) {
        count = 0;
        long mod = 1000000007;

        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                // Place Black Knight at (i, j)
                // Now find all valid spots for White Knight
                count = (count + countValidSpots(i, j, N, M)) % mod;
            }
        }
        return count;
    }

    private static long countValidSpots(int r, int c, int N, int M) {
        // Total possible spots for White Knight is (Total Squares - 1)
        long totalSpots = (long) N * M - 1;
        int attackingSpots = 0;

        // Check all 8 knight moves from (r, c)
        for (int k = 0; k < 8; k++) {
            int ni = r + dx[k];
            int nj = c + dy[k];

            if (ni >= 0 && ni < N && nj >= 0 && nj < M) {
                attackingSpots++;
            }
        }

        // Valid spots = Total available spots - those under attack
        return totalSpots - attackingSpots;
    }
}
