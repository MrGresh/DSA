class Solution {
    public void solveSudoku(char[][] board) {
        fillBoard(board, 0, 0);
    }
    boolean fillBoard(char[][] board, int i, int j) {
        if(i==board.length || j==board.length) return true;
        if(board[i][j]!='.') { // this cell is already filed
            return fillBoard(board, j==board.length-1 ? i+1 : i, j==board.length-1 ? 0 : j+1);
        }
        else {
            for(char n='1';n<='9';n++) {
                if(isSafe(board, i, j, n)) {
                    board[i][j] = n;
                    if(fillBoard(board, j==board.length-1 ? i+1 : i, j==board.length-1 ? 0 : j+1)) return true;
                    board[i][j] = '.';
                }
            }
        }
        return false;
    }
    boolean isSafe(char[][] board, int i, int j, char n) {
        for(int row=0;row<board.length;row++) {
            if(row!=i && board[row][j]==n) return false;
        }
        for(int col=0;col<board.length;col++) {
            if(col!=j && board[i][col]==n) return false;
        }
        for(int[] cell : getPos(i, j)) {
            if(cell[0]!=i && cell[1]!=j && board[cell[0]][cell[1]]==n) return false;
        }
        return true;
    }
    int[][] getPos(int i, int j) {
        if (i <= 2 && j <= 2) {
            return new int[][] {{0,0},{0,1},{0,2},{1,0},{1,1},{1,2},{2,0},{2,1},{2,2}};
        }
        if (i <= 2 && j >= 3 && j <= 5) {
            return new int[][] {{0,3},{0,4},{0,5},{1,3},{1,4},{1,5},{2,3},{2,4},{2,5}};
        }
        if (i <= 2 && j >= 6) {
            return new int[][] {{0,6},{0,7},{0,8},{1,6},{1,7},{1,8},{2,6},{2,7},{2,8}};
        }
        if (i >= 3 && i <= 5 && j <= 2) {
            return new int[][] {{3,0},{3,1},{3,2},{4,0},{4,1},{4,2},{5,0},{5,1},{5,2}};
        }
        if (i >= 3 && i <= 5 && j >= 3 && j <= 5) {
            return new int[][] {{3,3},{3,4},{3,5},{4,3},{4,4},{4,5},{5,3},{5,4},{5,5}};
        }
        if (i >= 3 && i <= 5 && j >= 6) {
            return new int[][] {{3,6},{3,7},{3,8},{4,6},{4,7},{4,8},{5,6},{5,7},{5,8}};
        }
        if (i >= 6 && j <= 2) {
            return new int[][] {{6,0},{6,1},{6,2},{7,0},{7,1},{7,2},{8,0},{8,1},{8,2}};
        }
        if (i >= 6 && j >= 3 && j <= 5) {
            return new int[][] {{6,3},{6,4},{6,5},{7,3},{7,4},{7,5},{8,3},{8,4},{8,5}};
        }
        if (i >= 6 && j >= 6) {
            return new int[][] {{6,6},{6,7},{6,8},{7,6},{7,7},{7,8},{8,6},{8,7},{8,8}};
        }
        return new int[0][0];
    }
}
