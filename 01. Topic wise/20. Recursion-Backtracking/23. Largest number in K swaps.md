class Solution {
    static String max;

    public String findMaximumNum(String s, int k) {
        max = s;
        solve(s.toCharArray(), k, 0);
        return max;
    }

    private static void solve(char[] str, int k, int idx) {
        if (k == 0 || idx == str.length) {
            return;
        }

        char maxChar = str[idx];
        // Find the maximum character in the remaining string
        for (int i = idx + 1; i < str.length; i++) {
            if (str[i] > maxChar) {
                maxChar = str[i];
            }
        }

        // If the current character is not the maximum, we MUST swap
        if (maxChar != str[idx]) {
            k--;
        }

        // Try swapping the current index with every occurrence of maxChar
        for (int i = str.length - 1; i >= idx; i--) {
            if (str[i] == maxChar) {
                swap(str, idx, i);
                
                String current = new String(str);
                if (current.compareTo(max) > 0) {
                    max = current;
                }

                // Recurse for the next index
                solve(str, k, idx + 1);

                // Backtrack
                swap(str, idx, i);
            }
        }
    }

    private static void swap(char[] str, int i, int j) {
        char temp = str[i];
        str[i] = str[j];
        str[j] = temp;
    }
}
