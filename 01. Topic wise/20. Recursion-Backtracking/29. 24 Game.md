**Logic:** Select 2 values each time
```java
class Solution {
    public boolean judgePoint24(int[] cards) {
        double[] nums = new double[4];
        for (int i = 0; i < 4; i++) nums[i] = cards[i];
        return helper(nums);
    }
    
    private boolean helper(double[] nums) {
        if (nums.length == 1) return Math.abs(nums[0] - 24) < 1e-6;
        
        for (int i = 0; i < nums.length; i++) {
            for (int j = 0; j < nums.length; j++) {
                if (i == j) continue;
                
                double[] next = new double[nums.length - 1];
                int idx = 0;
                for (int k = 0; k < nums.length; k++) {
                    if (k != i && k != j) next[idx++] = nums[k];
                }
                
                for (int op = 0; op < 4; op++) {
                    if (op == 0) next[idx] = nums[i] + nums[j];
                    if (op == 1) next[idx] = nums[i] - nums[j];
                    if (op == 2) next[idx] = nums[i] * nums[j];
                    if (op == 3 && nums[j] != 0) next[idx] = nums[i] / nums[j];
                    if (helper(next)) return true;
                }
            }
        }
        return false;
    }
}
```
