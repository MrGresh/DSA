class Solution {
    List<String> res;
    public List<String> generateParenthesis(int n) {
        res = new ArrayList<>();
        helper(new StringBuilder(), n, n);
        return res;
    }
    void helper(StringBuilder curr, int openLeft, int closeLeft) {
        if(openLeft==0 && closeLeft==0) {
            res.add(curr.toString());
        }
        if(openLeft>0) {
            curr.append("(");
            helper(curr, openLeft-1, closeLeft);
            curr.deleteCharAt(curr.length() - 1);
        }
        if(closeLeft>0 && closeLeft>openLeft) {
            curr.append(")");
            helper(curr, openLeft, closeLeft-1);
            curr.deleteCharAt(curr.length() - 1);
        }
    }
}
