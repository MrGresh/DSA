### Memoization
```java
class Solution {
    Integer[] memo;
    public int combinationSum4(int[] nums, int target) {
        int count = 0;
        memo = new Integer[target+1];
        for(int x : nums) {
            if(x>target) continue;
            count += rec(nums, target-x);
        }
        return count;
    }
    int rec(int[] nums, int target) {
        if(target==0) return 1;
        if(memo[target]!=null) return memo[target];
        int val = 0;
        for(int x : nums) {
            if(x>target) continue;
            val += rec(nums, target-x);
        }
        return memo[target] = val;
    }
}
```

### Tabulation
```java
class Solution {
    public int combinationSum4(int[] nums, int T) {
        int[] dp = new int[T+1];
        for(int target=0;target<=T;target++) {
            if(target==0) dp[target] = 1;
            else {
                int val = 0;
                for(int x : nums) {
                    if(x>target) continue;
                    val += dp[target-x];
                }
                dp[target] = val;
            }
        }
        return dp[T];
    }
}
```
