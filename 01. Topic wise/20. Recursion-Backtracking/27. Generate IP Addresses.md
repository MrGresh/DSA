class Solution {
    ArrayList<String> res;
    public ArrayList<String> generateIp(String s) {
        if(s.length()>12 || s.length()<4) return new ArrayList<String>();
        res = new ArrayList<String>();
        rec(s, 0, 0, new ArrayList<String>());
        return res;
    }
    void rec(String s, int idx, int n, ArrayList<String> curr) {
        if(n==4) {
            if(idx==s.length()) {
                res.add(String.join(".", curr));
                return;
            } else return;
        }
        for(int i=1;i<=3;i++) {
            if(i>1 && s.charAt(idx)=='0') break;
            if(idx+i>s.length()) break;
            String str = s.substring(idx, idx+i);
            if(Integer.parseInt(str)>255) continue;
            curr.add(str);
            rec(s, idx+i, n+1, curr);
            curr.remove(curr.size()-1);
        }
    }
}
