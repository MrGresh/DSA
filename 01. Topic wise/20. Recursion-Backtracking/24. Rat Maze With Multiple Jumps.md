class Solution {
    public int[][] ShortestDistance(int[][] matrix) {
        int n = matrix.length;
        int[][] res = new int[n][n];
        
        if (solve(0, 0, matrix, res, n)) {
            return res;
        }
        
        // If no path found, return a matrix containing -1
        return new int[][]{{-1}};
    }

    private boolean solve(int r, int c, int[][] mat, int[][] res, int n) {
        // Base Case: Reached Destination
        if (r == n - 1 && c == n - 1) {
            res[r][c] = 1;
            return true;
        }

        // Check if current cell is valid (non-zero jump allowed)
        if (r >= n || c >= n || mat[r][c] == 0) {
            return false;
        }

        res[r][c] = 1; // Mark current cell in path
        int maxJumps = mat[r][c];

        // Preference 1: Forward (Right)
        // We try jumping 1 step, then 2, up to maxJumps
        for (int i = 1; i <= maxJumps; i++) {
            if (c + i < n && solve(r, c + i, mat, res, n)) {
                return true;
            }
            
            // Preference 2: Downward
            // Only check downward for the same jump distance after checking right
            if (r + i < n && solve(r + i, c, mat, res, n)) {
                return true;
            }
        }

        // Backtrack: If no move leads to a solution
        res[r][c] = 0;
        return false;
    }
}
