class Solution {
    int count = 0;
    String result = "";

    public String getHappyString(int n, int k) {
        backtrack(n, k, new StringBuilder());
        return result;
    }

    private boolean backtrack(int n, int k, StringBuilder current) {
        if (current.length() == n) {
            count++;
            if (count == k) {
                result = current.toString();
                return true;
            }
            return false;
        }
        int len = current.length();
        for (char c : new char[]{'a', 'b', 'c'}) {
            if (len ==0 || (len > 0 && current.charAt(len - 1) != c)) {
                current.append(c);
                if(backtrack(n, k, current)) return true;
                current.deleteCharAt(current.length() - 1);
            }
        }
        return false;
    }
}
