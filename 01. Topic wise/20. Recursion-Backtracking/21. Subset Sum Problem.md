### Memoization
```java
class Solution {
    static Boolean[][] memo;
    static Boolean isSubsetSum(int arr[], int sum) {
        int n = arr.length;
        memo = new Boolean[n + 1][sum + 1];
        return rec(arr, 0, sum);
    }
    static boolean rec(int arr[], int i, int sum) {
        if (sum == 0) return true;
        if (i == arr.length) return false;
        if (memo[i][sum] != null) return memo[i][sum];
        return memo[i][sum] = (sum-arr[i]>=0 ? rec(arr, i+1, sum-arr[i]) : false) || rec(arr, i+1, sum);
    }
}
```

### Tabulation
```java
class Solution {
    static Boolean isSubsetSum(int arr[], int S) {
        int n = arr.length;
        boolean[][] dp = new boolean[n + 1][S + 1];
        for(int i=arr.length;i>=0;i--) {
            for(int sum=0;sum<=S;sum++) {
                if (sum == 0) dp[i][sum] = true;
                else if (i == arr.length) dp[i][sum] = false;
                else {
                    dp[i][sum] = (sum-arr[i]>=0 ? dp[i+1][sum-arr[i]] : false) || dp[i+1][sum];
                }
            }
        }
        return dp[0][S];
    }
}
```
