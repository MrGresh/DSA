class Solution {
    List<List<Integer>> adj;
    int[] color;
    boolean graphColoring(int v, int[][] edges, int m) {
        adj = new ArrayList<>();
        for(int i=0;i<v;i++) adj.add(new ArrayList<>());
        for(int[] e: edges) {
            adj.get(e[0]).add(e[1]);
            adj.get(e[1]).add(e[0]);
        }
        color = new int[v];
        Arrays.fill(color, 0);
        return solve(0, v, m);
    }
    boolean solve(int node, int v, int m) {
        if (node == v) return true;
        for (int c = 1; c <= m; c++) {
            if (isSafe(node, c)) {
                color[node] = c;
                if (solve(node + 1, v, m)) return true;
                color[node] = 0;
            }
        }
        return false;
    }
    boolean isSafe(int node, int c) {
        for (int nei : adj.get(node)) {
            if (color[nei] == c) return false;
        }
        return true;
    }
}
