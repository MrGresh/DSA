class Solution {
    List<List<Integer>> res;
    public List<List<Integer>> combine(int n, int k) {
        res = new ArrayList<>();
        List<Integer> curr = new ArrayList<>();
        for(int i=1;i<=n;i++) {
            curr.add(i);
            rec(curr, k, n, i+1);
            curr.removeLast();
        }
        return res;
    }
    void rec(List<Integer> curr, int k, int n, int idx) {
        if(curr.size()==k) {
            res.add(new ArrayList<>(curr));
            return;
        }
        for(int i=idx;i<=n;i++) {
            curr.add(i);
            rec(curr, k, n, i+1);
            curr.removeLast();
        }
    }
}
