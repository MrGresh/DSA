class Solution {
    public long sumXOR(int[] arr) {
        int n = arr.length;
        long totalSum = 0;

        for (int i = 0; i < 31; i++) {
            long countOn = 0;
            
            for (int j = 0; j < n; j++) {
                if ((arr[j] & (1 << i)) != 0) {
                    countOn++;
                }
            }

            long countOff = n - countOn;

            totalSum += (countOn * countOff) * (1L << i);
        }

        return totalSum;
    }
}
