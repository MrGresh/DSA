class Solution {
    static long isPallindrome(long N) {
        int pos = -1, cnt=0;
        long t = N;
        while(t>0) {
            if((t&1)==1) {
                pos = cnt;
            }
            t>>=1;
            cnt++;
        }
        int l = pos, r = 0;
        while(l>r) {
            long lBit = (N>>l)&1;
            long rBit = (N>>r)&1;
            if(lBit!=rBit) return 0; // false
            l--;
            r++;
        }
        return 1; // true
    }
}

class Solution {
    static long isPallindrome(long N) {
        String S = Long.toBinaryString(N);
        
        int l = 0;
        int r = S.length() - 1;
        
        while (l < r) {
            if (S.charAt(l) != S.charAt(r)) {
                return 0;
            }
            l++;
            r--;
        }
        
        return 1;
    }
}
