```java
class Solution {
    public static int max_toys(int arr[], int k) {
        Arrays.sort(arr);
        int cnt = 0;
        for(int x: arr) {
            if(x<=k) {
                cnt++;
                k -= x;
            } else break;
        }
        return cnt;
    }
}
```

```java
class Solution {
    public static int max_toys(int arr[], int k) {
        PriorityQueue<Integer> pq = new PriorityQueue<>();
        for(int x: arr) pq.add(x);
        int cnt = 0;
        while(!pq.isEmpty()) {
            int x = pq.poll();
            if(x<=k) {
                cnt++;
                k -= x;
            } else break;
        }
        return cnt;
    }
}
```
