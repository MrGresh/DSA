class Solution {
    public int mostBooked(int n, int[][] meetings) {
        Arrays.sort(meetings, (a, b) -> Integer.compare(a[0], b[0]));

        long[] roomUsageCount = new long[n];
        
        PriorityQueue<Integer> availableRooms = new PriorityQueue<>();
        for (int i = 0; i < n; i++) availableRooms.add(i);

        PriorityQueue<long[]> ongoingMeetings = new PriorityQueue<>((a, b) -> {
            if (a[0] != b[0]) return Long.compare(a[0], b[0]);
            return Long.compare(a[1], b[1]);
        });

        for (int[] meeting : meetings) {
            long start = meeting[0];
            long end = meeting[1];

            while (!ongoingMeetings.isEmpty() && ongoingMeetings.peek()[0] <= start) {
                availableRooms.add((int) ongoingMeetings.poll()[1]);
            }

            if (!availableRooms.isEmpty()) {
                int room = availableRooms.poll();
                roomUsageCount[room]++;
                ongoingMeetings.add(new long[]{end, room});
            } else {
                long[] earliestEnding = ongoingMeetings.poll();
                long roomFreeTime = earliestEnding[0];
                int room = (int) earliestEnding[1];
                
                roomUsageCount[room]++;
                ongoingMeetings.add(new long[]{roomFreeTime + (end - start), room});
            }
        }

        int resultRoom = 0;
        for (int i = 1; i < n; i++) {
            if (roomUsageCount[i] > roomUsageCount[resultRoom]) {
                resultRoom = i;
            }
        }

        return resultRoom;
    }
}
