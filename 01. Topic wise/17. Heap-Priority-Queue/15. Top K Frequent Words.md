class Solution {
    public List<String> topKFrequent(String[] words, int k) {
        // 1. Count frequencies
        Map<String, Integer> counts = new HashMap<>();
        for (String word : words) {
            counts.put(word, counts.getOrDefault(word, 0) + 1);
        }

        // 2. Min-Heap with custom comparator
        // Rule: Higher frequency is better. 
        // If frequency is same, smaller lexicographical string is better.
        // We use a Min-Heap to keep the "best" k elements.
        PriorityQueue<String> minHeap = new PriorityQueue<>((a, b) -> {
            int freqA = counts.get(a);
            int freqB = counts.get(b);
            if (freqA != freqB) {
                return freqA - freqB; // Lower frequency at top
            }
            return b.compareTo(a); // Larger lexicographical string at top for ties
        });

        // 3. Maintain size k in the heap
        for (String word : counts.keySet()) {
            minHeap.add(word);
            if (minHeap.size() > k) {
                minHeap.poll();
            }
        }

        // 4. Build result list and reverse
        List<String> result = new LinkedList<>();
        while (!minHeap.isEmpty()) {
            result.addFirst(minHeap.poll());
        }
        
        return result;
    }
}
