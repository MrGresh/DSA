class Solution {
    public static String rearrangeString(String s) {
        int n = s.length();
        int[] counts = new int[26];
        for (char c : s.toCharArray()) {
            counts[c - 'a']++;
        }

        // Max-Heap: stores characters sorted by their frequency descending
        PriorityQueue<Character> maxHeap = new PriorityQueue<>((a, b) -> counts[b - 'a'] - counts[a - 'a']);

        for (int i = 0; i < 26; i++) {
            if (counts[i] > 0) {
                // Impossible case check
                if (counts[i] > (n + 1) / 2) return "";
                maxHeap.add((char) (i + 'a'));
            }
        }

        StringBuilder res = new StringBuilder();
        Character prev = null;

        while (!maxHeap.isEmpty()) {
            char curr = maxHeap.poll();
            res.append(curr);
            counts[curr - 'a']--;

            // If a previous character still has frequency, add it back to heap 
            // now that we've put a different character in between.
            if (prev != null && counts[prev - 'a'] > 0) {
                maxHeap.add(prev);
            }

            // Update prev to the character we just used
            prev = curr;
        }

        return res.length() == n ? res.toString() : "";
    }
}
