class SubsetDifference {
    public static long maxDifference(int[] arr, int k) {
        int n = arr.length;
        if (2 * k > n) {
            throw new IllegalArgumentException("Subsets must be disjoint; 2k cannot exceed N.");
        }

        // Sort the array to easily pick extremes
        Arrays.sort(arr);

        long minSum = 0;
        long maxSum = 0;

        // Sum the first k elements
        for (int i = 0; i < k; i++) {
            minSum += arr[i];
        }

        // Sum the last k elements
        for (int i = n - k; i < n; i++) {
            maxSum += arr[i];
        }

        return maxSum - minSum;
    }
}
