class Element {
    int row, col;
    Element(int row, int col) {
        this.row = row;
        this.col = col;
    }
}

class Solution {
    public ArrayList<Integer> mergeArrays(int[][] mat) {
        ArrayList<Integer> result = new ArrayList<>();
        int k = mat.length;
        
        // Min-Heap based on the value of the elements
        PriorityQueue<Element> minHeap = new PriorityQueue<>((a, b) -> mat[a.row][a.col] - mat[b.row][b.col]);
        
        // 1. Put the first element of each of the k arrays into the heap
        for (int i = 0; i < k; i++) {
            if (mat[i].length > 0) {
                minHeap.add(new Element(i, 0));
            }
        }
        
        // 2. Process the heap
        while (!minHeap.isEmpty()) {
            Element curr = minHeap.poll();
            result.add(mat[curr.row][curr.col]);
            
            // 3. If there's a next element in the same row, add it to the heap
            if (curr.col + 1 < mat[curr.row].length) {
                minHeap.add(new Element(curr.row, curr.col + 1));
            }
        }
        
        return result;
    }
}
