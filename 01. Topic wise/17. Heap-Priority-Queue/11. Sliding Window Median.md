class Solution {
    public double[] medianSlidingWindow(int[] nums, int k) {
        double[] result = new double[nums.length - k + 1];
        // max heap for the left half
        PriorityQueue<Integer> small = new PriorityQueue<>(Collections.reverseOrder());
        // min heap for the right half
        PriorityQueue<Integer> large = new PriorityQueue<>();
        // map to track elements to be lazily removed
        Map<Integer, Integer> delayed = new HashMap<>();
        
        // Initialize the first window
        for (int i = 0; i < k; i++) {
            small.add(nums[i]);
        }
        for (int i = 0; i < k / 2; i++) {
            large.add(small.poll());
        }
        
        result[0] = getMedian(small, large, k);
        
        for (int i = k; i < nums.length; i++) {
            int outNum = nums[i - k];
            int inNum = nums[i];
            int balance = 0; // relative size change of small heap
            
            // 1. Mark the outgoing number for removal
            balance += (outNum <= small.peek() ? -1 : 1);
            delayed.put(outNum, delayed.getOrDefault(outNum, 0) + 1);
            
            // 2. Add the incoming number
            if (!small.isEmpty() && inNum <= small.peek()) {
                small.add(inNum);
                balance++;
            } else {
                large.add(inNum);
                balance--;
            }
            
            // 3. Rebalance the heaps
            if (balance < 0) { // large has too many
                small.add(large.poll());
            } else if (balance > 0) { // small has too many
                large.add(small.poll());
            }
            
            // 4. Remove "garbage" elements from tops of heaps
            prune(small, delayed);
            prune(large, delayed);
            
            result[i - k + 1] = getMedian(small, large, k);
        }
        
        return result;
    }
    
    private void prune(PriorityQueue<Integer> heap, Map<Integer, Integer> delayed) {
        while (!heap.isEmpty() && delayed.getOrDefault(heap.peek(), 0) > 0) {
            int top = heap.poll();
            delayed.put(top, delayed.get(top) - 1);
        }
    }
    
    private double getMedian(PriorityQueue<Integer> small, PriorityQueue<Integer> large, int k) {
        if (k % 2 == 1) {
            return (double) small.peek();
        } else {
            // Use long to prevent overflow before dividing
            return ((double) small.peek() + (double) large.peek()) / 2.0;
        }
    }
}
