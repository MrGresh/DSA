```java
public int largestCoPrimeDivisor(int x, int y) {
    // Keep removing common factors between x and y from x
    while (gcd(x, y) != 1) {
        x = x / gcd(x, y);
    }
    return x;
}
```

```java
private int gcd(int a, int b) {
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}
```
