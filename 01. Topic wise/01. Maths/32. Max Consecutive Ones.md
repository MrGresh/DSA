class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int ans = 0;
        int count = 0;
        for(int x : nums) {
            if(x!=1) {
                ans = Math.max(ans, count);
                count = 0;
            } else count++;
        }
        ans = Math.max(ans, count);
        return ans;
    }
}
