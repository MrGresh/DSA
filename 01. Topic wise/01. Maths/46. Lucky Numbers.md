class Solution {
    public static boolean isLucky(int n) {
        // We start by removing every 2nd number
        int counter = 2;
        
        while (counter <= n) {
            // If the current position is a multiple of the counter, 
            // the number at this position is eliminated.
            if (n % counter == 0) {
                return false;
            }
            
            // Calculate the new position of n after the current removal step.
            // (n / counter) represents how many numbers were removed before or at n.
            n = n - (n / counter);
            
            // Move to the next step (remove every 3rd, then 4th, etc.)
            counter++;
        }
        
        return true;
    }
}
