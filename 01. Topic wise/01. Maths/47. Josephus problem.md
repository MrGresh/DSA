class Solution {
    public int josephus(int n, int k) {
        int pos = -1;
        boolean[] eliminated = new boolean[n];
        int surviving = n;
        pos += k;
        pos %= n;
        while(surviving>1) {
            eliminated[pos] = true;
            surviving--;
            int count = 0;
            while(count<k) {
                pos++;
                pos %= n;
                if(!eliminated[pos]) count++;
            }
        }
        return pos+1;
    }
}

class Solution {
    public int josephus(int n, int k) {
        int result = 0;
        
        for (int i = 2; i <= n; i++) {
            result = (result + k) % i;
        }
        
        return result + 1;
    }
}
