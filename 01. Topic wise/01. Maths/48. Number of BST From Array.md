class Solution {
    public ArrayList<Integer> countBSTs(int[] arr) {
        int n = arr.length;
        // Precomputed Catalan numbers: C0, C1, C2, C3, C4, C5, C6
        int[] catalan = getCatalan(n);
        
        ArrayList<Integer> result = new ArrayList<>();
        
        for (int i = 0; i < n; i++) {
            int leftCount = 0;
            int rightCount = 0;
            
            // Count elements smaller and larger than the current root arr[i]
            for (int j = 0; j < n; j++) {
                if (i == j) continue;
                if (arr[j] < arr[i]) {
                    leftCount++;
                } else {
                    rightCount++;
                }
            }
            
            // Total BSTs = (Ways to form Left) * (Ways to form Right)
            result.add(catalan[leftCount] * catalan[rightCount]);
        }
        
        return result;
    }
    public int[] getCatalan(int n) {
        int[] dp = new int[n + 1];
        dp[0] = 1;
        dp[1] = 1;
    
        for (int i = 2; i <= n; i++) {
            for (int j = 0; j < i; j++) {
                // Formula: Ci = C0*Ci-1 + C1*Ci-2 + ... + Ci-1*C0
                dp[i] += dp[j] * dp[i - 1 - j];
            }
        }
        return dp;
    }
}
