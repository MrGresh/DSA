class Solution {
    public int maximum69Number (int n) {
		int digits = 0, targetPos = 0;
        int n1 = n;
        while(n>0) {
            digits++;
            if(n%10==6) targetPos = digits;
            n/=10;
        }
        n = n1;
        int res = 0;
        for(int i=1;i<=digits;i++) {
            int digit = (i==targetPos) ? 9 : n%10;
            res = ((int)Math.pow(10, i-1))*digit + res;
            n /= 10;
        }
        return res;        
    }
}
