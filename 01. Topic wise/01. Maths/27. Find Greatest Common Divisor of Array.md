class Solution {
    public int findGCD(int[] arr) {
        int smallest = arr[0], largest = arr[0];
        for(int x : arr) {
            if(x>largest) largest = x;
            if(x<smallest) smallest = x;
        }
        return calcGcd(smallest, largest);        
    }
    int calcGcd(int a, int b) {
        if (b == 0) return a;
        return calcGcd(b, a % b);
    }
}
