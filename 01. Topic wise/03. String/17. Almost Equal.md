class Solution {
    public int countChars(String s1, String s2) {
        // Since characters are lowercase 'a'-'z', use a size 26 array
        int[] freq1 = new int[26];
        int[] freq2 = new int[26];

        // Fill frequency for first string
        for (char c : s1.toCharArray()) {
            freq1[c - 'a']++;
        }

        // Fill frequency for second string
        for (char c : s2.toCharArray()) {
            freq2[c - 'a']++;
        }

        int deletions = 0;
        // The number of deletions needed is the sum of 
        // absolute differences for each character
        for (int i = 0; i < 26; i++) {
            deletions += Math.abs(freq1[i] - freq2[i]);
        }

        return deletions;
    }
}
