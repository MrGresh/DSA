class Solution {
    Map<String, String> dp;
    public boolean isSubsequence(String w1, String w2) {
        dp = new HashMap<>();
        String lcs = rec(w1, w2, 0, 0);
        return w1.equals(lcs);
    }
    String rec(String w1, String w2, int i, int j) {
        if(i==w1.length() || j==w2.length()) return "";
        String key = i+","+j;
        if(dp.containsKey(key)) return dp.get(key);
        String s;
        if(w1.charAt(i)==w2.charAt(j)) s = w1.charAt(i) + rec(w1, w2, i+1, j+1);
        else {
            String t1 = rec(w1, w2, i+1, j);
            String t2 = rec(w1, w2, i, j+1);
            if(t1.length() > t2.length()) s = t1;
            else s = t2;
        }
        dp.put(key, s);
        return s;
    }
}
