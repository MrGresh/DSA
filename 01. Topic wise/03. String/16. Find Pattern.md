class Solution {
    public static int findPattern(String s, String p) {
        for(int i=0;i<=s.length()-p.length();i++) {
            if(s.charAt(i)==p.charAt(0)) {
                int j = 1, k=i+1;
                while(j<p.length() && s.charAt(k)==p.charAt(j)) {
                    k++;
                    j++;
                }
                if(j==p.length()) return i;
            }
        }
        return -1;
    }
}
