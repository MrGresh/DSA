class Solution {
    public static String caseSort(String s) {
        int n = s.length();
        StringBuilder upper = new StringBuilder();
        StringBuilder lower = new StringBuilder();

        // Step 1: Separate characters into two groups
        for (int i = 0; i < n; i++) {
            char c = s.charAt(i);
            if (Character.isUpperCase(c)) {
                upper.append(c);
            } else {
                lower.append(c);
            }
        }

        // Step 2: Sort both groups
        char[] upperArr = upper.toString().toCharArray();
        char[] lowerArr = lower.toString().toCharArray();
        Arrays.sort(upperArr);
        Arrays.sort(lowerArr);

        // Step 3: Reconstruct the string
        StringBuilder result = new StringBuilder();
        int uIdx = 0, lIdx = 0;
        
        for (int i = 0; i < n; i++) {
            char original = s.charAt(i);
            if (Character.isUpperCase(original)) {
                result.append(upperArr[uIdx++]);
            } else {
                result.append(lowerArr[lIdx++]);
            }
        }

        return result.toString();
    }
}
