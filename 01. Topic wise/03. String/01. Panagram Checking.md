class Solution {
    public static boolean checkPangram(String s) {
        if (s.length() < 26) {
            return false;
        }

        boolean[] mark = new boolean[26];
        int index = 0;

        for (int i = 0; i < s.length(); i++) {
            char ch = s.charAt(i);

            if ('A' <= ch && ch <= 'Z') {
                index = ch - 'A';
            } else if ('a' <= ch && ch <= 'z') {
                index = ch - 'a';
            } else {
                continue;
            }
            
            mark[index] = true;
        }

        for (boolean present : mark) {
            if (!present) {
                return false;
            }
        }

        return true;
    }
}
