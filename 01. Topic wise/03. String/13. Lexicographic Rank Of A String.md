**Note:** S consisting of distinct lowercase characters
```java
class Solution {
    public long findRank(String s) {
        int n = s.length();

        long rank = 1;
        long totalFactorial = factorial(n);

        for (int i = 0; i < n; i++) {
            totalFactorial /= (n - i);
            
            int countSmaller = 0;
            for (int j = i + 1; j < n; j++) {
                if (s.charAt(j) < s.charAt(i)) {
                    countSmaller++;
                }
            }
            
            rank += (long) countSmaller * totalFactorial;
        }

        return rank;
    }

    private long factorial(int n) {
        long res = 1;
        for (int i = 2; i <= n; i++) res *= i;
        return res;
    }
}
```
