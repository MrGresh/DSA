class Solution {
    public long findRank(String s) {
        int n = s.length();
        
        // Step 1: Check for repeated characters
        int[] freq = new int[26];
        for (char c : s.toCharArray()) {
            if (++freq[c - 'a'] > 1) return 0;
        }

        long rank = 1;
        long totalFactorial = factorial(n);

        for (int i = 0; i < n; i++) {
            // How many permutations are there for the remaining characters?
            totalFactorial /= (n - i);
            
            // Count how many characters smaller than s.charAt(i) 
            // are still available to the right
            int countSmaller = 0;
            for (int j = i + 1; j < n; j++) {
                if (s.charAt(j) < s.charAt(i)) {
                    countSmaller++;
                }
            }
            
            rank += (long) countSmaller * totalFactorial;
        }

        return (int) rank;
    }

    private long factorial(int n) {
        long res = 1;
        for (int i = 2; i <= n; i++) res *= i;
        return res;
    }
}
