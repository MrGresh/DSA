class Solution {
    public boolean isCircularSentence(String s) {
        if(s.charAt(0)!=s.charAt(s.length()-1)) return false;
        int i = 0;
        while(i<s.length()) {
            if(s.charAt(i)==' ' && (s.charAt(i-1)!=s.charAt(i+1))) return false;
            i++;
        }
        return true;
    }
}
