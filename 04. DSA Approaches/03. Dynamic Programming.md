# Dice throw

## Memoization
```java
class Solution {
    static int[][] dp;
    static int noOfWays(int m, int n, int x) {
        if(m*n<x) return 0;
        dp = new int[n+1][x+1];
        for(int[] row: dp) Arrays.fill(row, -1);
        return rec(0, x, m, n);
    }
    static int rec(int completedDice, int x, int m, int n) {
        if(completedDice==n) return x==0 ? 1 : 0;
        if(x<=0) return 0;
        if(dp[completedDice][x] != -1) return dp[completedDice][x];
        int ans = 0;
        for(int i=1;i<=m;i++) {
            ans += rec(completedDice+1, x-i, m, n);
        }
        return dp[completedDice][x] = ans;
    }
};
```

## Tabulation: Mirror of Memoization
```java
class Solution {
    static long noOfWays(int m, int n, int x) {
        if ((long)m * n < x || x < n) return 0;
        long[][] dp = new long[n + 1][x + 1];
        dp[n][0] = 1;
        for (int i = n - 1; i >= 0; i--) {
            for (int j = 0; j <= x; j++) {
                for (int k = 1; k <= m; k++) {
                    if (j - k >= 0) dp[i][j] += dp[i + 1][j - k];
                }
            }
        }
        return dp[0][x];
    }
}
```
