## Find All Possible Stable Binary Arrays I

### Memoization
```java
class Solution {
    private Long[][][][] memo;
    private int limit;
    private final int MOD = 1_000_000_007;

    public int numberOfStableArrays(int zero, int one, int limit) {
        this.limit = limit;
        this.memo = new Long[zero + 1][one + 1][2][limit + 1];
        
        long count0 = solve(zero - 1, one, 0, 1);
        long count1 = solve(zero, one - 1, 1, 1);

        return (int) ((count0 + count1) % MOD);
    }

    private long solve(int z, int o, int last, int streak) {
        if (z == 0 && o == 0) return 1;
        if (memo[z][o][last][streak] != null) return memo[z][o][last][streak];
        long res = 0;
        if (z > 0) {
            if (last == 0) {
                if(streak < limit) res = (res + solve(z - 1, o, 0, streak + 1)) % MOD;
            }
            else res = (res + solve(z - 1, o, 0, 1)) % MOD;
        }
        if (o > 0) {
            if (last == 1) {
                if(streak < limit) res = (res + solve(z, o - 1, 1, streak + 1)) % MOD;
            }
            else res = (res + solve(z, o - 1, 1, 1)) % MOD;
        }
        return memo[z][o][last][streak] = res;
    }
}
```

### Tabulation
```java
class Solution {
    private final int MOD = 1_000_000_007;

    public int numberOfStableArrays(int zero, int one, int limit) {
        long[][][][] dp = new long[zero + 1][one + 1][2][limit + 1];

        // Starting states: place first zero or first one
        if (zero > 0) dp[1][0][0][1] = 1;
        if (one > 0) dp[0][1][1][1] = 1;

        for (int z = 0; z <= zero; z++) {
            for (int o = 0; o <= one; o++) {
                for (int last = 0; last < 2; last++) {
                    for (int streak = 1; streak <= limit; streak++) {
                        long val = dp[z][o][last][streak];
                        if (val == 0) continue;

                        // Place a zero
                        if (z < zero) {
                            if (last == 0) {
                                if (streak < limit) {
                                    dp[z + 1][o][0][streak + 1] =
                                        (dp[z + 1][o][0][streak + 1] + val) % MOD;
                                }
                            } else {
                                dp[z + 1][o][0][1] =
                                    (dp[z + 1][o][0][1] + val) % MOD;
                            }
                        }

                        // Place a one
                        if (o < one) {
                            if (last == 1) {
                                if (streak < limit) {
                                    dp[z][o + 1][1][streak + 1] =
                                        (dp[z][o + 1][1][streak + 1] + val) % MOD;
                                }
                            } else {
                                dp[z][o + 1][1][1] =
                                    (dp[z][o + 1][1][1] + val) % MOD;
                            }
                        }
                    }
                }
            }
        }

        // Collect all ways that used all zeros and ones
        long ans = 0;
        for (int last = 0; last < 2; last++) {
            for (int streak = 1; streak <= limit; streak++) {
                ans = (ans + dp[zero][one][last][streak]) % MOD;
            }
        }
        return (int) ans;
    }
}
```

### Space Optimized Tabulation
```java
class Solution {
    private final int MOD = 1_000_000_007;

    public int numberOfStableArrays(int zero, int one, int limit) {
        // dp[o][last][streak] for current z
        long[][][] dpCurr = new long[one + 1][2][limit + 1];
        long[][][] dpNext = new long[one + 1][2][limit + 1];

        // Start states: first placed element
        if (zero > 0) dpNext[0][0][1] = 1;  // placed one zero
        if (one > 0) dpCurr[1][1][1] = 1;   // placed one one

        for (int z = 0; z <= zero; z++) {
            for (int o = 0; o <= one; o++) {
                for (int last = 0; last < 2; last++) {
                    for (int streak = 1; streak <= limit; streak++) {
                        long val = dpCurr[o][last][streak];
                        if (val == 0) continue;

                        // Place a zero
                        if (z < zero) {
                            if (last == 0 && streak < limit) {
                                dpNext[o][0][streak + 1] =
                                    (dpNext[o][0][streak + 1] + val) % MOD;
                            } else if (last == 1) {
                                dpNext[o][0][1] =
                                    (dpNext[o][0][1] + val) % MOD;
                            }
                        }

                        // Place a one
                        if (o < one) {
                            if (last == 1 && streak < limit) {
                                dpCurr[o + 1][1][streak + 1] =
                                    (dpCurr[o + 1][1][streak + 1] + val) % MOD;
                            } else if (last == 0) {
                                dpCurr[o + 1][1][1] =
                                    (dpCurr[o + 1][1][1] + val) % MOD;
                            }
                        }
                    }
                }
            }
            // Roll arrays: move dpNext into dpCurr for next z
            if (z < zero) {
                dpCurr = dpNext;
                dpNext = new long[one + 1][2][limit + 1];
            }
        }

        long ans = 0;
        for (int last = 0; last < 2; last++) {
            for (int streak = 1; streak <= limit; streak++) {
                ans = (ans + dpCurr[one][last][streak]) % MOD;
            }
        }
        return (int) ans;
    }
}
```
